88 subroutine qpothess2(fix,n,at,xyz,h)
89  use xtb_mctc_accuracy, only : wp
90  use xtb_type_setvar
91  implicit none
92  type(fix_setvar),intent(in) :: fix
93  integer, intent(in) :: n
94  integer, intent(in) :: at(n)
95  real(wp),intent(in) :: xyz(3,n)
96  real(wp),intent(inout) :: h(n*3*(n*3+1)/2)
97 
98  real(wp) :: rij(3),dr,r0
99  real(wp) :: t1,t0,w0,w1,dx,dx2,dy,dy2,r,r2,r3
100  integer :: i,j,k,l,m,ii,jj,iik,iikl,ikjm,ijk,ij
101 
102  if(fix%n.eq.0) return
103 
104  ! calculate hessian
105  do i = 1, fix%n !loop over all atoms
106  ii = (fix%atoms(i)-1)*3
107  do k = 1, 3 !loop diagonal elements (xyz components)
108  do j = 1, fix%n !inner loop for sum over all atoms
109  if (i.ne.j) then
110  rij = xyz(:,fix%atoms(j))-xyz(:,fix%atoms(i))
111  r = norm2(rij)
112  r2 = r*r
113  r3 = r2*r
114  ij = lin(i-1,j-1)+1
115  r0 = fix%val(ij)
116  dr = r-r0
117  dx = xyz(k,fix%atoms(i))-xyz(k,fix%atoms(j))
118  dx2 = dx*dx
119  iik = lin(ii+k,ii+k)
120  h(iik) = h(iik) + 2.0_wp*fix%fc*(1.0_wp+(dx2/r2)-(dx2*dr/r3)-(r0/r))
121  do l = k+1, 3 !loop same-atom block-diagonal elements
122  dy = xyz(l,fix%atoms(i))-xyz(l,fix%atoms(j))
123  iikl = lin(ii+k,ii+l)
124  h(iikl) = h(iikl) + 2.0_wp*fix%fc*r0*dx*dy/r3
125  enddo !end loop same-atom block-diagonal elements
126  endif
127  enddo !end inner loop for sum over all atoms
128  do j = i+1, fix%n !loop over the rest (mixed atoms)
129  rij = xyz(:,fix%atoms(j))-xyz(:,fix%atoms(i))
130  r = norm2(rij)
131  r2 = r*r
132  r3 = r2*r
133  ij = lin(i-1,j-1)+1
134  r0 = fix%val(ij)
135  dr = r-r0
136  jj = (fix%atoms(j)-1)*3
137  do m = 1, 3
138  if (k.eq.m) then !same component case
139  dx = xyz(k,fix%atoms(i))-xyz(k,fix%atoms(j))
140  dx2 = dx*dx
141  ijk = lin(ii+k,jj+k)
142  h(ijk) = h(ijk) - 2.0_wp*fix%fc*(1.0_wp+dx2/r2-dx2*dr/r3-r0/r)
143  else !different component case
144  dx = xyz(k,fix%atoms(i))-xyz(k,fix%atoms(j))
145  dy = xyz(m,fix%atoms(i))-xyz(m,fix%atoms(j))
146  ikjm = lin(ii+k,jj+m)
147  h(ikjm) = h(ikjm) - 2.0_wp*fix%fc*r0*dx*dy/r3
148  endif
149  enddo
150  enddo
151  enddo !end loop diagonal elements (xyz components)
152  enddo !end loop atoms
153

219 
220  integer i,j,k,l,m,mm
221  real(wp)rij(3),dum,r0,ffc,vp(3),ra(3),rb(3)
222  real(wp)va(3),vb(3),vc(3),vab(3),vcb(3),deda(3),dedc(3),dedb(3)
223  real(wp)dda(3),ddb(3),ddc(3),ddd(3)
224  real(wp)c0,rab2,rcb2,theta,dt,rmul1,rmul2,deddt,rp,cosa,ea,vlen,d

226 
227  if(fix%n.eq.0) return
228 
229  do m = 1, fix%n
230  mm = 3*m-2
231  i = fix%atoms(mm)
232  j = fix%atoms(mm+1)
233  k = fix%atoms(mm+2)
234  c0 = fix%val(m)
235  va = xyz(1:3,i)
236  vb = xyz(1:3,j)
237  vc = xyz(1:3,k)
238  vab = va-vb
239  vcb = vc-vb
240  rab2 = vab(1)*vab(1) + vab(2)*vab(2) + vab(3)*vab(3)
241  rcb2 = vcb(1)*vcb(1) + vcb(2)*vcb(2) + vcb(3)*vcb(3)
242  call crprod(vcb,vab,vp)
243  rp = norm2(vp)+1.d-14
244  call impsc(vab,vcb,cosa)
245  cosa = dble(min(1.0d0,max(-1.0d0,cosa)))
246  theta= dacos(cosa)
247  dt = theta - c0
248  ea = fix%fc* dt**2
249  deddt = 2.d0 * fix%fc * dt
250  e = e + ea
251  call crprod(vab,vp,deda)
252  rmul1 = -deddt / (rab2*rp)
253  deda = deda*rmul1
254  call crprod(vcb,vp,dedc)
255  rmul2 = deddt / (rcb2*rp)
256  dedc = dedc*rmul2
257  dedb = deda+dedc
258  g(:,i) = g(:,i) + deda
259  g(:,j) = g(:,j) - dedb
260  g(:,k) = g(:,k) + dedc
261  enddo
262 
263 end subroutine constrain_angle
