314  end if
315 
316  cutoff2 = self%cutoff**2
317 
318  ntrans = 0
319  do itr1 = self%ranges(1, 1), self%ranges(2, 1)
320  do itr2 = self%ranges(1, 2), self%ranges(2, 2)
321  do itr3 = self%ranges(1, 3), self%ranges(2, 3)
322  if (self%excludeInversion) then
323  if (itr1 < 0) cycle
324  if (itr2 < 0 .and. itr1 == 0) cycle
325  if (itr3 < 0 .and. itr2 == 0 .and. itr1 == 0) cycle
326  end if
327  point(:) = self%lattice(:, 1) * itr1 &
328  & + self%lattice(:, 2) * itr2 &
329  & + self%lattice(:, 3) * itr3
330  r2 = sum(point**2)
331  if (r2 <= cutoff2) then
332  ntrans = ntrans + 1
333  self%dist2(ntrans) = r2
334  self%trans(:, ntrans) = [itr1, itr2, itr3]
335  end if
336  end do
337  end do
338  end do
339 
340  allocate(indx(ntrans))
341  call indexheapsort(indx(:ntrans), self%dist2(:ntrans), tolsamedist2)
342  self%dist2(:ntrans) = self%dist2(indx(:ntrans))
343 
344  self%nTrans = ntrans
345  call move_alloc(self%trans, trans)
346  allocate(self%trans(3, ntrans))
347  do itr = 1, ntrans
348  self%trans(:, itr) = trans(:, indx(itr))
349  end do
