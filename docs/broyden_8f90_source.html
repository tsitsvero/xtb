18 ! Modified Boryden mixer according to
19 ! D.D. Johnson, PRB 38, 12807 (1988)
20 ! Input Variables:
21 ! iter = current iteration
22 ! maxiter = max number of iterations
23 ! n = number of atoms (charges)
24 ! q = current iteration INPUT charges (before diag)
25 ! qlast = last iteration INPUT charges (before diag)
26 ! dqlast = charge difference of prev. iteration (q_in - q_out)
27 ! dq = current iteration charge difference (q_in - q_out)
28 ! df = previous |dF>
29 ! u = previous |u>
30 ! a = a matrix
31 ! Note: The modified Broyden method works on the INPUT charges of the sc
32 ! the difference produced by the SCC step. Especially the INPUT charge o
33 ! iteration is needed, which is usually not stored!
34 module xtb_broyden

37 subroutine broyden(n,q,qlast,dq,dqlast,iter,maxiter,&
38  & alpha,omega,df,u,a)
39  use xtb_mctc_accuracy, only : wp
40  implicit none
41 
42  integer i,j
43  integer n,iter,maxiter,it1
44  real(wp) inv
45 
46  ! charge arrays
47  real(wp) q(n),qlast(n),dq(n),dqlast(n)
48 
49 
50  ! |dF>, |u> and a from prev. iterations
51  real(wp) df(maxiter,n),u(maxiter,n),a(maxiter,maxiter)
52  real(wp) omega(maxiter)
53  ! on the fly variables needed for the mixing
54  real(wp), allocatable :: beta(:,:), c(:), gamma(:)
55  ! real(wp), allocatable :: beta1(:,:)
56 
57  ! mixing parameters
58 
59  real(wp) alpha
60  real(wp) omega0
61  real(wp) minw,maxw,wfac
62  real(wp) tmp(n)
63 
64 
65 
66 
67  it1 = iter - 1
68 
69 
70 
71  ! set parameters
72 
73  ! alpha = 0.25d0
74  omega0 = 0.01d0
75  minw = 1.0d0
76  maxw = 100000.0d0
77  wfac = 0.01d0
78  ! wfac = 0.05d0
79 
80  ! if case for first iteration: simple damping
81  if (iter == 1 ) then
82  dqlast = dq
83  qlast = q
84  q = q + alpha * dq
85  return
86  endif
87 
88  ! allocate
89  allocate(beta(it1,it1),c(it1),gamma(it1))
90  ! allocate(beta1(it1,it1))
91 
92 
93 
94  ! create omega (weight) for the current iteration
95  omega(it1) = sqrt(dot_product(dq,dq))
96 
97  if (omega(it1) .gt. (wfac / maxw)) then
98  omega(it1) = wfac / omega(it1)
99  else
100  omega(it1) = maxw
101  endif
102  if (omega(it1) .lt. minw) then
103  omega(it1) = minw
104  endif
105 
106 
107 
108 
109  ! Build dF(iter-1)
110  df(it1,:)=dq-dqlast
111  inv=1.0d0 / sqrt(dot_product(df(it1,:),df(it1,:)))
112  df(it1,:)=inv*df(it1,:)
113 
114 
115 
116  ! Next: build a, beta, c, gamma
117  do i=1,it1
118  a(i,it1) = dot_product(df(i,:),df(it1,:))
119  a(it1,i) = a(i,it1)
120  c(i) = omega(i) * dot_product(df(i,:),dq)
121  enddo
122 
123 
124 
125  ! Build beta from a and omega
126  do i = 1, it1
127  beta(:it1, i) = omega(:it1) * omega(i) * a(:it1,i)
128  beta(i, i) = beta(i, i) + omega0*omega0
129  enddo
130 
131  ! build beta^-1
132  call matinv(beta,it1)
133 
134  ! build gamma
135  gamma = matmul(c,beta)
136 
137  ! Build |u>
138  u(it1,:) = alpha * df(it1,:) + inv * (q-qlast) !!!
139 
140  ! save charges and deltas
141  dqlast = dq
142  qlast = q
143 
144  ! calculate new charges
145  q = q + alpha * dq
146 
147  do i=1, it1
148  q = q - omega(i) * gamma(i) * u(i,:)
149  enddo
150 
151  deallocate(beta,c,gamma)
152 
153 end subroutine broyden
