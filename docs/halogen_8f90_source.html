48  real(wp) :: dtermlj,termlj,prefactor,numerator,denominator,rbx
49  alp=6.0_wp
50  lj2=0.50_wp*a
51 
52  exb = 0.0_wp
53  if(nxb.lt.1) return
54 
55  ! B-X...A
56  do k=1,nxb
57  x =xblist(1,k)
58  aa=xblist(2,k)
59  b=xblist(3,k)
60  ati=at(x)
61  atj=at(aa)
62  cc=haldata%bondStrength(ati)
63  r0ax=haldata%radScale*(haldata%atomicRad(ati)+haldata%atomicRad(atj))
64  dxa=xyz(:,aa)-xyz(:,x) ! acceptor - halogen
65  dxb=xyz(:, b)-xyz(:,x) ! neighbor - halogen
66  dba=xyz(:,aa)-xyz(:, b) ! acceptor - neighbor
67  d2ax=sum(dxa*dxa)
68  d2bx=sum(dxb*dxb)
69  d2ab=sum(dba*dba)
70  rax=sqrt(d2ax)
71  ! angle part. term = cos angle B-X-A
72  xy = sqrt(d2bx*d2ax)
73  term = (d2bx+d2ax-d2ab) / xy
74  aterm = (0.5_wp-0.25_wp*term)**alp
75  t13 = r0ax/rax
76  t14 = t13**a
77  exb = exb + aterm*cc*(t14-haldata%dampingPar*t13**lj2) / (1.0_wp+t14)
78  enddo
79 
80  ! analytic gradient
81  do k=1,nxb
82  x =xblist(1,k)
83  aa=xblist(2,k)
84  b=xblist(3,k)
85  ati=at(x)
86  atj=at(aa)
87  cc=haldata%bondStrength(ati)
88  r0ax=haldata%radScale*(haldata%atomicRad(ati)+haldata%atomicRad(atj))
89 
90  dxa=xyz(:,aa)-xyz(:,x) ! acceptor - halogen
91  dxb=xyz(:, b)-xyz(:,x) ! neighbor - halogen
92  dba=xyz(:,aa)-xyz(:, b) ! acceptor - neighbor
93 
94  d2ax=sum(dxa*dxa)
95  d2bx=sum(dxb*dxb)
96  d2ab=sum(dba*dba)
97  rax=sqrt(d2ax)+1.0e-18_wp
98  rbx=sqrt(d2bx)+1.0e-18_wp
99 
100  xy = sqrt(d2bx*d2ax)
101  term = (d2bx+d2ax-d2ab) / xy
102  ! now compute angular damping function
103  aterm = (0.5_wp-0.25_wp*term)**alp
104 
105  ! set up weighted inverted distance and compute the modified Lennard-Jones potential
106  t14 = (r0ax/rax)**lj2 ! (rov/r)^lj2 ; lj2 = 6 in GFN1
107  numerator = (t14*t14 - haldata%dampingPar*t14)
108  denominator = (1.0_wp + t14*t14)
109  termlj= numerator/denominator
110 
111  ! ----
112  ! LJ derivative
113  ! denominator part
114  dtermlj=2.0_wp*lj2*numerator*t14*t14/(rax*denominator*denominator)
115  ! numerator part
116  dtermlj=dtermlj+lj2*t14*(haldata%dampingPar - 2.0_wp*t14)/(rax*denominator)
117  ! scale w/ angular damping term
118  dtermlj=dtermlj*aterm*cc/rax
119  ! gradient for the acceptor
120  g(:,aa)=g(:,aa)+dtermlj*dxa(:)
121  ! halogen gradient
122  g(:,x)=g(:,x)-dtermlj*dxa(:)
123  ! ----
124  ! cosine term derivative
125  prefactor=-0.250_wp*alp*(0.5_wp-0.25_wp*term)**(alp-1.0_wp)
126  prefactor=prefactor*cc*termlj
127  ! AX part
128  dcosterm=2.0_wp/rbx - term/rax
129  dcosterm=dcosterm*prefactor/rax
130  ! gradient for the acceptor
131  g(:,aa)=g(:,aa)+dcosterm*dxa(:)
132  ! halogen gradient
133  g(:,x)=g(:,x)-dcosterm*dxa(:)
134  ! BX part
135  dcosterm=2.0_wp/rax - term/rbx
136  dcosterm=dcosterm*prefactor/rbx
137  ! gradient for the acceptor
138  g(:,b)=g(:,b)+dcosterm*dxb(:)
139  ! halogen gradient
140  g(:,x)=g(:,x)-dcosterm*dxb(:)
141  ! AB part
142  t13=2.0_wp*prefactor/xy
143  ! acceptor
144  g(:,aa)=g(:,aa)-t13*dba(:)
145  ! neighbor
146  g(:,b)=g(:,b)+t13*dba(:)
147 
148  enddo
