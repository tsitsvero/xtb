140 subroutine generatefilemetainfo(Name, path, basename, extension)
141 
143  character(len=*), intent(in) :: name
144 
146  character(len=:), allocatable, intent(out) :: path
147 
149  character(len=:), allocatable, intent(out) :: basename
150 
152  character(len=:), allocatable, intent(out) :: extension
153 
154  integer :: idot, islash
155 
156  idot = index(name, '.', back=.true.)
157  islash = index(name, '/', back=.true.)
158 
159  ! When given a name like "Gau-3333.Ein", the basename should be "Gau-3333",
160  ! so if there is no '/', we assume there is an implicit './' at the
161  ! beginning of name.
162  if (islash > 0) then
163  path = name(:islash)
164  else
165  path = ''
166  endif
167 
168  if (idot > islash .and. idot > 0) then
169  extension = name(idot+1:)
170  else ! means point is somewhere in the path or absent
171  idot = len(name)+1
172  extension = ''
173  endif
174 
175  if (idot > islash) then
176  basename = name(islash+1:idot-1)
177  else
178  basename = ''
179  endif
180 
181 
182 end subroutine generatefilemetainfo
