19 ! compute localized "normal" modes according to the Pipek-Mezey algo

22 subroutine locmode(n,n3,at,xyz,vthr,freq,rmass,uin,ng,group)
23  use xtb_mctc_accuracy, only : wp
24  use xtb_mctc_blas, only : blas_gemm
25  implicit none
26  integer n,n3,at(n),ng,group(n)
27  real(wp) xyz(3,n),freq(n3),rmass(n3),uin(n3,n3),vthr
28 
29  integer nvar,nvsq,r,s,a,ii,ij,ija,k,i,j,highest,ga,n36
30  real(wp) dum1,dum2,dd,vtyp(3),tors(n3)
31  real(wp),allocatable:: ul(:,:), fl(:), pop(:,:), rl(:), pp(:)
32  real(wp),allocatable:: d(:,:),u(:,:),bmat(:,:),rr(:), fl2(:)
33  integer,allocatable::ind(:),ind2(:)
34 
35  write(*,*)
36  write(*,*)'Pipek-Mezey localization of normal modes'
37  do i=1,n3
38  if(freq(i).lt.vthr) highest=i
39  enddo
40 
41  nvar=highest-6
42  write(*,*) 'number of localized modes ',nvar
43  write(*,*) 'frequency thrshold ',vthr
44  write(*,*) 'number of groups ',ng
45  write(*,*) 'atom-group assingment : '
46  write(*,'(30i3)') group
47 
48  nvsq=nvar*(nvar+1)/2
49  allocate ( pop(nvsq,ng),ul(n3,n3),u(n3,nvar),&
50  & fl(nvar),rl(nvar),rr(n3),fl2(n3),pp(ng),&
51  & d(nvar,nvar),ind(nvar),ind2(n3) )
52 
53  ! skip tr/rot
54  do i=7,6+nvar
55  u(1:n3,i-6)=uin(1:n3,i)
56  enddo
57 
58  pop = 0
59  ! do the popmat
60  k=0
61  do r=1,nvar
62  do s=1,r-1
63  k=k+1
64  pp=0
65  do a=1, n
66  ga=group(a)
67  do ij=1,3
68  ija=ij+(a-1)*3
69  pp(ga)=pp(ga)+u(ija,r)*u(ija,s)
70  enddo
71  enddo
72  pop(k,1:ng)=pp(1:ng)*2.0d0
73  enddo
74  k=k+1
75  do a=1, n
76  ga=group(a)
77  do ij=1,3
78  ija=ij+(a-1)*3
79  pop(k,ga)=pop(k,ga)+u(ija,r)*u(ija,r)
80  enddo
81  enddo
82  enddo
83  write(*,*) 'pop matrix done.'
84  ! do the trafo
85  call lopt(.true.,nvar,ng,1.d-6,pop,d)
86  ! new freq, rmass
87  do i=1,nvar
88  dum1=0.0d0
89  dum2=0.0d0
90  do k=1,nvar
91  dd=d(k,i)*d(k,i)
92  dum1=dum1+freq(k+6)*dd
93  dum2=dum2+rmass(k+6)*dd
94  enddo
95  fl(i)=dum1
96  rl(i)=dum2
97  enddo
98  ! the local modes in ul
99  CALL blas_gemm('N','N',n3,nvar,nvar,1.d0,u,n3,d,nvar,0.d0,ul,n3)
100 
101  ! sort
102  do i=1,nvar
103  ind(i)=i
104  enddo
105  call qsort(fl, 1, nvar, ind)
106 
107  do i=1,nvar
108  freq(i+6)=fl( i )
109  rmass(i+6)=rl(ind(i))
110  uin(1:n3,i+6)=ul(1:n3,ind(i))
111  enddo
112 
113 end subroutine locmode
