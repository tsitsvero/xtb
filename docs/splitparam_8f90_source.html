69  allocate(bond(nat,nat))
70 
71 ! determine covalent neighbours
72  bond = 0
73  do i=1,nat
74  do j=1,nat
75  r=sqrt(sum((xyz(:,i)-xyz(:,j))**2))
76  rco=(atomicrad(at(i))+atomicrad(at(j)))*autoaa
