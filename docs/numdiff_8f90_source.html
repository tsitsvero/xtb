91  chk = chk0
92  call calc%singlepoint(env, mol, chk, -1, .true., el, gl, sl, egap, rl)
93  dl = rl%dipole

95  dipgrad(:, ii) = (dr - dl) * step2
96  do jat = 1, mol0%n

99  hessian(jj, ii) = hessian(jj, ii) &
100  & + (gr(jc, jat) - gl(jc, jat)) * step2
101  end do
102  end do
