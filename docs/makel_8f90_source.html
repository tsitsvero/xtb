26 subroutine makel(nao, s, can, clo )
27  use xtb_mctc_lapack, only : lapack_syev
28  use xtb_mctc_blas, only : blas_gemm
29  implicit real*8 (a-h,o-z)
30  dimension s(nao,nao)
31  dimension can(nao,nao)
32  dimension clo(nao,nao)
33  real*8, allocatable ::aux(:),vecs(:,:),e(:),cc(:,:),x(:,:)
34 
35  lwork = 1 + 6*nao + 2*nao**2
36  allocate (vecs(nao,nao),e(nao),aux(lwork),cc(nao,nao))
37  allocate (x(nao,nao))
38 
39  vecs = s
40 
41  call lapack_syev ('V','U',nao,vecs,nao,e,aux,lwork,info)
42 
43  do i=1,nao
44  if(e(i).lt.0) stop 'sorry, must stop in S^1/2!'
45  e(i)=dsqrt(e(i))
46  enddo
47 
48  do m=1,nao
49  do i=1,nao
50  x(i,m)= vecs(i,m)
51  cc(i,m)=e(m)*vecs(i,m)
52  enddo
53  enddo
54 
55  call blas_gemm('N','T',nao,nao,nao,1.0d0,x, nao,cc,nao,0.0d0,vecs,nao)
56 
57  x = vecs
58  deallocate(e,aux,cc,vecs)
59 
60  moci=nao
61  call blas_gemm('n','n',nao,moci,nao,1.d0,x,nao,can,nao,0.d0,clo,nao)
62 
63  deallocate(x)
64  return
65 end subroutine makel
