168  do i = 1, nno
169  ia = nnlists(i,iat)
170  ! compute the distance to the atom
171  tj(:)=xyzp(:)-xyza(:,ia)
172  tj2=tj(1)*tj(1)+tj(2)*tj(2)+tj(3)*tj(3)
173  ! if within the outer cut-off compute
174  if(tj2.lt.trj2(2,ia)) then
175  if(tj2.le.trj2(1,ia)) then
176  sasap=0.0_wp
177  return
178  else
179  sqtj = sqrt(tj2)
180  uj = sqtj - vdwsa(ia)
181  ah3uj2 = ah3*uj*uj
182  dsasaij = ah1+3.0_wp*ah3uj2
183  sasaij = ah0+(ah1+ah3uj2)*uj
184 
185  ! accumulate the molecular surface
186  sasap = sasap*sasaij
187  ! compute the gradient wrt the neighbor
188  dsasaij = dsasaij/(sasaij*sqtj)
189  nni=nni+1
190  grdi(nni)=ia
191  grds(:,nni) = dsasaij*tj(:)
192  endif
193  ! check if the point is already buried
194 ! if(sasap.lt.tolsesp) return
195  endif
196  enddo
