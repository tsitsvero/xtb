30 subroutine pseudodiag(n,nocc,fmo,eig)
31  use xtb_mctc_blas, only : mctc_rot
32 !$ use omp_lib
33  implicit none
34  !Dummy Arguments
35  integer, intent(in) :: n
36  integer, intent(in) :: nocc
37  real*8, intent(inout):: fmo(n,n)
38  real*8, intent(out) :: eig(n)
39  !Stack Variables
40  integer:: i
41  integer:: j
42  integer:: m
43  integer:: aux_occ,aux_virt, loop_count
44  integer:: blocksize, batch, dest, nproc
45  integer:: i_virt
46  integer:: nvirt
47  integer:: i_occ
48  integer :: bound
49  real*8:: a
50  real*8:: b
51  real*8:: c
52  real*8:: d
53  real*8:: e
54  !real*8:: alpha
55  !real*8:: beta
56  !Heap Variables and pointer targets
57  integer, allocatable, save :: indizes(:,:)
58  real*4, allocatable :: vector(:,:) ! R4 saves 30 % due to less cache misses for 1000 atoms
59  !real*8, allocatable, target :: occ(:,:)
60  !real*8, allocatable, target :: virt(:,:)
61  real*4,allocatable :: alphaarr(:,:)
62  real*4,allocatable :: betaarr(:,:)
63 
64 
65  allocate(vector(n,n))
66  vector = 0
67  nvirt = n - nocc
68 
69  allocate (alphaarr(nocc,nvirt))
70  allocate (betaarr(nocc,nvirt))
71  alphaarr = 0.0d0
72  betaarr = 0.0d0
73 
74  do i=1,n
75  vector(i,i)=1
76  enddo
77  do i=1,n
78  eig(i) = fmo(i,i)
79  enddo
80 
81  !$omp parallel SHARED(alphaarr, betaarr, fmo) PRIVATE(a,c,d,e, i_occ, i_virt)
82  !$omp do
83  do i_occ = 1, nocc
84  a=fmo(i_occ, i_occ)
85  !DIR$ IVDEP
86  do i_virt = 1, nvirt
87  c=fmo(i_occ,nocc + i_virt)
88  d=a-fmo(nocc + i_virt,nocc + i_virt)
89  !if(abs(c/d).lt.1.d-6) cycle
90  e=sign(sqrt(4.d0*c*c+d*d),d)
91  alphaarr(i_occ,i_virt) = sqrt(0.5d0*(1.d0+d/e))
92  betaarr(i_occ,i_virt) = -sign(sqrt(1.d0-alphaarr(i_occ,i_virt)*alphaarr(i_occ,i_virt)),c)
93  end do ! End Loop over i_virt from 1 to nvirt
94  end do ! End Loop over i_occ from 1 to nocc
95  !$OMP END DO
96  !$OMP END PARALLEL
97 
98 
99  !blocksize = 28
100  nproc = 1
101  !$ nproc = omp_get_max_threads()
102  !blocksize = 30 * nproc !for cluster
103  blocksize = 6 * nproc !for Desktop
104 
105  !Batched Loop
106  do batch = 0, nvirt/blocksize-1
107  do aux_occ = 1, nocc
108  !$omp parallel SHARED(alphaarr, betaarr, vector, aux_occ, n, nocc), &
109  !$OMP& PRIVATE(i_occ, i_virt, aux_virt)
110  !$omp do
111 
112  !DIR$ IVDEP
113  do aux_virt = 1, blocksize
114  i_virt = batch*blocksize + aux_virt
115  i_occ = i_virt+aux_occ-1 - (((i_virt-1+aux_occ)/(nocc+1))*nocc)
116  ! rotation of pseudo-eigenvectors
117  call mctc_rot(vector(:,i_occ), vector(:,nocc+i_virt), alphaarr(i_occ,i_virt), betaarr(i_occ,i_virt) )
118  end do ! End Loop over j from 1 to nocc
119  !$OMP END DO
120  !$OMP END PARALLEL
121  end do ! End Loop over i from 1 to nvirt
122  end do ! End Loop over batch from 1 to nvirt / blocksize + 1
123 
124  !Residual Loop
125  do aux_occ = 1, nocc
126  !$omp parallel SHARED(alphaarr, betaarr, vector, aux_occ, n, nocc), &
127  !$OMP& PRIVATE(i_occ, i_virt)
128  !$omp do
129 
130  !DIR$ IVDEP
131  do aux_virt = 1, mod(nvirt,blocksize)
132  i_virt = batch*blocksize + aux_virt
133  i_occ = i_virt+aux_occ-1 - (((i_virt-1+aux_occ)/(nocc+1))*nocc)
134  !if ((alphaarr(i_occ,i_virt)==0.0) .and. (betaarr(i_occ,i_virt)==0.0)) cycle
135  call mctc_rot(vector(:,i_occ), vector(:,nocc+i_virt), alphaarr(i_occ,i_virt), betaarr(i_occ,i_virt))
136  end do ! End Loop over j from 1 to nocc
137  !$OMP END DO
138  !$OMP END PARALLEL
139  end do ! End Loop over i from 1 to nvirt
140 
141  fmo = vector
142  deallocate(vector, alphaarr, betaarr)
143 end
