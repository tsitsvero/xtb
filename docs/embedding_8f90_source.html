23 !! ========================================================================
24 ! Embedding Stuff -- for QM/MM and subsystem-DFTB
25 !! ------------------------------------------------------------------------
26 ! in our tight binding model we can do much more than just point charge
27 ! embedding. Since we have a multipole electrostatic we can also use
28 ! these multipoles (dipole and quadrupoles). Also there is the dispersion
29 ! potential so we can embed the system in polarizibilities (coordination).
30 ! Given this fact the pcharges input will be more complicated
31 ! 1> <npc>
32 ! 2> <q> <x> <y> <z> [at [dx dy dz [qxx qyy qzz qxy qxz qxy [CN]]]]
33 ! ...
34 ! where q is the partial charge and x,y,z are it respective coordinates
35 ! at the moment the atom type is optional and by default nitrogen (7).
36 !! ------------------------------------------------------------------------
37 ! by this embedding scheme we can get in a subsystem-DFTB scheme
38 ! gradients for:
39 ! - monopole electrostatic
40 ! - anisotropic electrostatic
41 ! - dispersion
42 ! still missing is
43 ! - polynomial wavefunction terms
44 ! - repulsion terms
45 ! - shellwise electrostatic
46 ! - CN-shift of H0 (maybe via divide and conquere with P)
47 ! - nonadditive dispersion (ATM)
48 ! GBSA might need some special attention
49 !! ========================================================================

66 subroutine read_pcem(iunit,env,pcem,jData)
67  use xtb_mctc_strings
68  use xtb_mctc_convert
69  use xtb_type_pcem
70  use xtb_type_environment, only : tenvironment
71  use xtb_setparam
72  use xtb_readin, only : getline => strip_line
73  implicit none
74  integer, intent(in) :: iunit
75  type(tenvironment), intent(inout) :: env
76  type(tb_pcem),intent(inout) :: pcem
77  type(tcoulombdata), intent(in) :: jData
78 
79  integer :: narg
80  character(len=p_str_length),dimension(p_arg_length) :: argv
81  character(len=:),allocatable :: line
82  integer :: err
83  integer :: npc,n
84  integer :: i,iat
85  real(wp) :: xyz(3),q,gami
86  real(wp) :: conv
87 
88  conv = 1.0_wp ! input is usually in *Bohr*
89  if (pcem_orca) then
90  conv = aatoau ! except when we try to read ORCA's pc-files
91  endif
92 
93  ! file is not open and error was not catched from caller
94  if (iunit.eq.-1) return
95 
96  ! reset file position
97  rewind(iunit)
98 
99  ! load first line to buffer and attempt to parse
100  call getline(iunit,line,err)
101  read(line,*,iostat=err) npc
102  if (err.ne.0) then
103  return
104  endif
105 
106  call pcem%allocate(npc)
107 
108  ! we now the number of lines that should follow, but we don't care
109  ! -> read until EOF and count lines
110  n = 0
111  do
112 
113  ! load a new line into buffer
114  ! exit if EOF occurs
115  call getline(iunit,line,err)
116  if (err.ne.0) exit
117 
118  ! we skip empty lines (might been commented out)
119  if (len(line).eq.0) cycle
120 
121  ! try to parse the lines into argv, this has a hard limit on length and items
122  ! let's assume a well-behaving user for now
123  call parse(line,' ',argv,narg)
124 
125  ! user provided crap input, bad user
126  if (narg.lt.4) then
127  call raise('E',"Not enough entries for PC, please check!")
128  endif
129 
130  ! we count this line
131  n = n+1
132  ! before we start parsing we check the number of lines
133  if (n > npc) then ! bad user
134  call raise('E',"Wrong dimension input for PC, too many lines provided")
135  endif
136 
137  ! first position should be the partial charge
138  read(argv(1),*,iostat=err) q
139  ! get the position from the next three entries
140  do i = 1, 3
141  read(argv(i+1),*,iostat=err) xyz(i)
142  enddo
143 
144  ! now our well-behaving user has three options:
145  ! -> provide no fifth argument -> fallback to default from $pcem group
146  ! -> provide a atom number -> use gfn_method gam value (aoparam)
147  ! -> provide a gam value -> double check this one!!!
148  if (narg.eq.5) then
149  read(argv(5),*,iostat=err) gami
150  if (err.ne.0) then
151  call elem(argv(5),iat)
152  if (iat.eq.0) then ! so much for the well-behaved user
153  call raise('E',"Invalid PC input: '"//trim(argv(5))//"'")
154  endif
155  gami = jdata%chemicalHardness(iat)
156  endif
157  ! GFN0-xTB has negative gam-values since they are internally combined
158  ! with the atom radius for the full chemical hardness (overall > 0).
159  ! This is certainly a problem here, but since GFN0-xTB does
160  ! not provide the possibility to use a PC potential, we don't care...
161  ! But if YOU are reading this comment and are in charge of implementing
162  ! a PC potential for GFN0-xTB, than this is YOUR problem now.
163  if (gami < 0.0_wp) then
164  call raise('S',"Found negative gam-value in user input: '"//&
165  & trim(argv(5))//"'")
166  endif
167  else
168  ! we trust the dummy atom from xcontrol, since xcontrol could already
169  ! have handled potential errors, which is usually not the case...
170  gami = jdata%chemicalHardness(pcem_dummyatom)
171  endif
172  pcem%xyz(:,n) = xyz*conv
173  pcem%q(n) = q
174  pcem%gam(n) = gami
175  enddo
176 
177  ! user screwed up, we could rescue this by resetting npc
178  ! or we rub it in his/her face :)
179  if (n /= npc) then
180  call raise('E',"Wrong dimension input for PC, too few lines provided")
181  endif
182 
183 end subroutine read_pcem

185 !! ========================================================================
186 ! J potentials for GFN1 including the point charge stuff
187 !! ========================================================================
188 subroutine jpot_pcem_gfn1(jData,n,pcem,nshell,at,xyz,alphaj,Vpc)
189  use xtb_type_pcem
190  implicit none
191  type(tcoulombdata), intent(in) :: jData
192  integer, intent(in) :: n
193  type(tb_pcem),intent(inout) :: pcem
194  integer, intent(in) :: nshell(:)
195  integer, intent(in) :: at(n)
196  real(wp),intent(in) :: xyz(3,n)
197  real(wp),intent(in) :: alphaj
198  real(wp),intent(inout) :: Vpc(:)
199  integer :: ish,ii,iat,ati,kk
200  real(wp) :: eh1,dum,gi,gj,xj,rab
201 
202  ii = 0
203  do iat = 1, n
204  ati = at(iat)
205  do ish = 1, nshell(ati)
206  gi = jdata%shellHardness(ish,ati)
207  eh1 = 0.0_wp
208  do kk = 1, pcem%n
209  gj = pcem%gam(kk)
210  rab = sqrt(sum((pcem%xyz(:,kk) - xyz(:,iat))**2))
211  xj = 2.0_wp/(1./gi+1./gj)
212  dum = 1.0_wp/(rab**alphaj+1._wp/xj**alphaj)**(1._wp/alphaj)
213  eh1 = eh1+pcem%q(kk)*dum
214  end do
215  vpc(ii+ish) = eh1
216  end do
217  ii = ii + nshell(ati)
218  end do
219 
220 end subroutine jpot_pcem_gfn1

225 subroutine jpot_pcem_gfn2(jData,n,pcem,nshell,at,xyz,Vpc)
226  use xtb_type_pcem
227  implicit none
228  type(tcoulombdata), intent(in) :: jData
229  integer, intent(in) :: n
230  type(tb_pcem),intent(in) :: pcem
231  integer, intent(in) :: nshell(:)
232  integer, intent(in) :: at(n)
233  real(wp),intent(in) :: xyz(3,n)
234  real(wp),intent(inout) :: Vpc(:)
235  integer :: ish,ii,iat,ati,kk
236  real(wp) :: eh1,dum,gi,gj,xj,r2
237 
238  ii = 0
239  do iat = 1, n
240  ati = at(iat)
241  do ish = 1, nshell(ati)
242  gi = jdata%shellHardness(ish,ati)
243  eh1 = 0.0_wp
244  do kk = 1, pcem%n
245  gj = pcem%gam(kk)
246  r2 = sum((pcem%xyz(:,kk) - xyz(:,iat))**2)
247  xj = 0.5_wp*(gi+gj)
248  dum = 1.0_wp/sqrt(r2+1._wp/xj**2)
249  eh1 = eh1+pcem%q(kk)*dum
250  enddo
251  vpc(ii+ish) = eh1
252  enddo
253  ii = ii + nshell(ati)
254  enddo
255 
256 end subroutine jpot_pcem_gfn2

284 !! ========================================================================
285 subroutine pcem_grad_gfn1(jData,g,gpc,n,pcem,at,nshell,xyz,alphaj,qsh)
286  use xtb_type_pcem
287  implicit none
288  type(tcoulombdata), intent(in) :: jData
289  integer, intent(in) :: n
290  type(tb_pcem),intent(in) :: pcem
291  real(wp),intent(inout) :: g(3,n)
292  real(wp),intent(inout) :: gpc(3,pcem%n)
293  integer, intent(in) :: at(n)
294  integer, intent(in) :: nshell(:)
295  real(wp),intent(in) :: xyz(3,n)
296  real(wp),intent(in) :: alphaj
297  real(wp),intent(in) :: qsh(:)
298 
299  integer :: ish,ii,j,iat,ati
300  real(wp) :: dr(3)
301  real(wp) :: gi,gj,r2,rr,yy,ff
302 
303  ii = 0
304  do iat = 1, n
305  ati = at(iat)
306  do ish = 1, nshell(ati)
307  gi = jdata%shellHardness(ish,ati)
308  do j = 1, pcem%n
309  dr = xyz(:,iat) - pcem%xyz(:,j)
310  r2 = sum(dr**2)
311  gj = pcem%gam(j)
312  rr = 2.0_wp/(1.0_wp/gi+1.0_wp/gj)
313  rr = 1.0_wp/rr**alphaj
314  ff = r2**(alphaj/2.0_wp-1.0_wp)*&
315  &(r2**(alphaj*0.5_wp)+rr)**(-1.0_wp/alphaj-1.0_wp)
316  yy = ff*qsh(ii+ish)*pcem%q(j)
317  g(:,iat)=g(:,iat)-dr*yy
318  gpc(:,j)=gpc(:,j)+dr*yy
319  end do
320  end do
321  ii = ii + nshell(ati)
322  end do
323 
324 end subroutine pcem_grad_gfn1
