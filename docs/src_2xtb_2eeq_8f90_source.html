312  do ish = 1, coulomb%itbl(2, iat)
313  tmp = self%kcn(ish, iid) / (sqrt(cn(iat)) + 1.0e-14_wp)
314  xvec(ii+ish) = -self%chi(ish, iid) + tmp*cn(iat)
315  dxdcn(ii+ish) = 0.5_wp*tmp
316  jmat(ii+ish, ii+ish) = jmat(ii+ish, ii+ish) + self%gam(ish, iid)

387  do ish = 1, coulomb%itbl(2, iat)
388  djdr(:, iat, ii+ish) = djdr(:, iat, ii+ish) + djdtr(:, ii+ish)
389  end do
390  end do

487 subroutine invert_sytri(env, mat)
488 
490  character(len=*), parameter :: source = 'xtb_eeq_sytri'
491 
493  type(tenvironment), intent(inout) :: env
494 
495  real(wp), intent(inout) :: mat(:, :)
496 
497  integer :: ii, jj
498  integer :: lwork, ndim, info
499  real(wp) :: test(1)
500  real(wp), allocatable :: work(:)
501  integer, allocatable :: ipiv(:)
502 
503  ndim = size(mat, dim=1)
504  if (size(mat, dim=2) < ndim) then
505  call env%error("A not so carefully crafted algorithm did mess up", source)
506  return
507  end if
508 
509  allocate(ipiv(ndim))
510 
511  ! assume work space query, set best value to test after first dsytrf call
512  call lapack_sytrf('L', ndim, mat, ndim, ipiv, test, -1, info)
513  lwork = int(test(1))
514  allocate(work(lwork))
515 
516  ! Bunch-Kaufman factorization A = L*D*L**T
517  call lapack_sytrf('L', ndim, mat, ndim, ipiv, work, lwork, info)
518  if(info > 0)then
519  call env%error('Bunch-Kaufman factorization failed', source)
520  return
521  endif
522 
523  ! A⁻¹ from factorized L matrix, save lower part of A⁻¹ in matrix
524  ! matrix is overwritten with lower triangular part of A⁻¹
525  call lapack_sytri('L', ndim, mat, ndim, ipiv, work, info)
526  if (info > 0) then
527  call env%error('Bunch-Kaufman inversion failed', source)
528  return
529  endif
530 
531  ! symmetrizes A⁻¹ matrix from lower triangular part of inverse matrix
532  do ii = 1, ndim
533  do jj = ii+1, ndim
534  mat(ii,jj) = mat(jj,ii)
535  enddo
536  enddo
537 
538 end subroutine invert_sytri
