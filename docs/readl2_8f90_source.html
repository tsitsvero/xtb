18 !cuts the at blanks and tabstops and returns all floats and strings in order of occurence
19 subroutine cutline(line,floats,strings)
20  use xtb_mctc_accuracy, only : wp
21  implicit none
22  real(wp) floats(*),num
23  character(len=128) line,str,stmp
24  character(len=80) strings(3)
25  character(len=1) digit
26  integer i,ty,cs,cf
27 
28  stmp=''
29  cs=1
30  cf=1
31  strings(:)=''
32  do i=1,len(trim(line))
33  digit=line(i:i)
34  if(digit.ne.' '.and.digit.ne.char(9)) then !should exclude tabstops and blanks, 9 is ascii code for tab
35  stmp=trim(stmp)//trim(digit)
36  elseif(stmp.ne.'')then
37  call checktype(stmp,num,str,ty) !get type of string, 0=number, 1=character
38  if(ty.eq.0) then
39  floats(cf)=num
40  cf=cf+1
41  elseif(ty.eq.1) then
42  strings(cs)=trim(str)
43  cs=cs+1
44  else
45  write(*,*)'Problem in checktype, must abort'
46  exit
47  endif
48  stmp=''
49  endif
50  if(i.eq.len(trim(line))) then !special case: end of line
51  call checktype(stmp,num,str,ty)
52  if(ty.eq.0) then
53  floats(cf)=num
54  cf=cf+1
55  elseif(ty.eq.1) then
56  strings(cs)=trim(str)
57  cs=cs+1
58  else
59  write(*,*)'Problem in checktype, must abort'
60  exit
61  endif
62  stmp=''
63  endif
64  enddo
65 end subroutine cutline

68 !this checks the type of the string and returns it cast to real or as string.
69 subroutine checktype(field,num,str,ty)
70  use xtb_mctc_accuracy, only : wp
71  implicit none
72  character(len=*) field,str
73  real(wp) num
74  integer i,e,ty
75  logical is_num
76 
77  ty=99
78  str=''
79  is_num=.false.
80  read(field,'(F10.5)',iostat=e)num !cast string on real and get error code; 0 means success.
81  if(e.eq.0)is_num=.true.
82  if(is_num)then
83  if(index(field,'.').ne.0) then !check for integer/real
84  read(field,'(F30.16)')num
85  ty=0
86  else !if integer, add .0 to string; otherwise cast to real does not work
87  str=trim(field)//'.0'
88  read(str,'(F30.16)')num
89  str=''
90  ty=0
91  endif
92  else
93  str=field
94  ty=1
95  endif
96 end subroutine checktype
