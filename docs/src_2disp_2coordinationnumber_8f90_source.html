252  !$omp parallel do default(none) private(den) shared(enscale, rcov, en)&
253  !$omp reduction(+:cn, dcndr, dcndL) shared(mol, kcn, neighlist, neighs) &
254  !$omp private(ij, img, jat, ati, atj, r2, rij, r1, rc, countf, countd, stress)
255  do iat = 1, len(mol)
256  ati = mol%at(iat)
257  do ij = 1, neighs(iat)
258  img = neighlist%iNeigh(ij, iat)
259  r2 = neighlist%dist2(ij, iat)
260  rij = neighlist%coords(:, iat) - neighlist%coords(:, img)
261  jat = neighlist%image(img)
262  atj = mol%at(jat)
263  r1 = sqrt(r2)
264 
265  rc = rcov(ati) + rcov(atj)
266 
267  if (enscale) then
268  den = k4*exp(-(abs(en(ati)-en(atj)) + k5)**2/k6)
269  else
270  den = 1.0_wp
271  endif
272 
273  countf = den * cfunc(kcn, r1, rc)
274  countd = den * dfunc(kcn, r1, rc) * rij/r1
275 
276  cn(iat) = cn(iat) + countf
277  if (iat /= jat) then
278  cn(jat) = cn(jat) + countf
279  endif
280 
281  dcndr(:, iat, iat) = dcndr(:, iat, iat) + countd
282  dcndr(:, jat, jat) = dcndr(:, jat, jat) - countd
283  dcndr(:, iat, jat) = dcndr(:, iat, jat) + countd
284  dcndr(:, jat, iat) = dcndr(:, jat, iat) - countd
285 
286  stress = spread(countd, 1, 3) * spread(rij, 2, 3)
287 
288  dcndl(:, :, iat) = dcndl(:, :, iat) + stress
289  if (iat /= jat) then
290  dcndl(:, :, jat) = dcndl(:, :, jat) + stress
291  endif
292 
293  enddo
294  enddo
295  !$omp end parallel do
