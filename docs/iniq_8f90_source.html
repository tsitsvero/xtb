32 !! ========================================================================
33 ! driver for calculation of Grimme-Gasteiger charges, which come in
34 ! three flavors: GG(vTB), GG(GFN1) and GG(GFN2). They differ in the way
35 ! the coordination number is taken into account.
36 ! INPUT
37 ! nat :: number of atoms
38 ! nel :: number of electrons
39 ! at :: ordinal number of atoms
40 ! xyz :: coordinates in Bohr
41 ! z :: nuclear charges
42 ! chrg :: system charge
43 ! OUTPUT
44 ! q :: partial charges
45 ! cn :: coordination number
46 ! PARAMETER
47 ! kchrg1 :: for GFN1-type Grimme-Gasteiger charges
48 ! OPTIONS
49 ! version :: vTB(=0),GFN1(=1),GFN2(>1)
50 ! verbose :: prints some header and makes a bit of analysis in the end

97  if (pr) then

100  dipole(1)=dipole(1)+xyz(1,i)*q(i)
101  dipole(2)=dipole(2)+xyz(2,i)*q(i)
102  dipole(3)=dipole(3)+xyz(3,i)*q(i)

105  write(id,'(1x,''point charge moment (au)'')')
106  write(id,'(5x,''X'',7x,''Y'',7x,''Z'')')
107  write(id,'(3f9.4,'' total (Debye): '',f8.3)') &
108  & dipole(1), dipole(2), dipole(3), &
109  & sqrt(sum(dipole**2))*2.5418_wp
110  endif

320 !! ========================================================================
321 ! non-iterative, distance dependent Gasteiger-Grimme type charges
322 ! INPUT
323 ! nat :: number of atoms
324 ! at :: ordinal number of atoms
325 ! xyz :: coordinates in Bohr
326 ! en :: systemspecific EN
327 ! OUTPUT
328 ! q :: EN based population
329 pure subroutine gasteiger_partition(nat,at,xyz,en,q)
330 
331 ! get data from parameter modules
332  use xtb_mctc_param, only : rcov => covalent_radius_d3
333 
334  implicit none
335 
336  integer, intent(in) :: nat
337  integer, intent(in) :: at(nat)
338  real(wp),intent(in) :: xyz(3,nat)
339  real(wp),intent(in) :: en(nat)
340  real(wp),intent(inout) :: q(nat)
341 
342  integer :: i,j
343  real(wp) :: tmp,r2,rav,den,rexp
344 
345  do i = 1, nat
346  tmp = 0.0_wp
347  do j = 1, nat
348  if (i.eq.j) cycle
349  r2 = sqrt(sum( (xyz(:,j) - xyz(:,i))**2 ))
350  rav = 0.5_wp * ( rcov(at(i))+rcov(at(j)) )
351  den = en(i) - en(j)
352  rexp = (rav/r2)**6
353  tmp = tmp + den*rexp
354  enddo
355  q(i) = q(i) + tmp
356  enddo
357 
358 end subroutine gasteiger_partition
