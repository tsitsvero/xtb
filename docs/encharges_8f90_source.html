168  nid = maxval(mol%id)

172  do iat = 1, mol%n

180

182  call env%check(exitrun)

185  return
186  end if
187

189  call env%check(exitrun)

192  return
193  end if

196  & energy, gradient, sigma, qat=qat, dqdr=dqdr, dqdl=dqdl)
