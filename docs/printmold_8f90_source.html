64  ! print out atoms & coordinates c
65  !cccccccccccccccccccccccccccccccccccccccccccccccc
66 
67  write(iwfn,'(A)',advance='yes')'[Atoms] AU'
68  ! either coordinates are given in a.u.
69  do i = 1,ncent
70  atyp = tosymbol(at(i))
71  z=at(i)
72  ! atom character, running number, nuclear charge, x,y,z coordin
73  write (iwfn,'(a2,2i6,3E22.14)') &
74  & atyp,i,z,xyz(1,i),xyz(2,i),xyz(3,i)
75  enddo
76  !ccccccccccccccccccccccccccccccccccccccccccccccccccccccccccccc
77  ! Now print basis set data c
78  ! c
79  ! go through atoms and assign orbitals (contracted & prims) c
80  ! search for [GTO] statement to read basis set data c
81  ! for each set of atomic orbitals c
82  !ccccccccccccccccccccccccccccccccccccccccccccccccccccccccccccc
83  write(iwfn,'(A)',advance='yes') '[GTO]'
84  !ccccccccccccccccccc
85  ! go through atoms c
86  !ccccccccccccccccccc
87  icount=1
88  do i=1,ncent
89  write(iwfn,*) i,'0' ! I don't know what the zero is needed for
90  ! now go trough nbfs located on atom i
91  do j=basis%fila(1,i),basis%fila(2,i)
92  call ang2chr(basis%lao(j),aang,skip)
93  if(skip)then
94  do k=1,basis%nprim(j)
95  icount=icount+1
96  enddo
97  else
98  write(iwfn,*) aang,basis%nprim(j),1.00
99  do k=1,basis%nprim(j)
100  write(iwfn,*) basis%alp(icount),basis%cont(icount)
101  icount=icount+1
102  enddo
103  endif
104  enddo
105  write(iwfn,*)
106  enddo
107 
108  nmomax=0
109  do i=1,nmo
110  if(eval(i).gt.thr.and.nmomax.eq.0)nmomax=i-1
111  enddo
112  if(nmomax.eq.0) nmomax=nmo
113 
114  !ccccccccccccccccccccccccccccccccccccccccccccccccccccccccccc
115  ! print occupation number of orbitals and orbital energies c
116  !ccccccccccccccccccccccccccccccccccccccccccccccccccccccccccc
117  write(iwfn,'(A)',advance='yes') '[MO]'
118  ! restricted xtb_printout
119  do i=1,nmomax
120  ! MO info
121  write(iwfn,'(A)',advance='no') 'Sym= '
122  write(iwfn,'(i5,a1)') i,'a'
123  write(iwfn,'(A)',advance='no') 'Ene= '
124  dum=0
125  write(iwfn,*) eval(i)
126  write(iwfn,'(A)',advance='no') 'Spin= '
127  write(iwfn,'(A)',advance='yes') 'Alpha' ! for now just conside
128  write(iwfn,'(A)',advance='no') 'Occup= '
129  write(iwfn,'(F14.8)') occ(i)
130  !now coefficients
131  ! a notion: for l>0, the ordering is p: x,y,z ; d: xx,yy,zz,xy
132  do j=1,nbf
133  write(iwfn,*) j,cmo(j,i)
134  enddo
135  enddo
136  call close_file(iwfn)
