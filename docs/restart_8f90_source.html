32 subroutine readrestart(env,wfx,fname,n,at,gfn_method,success,verbose)
33  character(len=*), parameter :: source = 'restart_readRestart'
34  type(tenvironment), intent(inout) :: env
35  type(twavefunction),intent(inout) :: wfx
36  character(len=*),intent(in) :: fname
37  integer,intent(in) :: n

44  integer :: ich ! file handle

47 
48  success = .false.
49  call open_binary(ich,fname,'r')
50  if (ich.ne.-1) then
51  ! read the first 48 byte, which identify the calculation specs
52  read(ich,iostat=err) iver8,idum8,n8,nshell8,nel8,nopen8
53  if(err.eq.0) then
54  if (iver8.ne.int(gfn_method,i8).and.verbose) &
55  & call env%warning('Version number missmatch in restart file.', source)
56  if (nel8.ne.int(wfx%nel,i8).and.verbose) &
57  & call env%warning('Number of electron missmatch in restart file.', source)
58  if (nopen8.ne.int(wfx%nopen,i8).and.verbose) &
59  & call env%warning('Multiplicity missmatch in restart file.', source)
60  if ((n8.eq.int(wfx%n,i8)).and.(nshell8.eq.int(wfx%nshell,i8))) then
61  success = .true.
62  read(ich) wfx%qsh
63  if (verbose) &
64  write(stdout,'("q/qsh data taken from xtbrestart")')
65  if ((gfn_method.gt.1).and.(iver8.gt.1)) then
66 ! read dipole and qpole CAMM
67  read(ich) wfx%dipm
68  read(ich) wfx%qp
69  if (verbose) &
70  write(stdout,'("CAMM data taken from xtbrestart")')
71  endif
72  else
73  if (verbose) &
74  call env%warning('Dimension missmatch in restart file.', source)
75  success = .false.
76  endif
77  else
78  if (verbose) &
79  call env%warning("Dimension missmatch in restart file.", source)
80  success = .false.
81  end if
82  call close_file(ich)
83  end if
