48 subroutine dhtosq(n,a,b)
49  implicit none
50 ! ---------------------------------------------------------------------
51 ! expand trigonal matrix b to full matrix a
52 ! a and b may refer to the same address
53 ! ---------------------------------------------------------------------
54  integer, intent(in) :: n
55  real(dp),intent(out) :: a(n,n)
56  real(dp),intent(in) :: b(n*(n+1)/2)
57  integer :: i,j,ioff
58 
59  do i=n,1,-1
60  ioff=i*(i-1)/2
61  do j=i,1,-1
62  a(j,i)=b(ioff+j)
63  enddo
64  enddo
65 
66  do i=1,n
67  do j=1,i-1
68  a(i,j)=a(j,i)
69  enddo
70  enddo
71 
72 end subroutine dhtosq

74 subroutine dsyprj(nbdim,m,bmat,n,asym)
75 !----------------------------------------------------------------------
76 ! Purpose:
77 ! Performs projection of a real symmetric matrix asym packed in an
78 ! upper triangular form:
79 ! asym = (1 - bmat*bmat')*asym*(1 - bmat*bmat')
80 ! where (1 - bmat*bmat') is a projector constructed from bmat
81 ! The matrix asym is overwritten on output; bmat is left unchanged.
82 ! Operation count is proportional to n*n*m
83 !
84 ! Input/Output:
85 ! nbdim - First dimension of bmat as declared in the calling routine
86 ! m - Number of columns in the matrix bmat
87 ! bmat - n*m matrix used to build projector
88 ! n - Dimension of the matrix asym
89 ! asym - Symmetric matrix in packed upper triangular form
90 !----------------------------------------------------------------------
91  implicit none
92 
93 ! Input/Output:
94  integer, intent(in) :: nbdim,m,n
95  real(dp),intent(in) :: bmat(nbdim,m)
96  real(dp),intent(inout) :: asym(n*(n+1)/2)
97 
98 ! Local:
99  integer :: i,j,ij
100  real(dp),allocatable :: scrb(:,:)
101  real(dp),allocatable :: scra(:,:)
102 !----------------------------------------------------------------------
103  allocate( scrb(n,m), scra(n,n) )
104 
105 ! Expand trigonal matrix asym to full matrix on scra
106  call htosq(n,scra,asym)
107 
108 ! Calculate scrb = asym*bmat
109  call blas_symm('l','u',n,m,1.0_dp,scra,n,bmat,nbdim,0.0_dp,scrb,n)
110 
111 ! Calculate scra = scrb*bmat'
112  call blas_gemm('n','t',n,n,m,1.0_dp,scrb,n,bmat,nbdim,0.0_dp,scra,n)
113 
114 ! Calculate asym = asym - scra - scra'
115  do i=1,n
116  do j=1,i
117  ij = i*(i-1)/2 + j
118  asym(ij) = asym(ij) - scra(i,j) - scra(j,i)
119  end do
120  end do
121 
122 ! Calculate scrb' = scra'*bmat
123  call blas_gemm('t','n',n,m,n,1.0_dp,scra,n,bmat,nbdim,0.0_dp,scrb,n)
124 
125 ! Calculate scra = bmat*scrb'
126  call blas_gemm('n','t',n,n,m,1.0_dp,bmat,nbdim,scrb,n,0.0_dp,scra,n)
127 
128 ! Calculate asym = asym + scra
129  do i=1,n
130  do j=1,i
131  ij = i*(i-1)/2 + j
132  asym(ij) = asym(ij) + scra(i,j)
133  end do
134  end do
135 
136  deallocate( scra, scrb )
137 
138 end subroutine dsyprj

140 subroutine dblckmgs(m,n,ndim,darray)
141 !----------------------------------------------------------------------
142 ! Purpose:
143 ! Subroutine performs modified Gramm-Schmidt orthonormalization of a
144 ! real matrix. Orthonormalization is done in-place, so the darray is
145 ! overwritten on exit. Linearly dependent vectors are set to zero.
146 !
147 ! Input:
148 ! m - Number of rows in the matrix darray
149 ! n - Number of columns in the matrix darray
150 ! ndim - First array dimension as declared in the calling routine
151 ! darray - Array to be orthonormalized
152 !
153 ! Output:
154 ! darray - Orthonormalized array
155 !----------------------------------------------------------------------
156  implicit none
157 
158 ! Input/Output:
159  integer, intent(in) :: m,n,ndim
160  real(dp), dimension(ndim,n), intent(out) :: darray
161 !----------------------------------------------------------------------
162 ! Local:
163  integer :: ii,jj,kk,ll,ibsize,nblcks,istrt,jstrt,iend,ncol,ierr
164  real(dp) :: tmp
165  real(dp), dimension(:,:), allocatable :: smat
166 !----------------------------------------------------------------------
167 ! Local parameters
168  real(dp) thr
169 ! use BLAS directly without MCTC interfaces, for dirty hacks
170  external dgemm ! I hope you know what you are doing...
171 !----------------------------------------------------------------------
172 
173  thr = epsilon(1.0_dp) !Threshold for zero vectors
174 !----------------------------------------------------------------------
175 ! Block size optimized for Athlon 1200 MHz with 2.0GB memory for
176 ! matrices up to 5000x5000
177  ibsize = 60
178 !----------------------------------------------------------------------
179 
180 !-----------------------------------------------------------------
181 ! Allocate overlap matrix
182 !-----------------------------------------------------------------
183  allocate(smat(ibsize,ibsize),stat=ierr)
184  if(ierr /= 0) call raise('E','Memory allocation error in blckmgs')
185 
186 !-----------------------------------------------------------------
187 ! Calculate the number of blocks
188 !-----------------------------------------------------------------
189  nblcks = (n+ibsize-1)/ibsize
190  ibsize = min(n,ibsize)
191 
192 !-----------------------------------------------------------------
193 ! Orthogonalize the first block using modified schmidt
194 !-----------------------------------------------------------------
195  do ii=1,ibsize
196 
197  tmp = blas_dot(m,darray(:,ii),1,darray(:,ii),1)
198 
199 ! Linear dependence
200  if(tmp < thr) then
201  darray(1:m,ii) = 0.0_dp
202  cycle
203  end if
204 
205  tmp = 1.0_dp/sqrt(tmp)
206  call blas_scal(m,tmp,darray(:,ii),1)
207 
208  do jj=ii+1,ibsize
209  tmp = blas_dot(m,darray(:,ii),1,darray(:,jj),1)
210  call blas_axpy(m,-tmp,darray(:,ii),1,darray(:,jj),1)
211  end do
212 
213  end do
214 
215 !-----------------------------------------------------------------
216 ! Loop over remaining blocks
217 !-----------------------------------------------------------------
218  do ii=1,nblcks-1
219 
220 ! Initial and final column and number of columns in the block ii+1
221  istrt = ii*ibsize+1
222  iend = (ii+1)*ibsize
223  iend = min(n,iend)
224  ncol = iend - istrt + 1
225 
226 ! Orthogonalize the block ii+1 against the previous ones
227  do jj=1,ii
228 
229 ! Initial index of the block jj
230  jstrt = (jj-1)*ibsize+1
231 
232  call dgemm('t','n',ibsize,ncol,m,1.0_dp,darray(1,jstrt),ndim,darray(1,istrt),ndim,0.0_dp,smat,ibsize)
233  call dgemm('n','n',m,ncol,ibsize,-1.0_dp,darray(1,jstrt),ndim,smat,ibsize,1.0_dp,darray(1,istrt),ndim)
234 
235  end do
236 
237 ! Othogonalize vectors on the block ii+1 among themself using modified schmidt
238  do kk=istrt,iend
239 
240  tmp = blas_dot(m,darray(:,kk),1,darray(:,kk),1)
241 
242 ! Linear dependence
243  if(tmp < thr) then
244  darray(1:m,kk) = 0.0_dp
245  cycle
246  end if
247 
248  tmp = 1.0_dp/sqrt(tmp)
249  call blas_scal(m,tmp,darray(:,kk),1)
250 
251  do ll=kk+1,iend
252  tmp = blas_dot(m,darray(:,kk),1,darray(:,ll),1)
253  call blas_axpy(m,-tmp,darray(:,kk),1,darray(:,ll),1)
254  end do
255 
256  end do
257 
258  end do
259 
260 ! Clean up
261  deallocate(smat,stat=ierr)
262  if(ierr /= 0) call raise('E','Memory deallocation error in blckmgs')
263 
264 end subroutine dblckmgs

266 function ssyluinv(Amat,m) result(info)
267  implicit none
268  integer, intent(in) :: m
269  real(sp), intent(inout) :: amat(m,m)
270  integer, allocatable :: ipiv(:)
271  real(sp),allocatable :: temp(:)
272  real(sp),allocatable :: work(:)
273  integer :: lwork
274  integer :: info
275  real(sp) :: test(1)
276  integer :: i,j
277 
278  ! assume work space query, set best value to test after first dsytrf call
279  call lapack_sytrf('L',m,amat,m,ipiv,test,-1,info)
280  lwork=int(test(1))
281  allocate( work(lwork), source = 0.0_sp )
282 
283  ! Bunch-Kaufman factorization A = L*D*L**T
284  call lapack_sytrf('L',m,amat,m,ipiv,work,lwork,info)
285  if(info > 0) return
286 
287  ! A⁻¹ from factorized L matrix, save lower part of A⁻¹ in Amat matrix
288  ! Amat matrix is overwritten with lower triangular part of A⁻¹
289  call lapack_sytri('L',m,amat,m,ipiv,work,info)
290  if (info > 0) return
291 
292  ! symmetrizes A⁻¹ matrix from lower triangular part of inverse matrix
293  do i = 1, m
