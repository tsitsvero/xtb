28 subroutine onetri(ity,s,s1,array,n,ival)
29  ! ******designed for abelian groups only******
30  !
31  ! calling sequence:
32  ! ity =1 sym ao
33  ! =-1 anti sym ao
34  ! s input property matrix over ao's
35  ! s1 transformed integrals on output
36  ! s2 scratch arrays large enough to hold a square matrix
37  ! array mo matrix over so's
38  ! n linear dimension of arrays
39  ! bernd hess, university of bonn, january 1991
40  integer, intent(in) :: n
41  integer, intent(in) :: ity
42  integer, intent(in) :: ival
43  real(wp), intent(in) :: s(*)
44  real(wp), intent(inout) :: s1(*)
45  real(wp), intent(in) :: array(n,ival)
46  real(wp) :: s2(n,n)
47  external :: dsymm, dgemm ! can't use wrappers due to change of leading dim
48 
49  !
50  ! determine if we have an antisymmetric integral
51  if (ity /= -1) then
52  !
53  ! blow up symmetric matrix s
54  call blowsy(ity,s,s1,n)
55 
56  !
57  ! transformation of s
58  call dsymm('l','l',n,ival,1.d0,s1,n,array,n,0.d0,s2,n)
59  call dgemm('t','n',ival,ival,n,1.d0,array,n,s2,n,0.d0,s1,ival)
60  else
61  !
62  ! blow up anti-symmetric matrix s
63  call blowsy(ity,s,s1,n)
64  !
65  ! transformation of s
66  call dgemm('n','n',n,ival,n,1.d0,s1,n,array,n,0.d0,s2,n)
67  call dgemm('t','n',ival,ival,n,1.d0,array,n,s2,n,0.d0,s1,ival)
68  end if
69 end subroutine
