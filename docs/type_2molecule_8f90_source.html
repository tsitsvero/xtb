20 ! contains information about the molecular geometry, the atom types
21 ! the nuclear and total charge, atomic masses and all interatomic distances
22 ! In periodic calculations the lattice parameters, a Wigner--Seitz cell
23 ! as well as fractional coordinates are attached to the data type
24 !
25 ! Additionally the molecular structure can keep a topology information
26 ! usually containing the bonds of the system and a list of non-overlapping
27 ! fragments. Both data containers are optional and need not to be filled
28 ! by the provider of the structure data.
29 !
30 ! Vendor specific information can be stored along with the molecular structure
31 ! in auxilary data objects. Data objects like PDB or SDF atomic data should
32 ! be kept as light-weighted as possible and the user of the structure class
33 ! is not required to care about it existence

470  use xtb_mctc_accuracy, only : wp
471  use xtb_pbc_tools
472  implicit none
473  class(tmolecule),intent(inout) :: self
474  integer :: i,j
475  if (self%npbc > 0) then
476  do i = 1, self%n
477  do j = 1, i-1
478  self%dist(j,i) = minimum_image_distance(.false.,self%abc(:,i), &
479  & self%abc(:,j),self%lattice,self%pbc)
480  self%dist(i,j) = self%dist(j,i)

482  self%dist(j,i) = minimum_image_distance(.true.,self%abc(:,i), &

484  enddo
485  else

488  self%dist(j,i) = sqrt(sum((self%xyz(:,j)-self%xyz(:,i))**2))
489  self%dist(i,j) = self%dist(j,i)
490  enddo
491  self%dist(i,i) = 0.0_wp

493  endif

617 pure function moments_of_inertia(self) result(moments)
618  use xtb_mctc_accuracy, only : wp
619  use xtb_mctc_math, only : eigval3x3
620  implicit none
621  class(tmolecule),intent(in) :: self
622  real(wp) :: moments(3)
623  real(wp) :: center(3),atmass,inertia(3,3),work(9),tmp(3,3)
624  real(wp) :: x,x2,y,y2,z,z2
625  integer :: iat,info
626  ! currently not supported
627  if (self%npbc > 0) then
628  moments = -1.0_wp
629  return
630  endif
631  center = self%center_of_mass()
632 
633  inertia(:, :) = 0.0_wp
634 
635  do iat = 1, self%n
636  atmass = self%atmass(iat)
637  x = self%xyz(1,iat)-center(1); x2 = x**2
638  y = self%xyz(2,iat)-center(2); y2 = y**2
639  z = self%xyz(3,iat)-center(3); z2 = z**2
640  inertia(1,1) = inertia(1,1) + atmass * (y2+z2)
641  inertia(1,2) = inertia(1,2) - atmass * x*y
642  inertia(2,2) = inertia(2,2) + atmass * (x2+z2)
643  inertia(1,3) = inertia(1,3) - atmass * x*z
644  inertia(2,3) = inertia(2,3) - atmass * y*z
645  inertia(3,3) = inertia(3,3) + atmass * (x2+y2)
646  enddo
647 
648  call eigval3x3(inertia, moments)
649 
650 end function moments_of_inertia

665 pure subroutine align_to_principal_axes(self,break_symmetry)
666  use xtb_mctc_accuracy, only : wp
667  use xtb_mctc_math, only : eigvec3x3, matdet3x3
668  use xtb_pbc_tools
669  implicit none
670  class(tmolecule),intent(inout) :: self
671  logical, optional, intent(in) :: break_symmetry
672  real(wp) :: moments(3),det
673  real(wp) :: center(3),atmass,inertia(3,3),axes(3,3)
674  real(wp) :: x,x2,y,y2,z,z2
675  integer :: i,iat
676  ! currently not supported
677  if (self%npbc > 0) then
678  return
679  endif
680  center = self%center_of_mass()
681  inertia(:, :) = 0.0_wp
682  if (present(break_symmetry)) then
683  if (break_symmetry) then
684  inertia(1,1) = 1.0e-10_wp
685  inertia(1,2) = 2.0e-10_wp
686  inertia(2,2) = 3.0e-10_wp
687  inertia(1,3) = 4.0e-10_wp
688  inertia(2,3) = 5.0e-10_wp
689  inertia(3,3) = 6.0e-10_wp
690  end if
691  endif
692 
693  do iat = 1, self%n
694  atmass = self%atmass(iat)
695  x = self%xyz(1,iat)-center(1); x2 = x**2
696  y = self%xyz(2,iat)-center(2); y2 = y**2
697  z = self%xyz(3,iat)-center(3); z2 = z**2
698  inertia(1,1) = inertia(1,1) + atmass * (y2+z2)
699  inertia(1,2) = inertia(1,2) - atmass * x*y
700  inertia(2,2) = inertia(2,2) + atmass * (x2+z2)
701  inertia(1,3) = inertia(1,3) - atmass * x*z
702  inertia(2,3) = inertia(2,3) - atmass * y*z
703  inertia(3,3) = inertia(3,3) + atmass * (x2+y2)
704  enddo
705 
706  call eigvec3x3(inertia,moments,axes)
707 
708  det = matdet3x3(axes)
709  if (det < 0) axes(:,1) = -axes(:,1)
710
