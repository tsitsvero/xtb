171  case(0) ! ij do not overlap; ji do not overlap
172  ! nonoverlaping spheres
173  if(abs(rhoi-rhoj).lt.1.d-8) then
174  ! equal reduced radii
175  r1 = 1.0_wp/r
176  ap = r+rhoj
177  am = r-rhoj
178  ab = ap*am
179  rhab = rhoj/ab
180  lnab = 0.5_wp*log(am/ap)*r1
181  gi = rhab+lnab
182  dgi = -2.0_wp*rhab/ab+(rhab-lnab)*r1*r1

199  rhab = rhoj/ab
200  lnab = 0.5_wp*log(am/ap)*r1
201  gi = rhab+lnab
202  dgi = -2.0_wp*rhab/ab+(rhab-lnab)*r1*r1
203  ! ji contribution

218

220  dpsitr(:, jj) = dpsitr(:, jj)-drjj(:)
221  dpsidr(:, ii, jj) = dpsidr(:, ii, jj)+drjj(:)
222  endif
223 
224  case(10) ! ij do not overlap; ji overlap

228  ap = r+rhoj
229  am = r-rhoj
230  ab = ap*am
231  rhab = rhoj/ab

270

272  ! ij contribution
273  r1 = 1.0_wp/r
274  r12 = 0.5_wp*r1

303  lnab = 0.5_wp*log(am/ap)*r1

346  r24 = r12*r12
