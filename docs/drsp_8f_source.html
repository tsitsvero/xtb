34  SUBROUTINE rsp(A,N,MATZ,W,Z)
35  use xtb_mctc_accuracy, only : wp
36  IMPLICIT DOUBLE PRECISION (a-h,o-z)
37  dimension a(n*(n+1)/2), w(n), z(n,n)
38  dimension :: fv1(2*n),fv2(2*n)
39 *******************************************************************
40 *
41 * EISPACK DIAGONALIZATION ROUTINES: TO FIND THE EIGENVALUES AND
42 * EIGENVECTORS (IF DESIRED) OF A REAL SYMMETRIC PACKED MATRIX.
43 * ON INPUT- N IS THE ORDER OF THE MATRIX A,
44 * A CONTAINS THE LOWER TRIANGLE OF THE REAL SYMMETRIC
45 * PACKED MATRIX STORED ROW-WISE,
46 * MATZ IS AN INTEGER VARIABLE SET EQUAL TO ZERO IF ONLY
47 * EIGENVALUES ARE DESIRED, OTHERWISE IT IS SET TO
48 * ANY NON-ZERO INTEGER FOR BOTH EIGENVALUES AND
49 * EIGENVECTORS.
50 * ON OUTPUT- W CONTAINS THE EIGENVALUES IN ASCENDING ORDER,
51 * Z CONTAINS THE EIGENVECTORS IF MATZ IS NOT ZERO,
52 *
53 *******************************************************************
54 * THIS SUBROUTINE WAS CHOSEN AS BEING THE MOST RELIABLE. (JJPS)
55 C QUESTIONS AND COMMENTS SHOULD BE DIRECTED TO B. S. GARBOW,
56 C APPLIED MATHEMATICS DIVISION, ARGONNE NATIONAL LABORATORY
57 C
58 C ------------------------------------------------------------------
59 C
60  real(wp) esp,eta
61 
62  if(n.eq.1)then
63  z(1,1)=1.0d0
64  w(1)=a(1)
65  return
66  endif
67 
68 c CALL EPSETA(EPS,ETA)
69 c write(*,*) eps,eta
70  eps=1.000000000000000d-016
71  eta=1.000000000000000d-039
72 
73  nv=(n*(n+1))/2
74  nm=n
75  CALL tred3(n,nv,a,w,fv1,fv2,eps,eps)
76  IF (matz .NE. 0) GO TO 10
77 C ********** FIND EIGENVALUES ONLY **********
78  CALL tqlrat(n,w,fv2,ierr,eps)
79  GO TO 40
80 C ********** FIND BOTH EIGENVALUES AND EIGENVECTORS **********
81  10 DO 30 i = 1, n
82 C
83  DO 20 j = 1, n
84  z(j,i)=0.0d0
85  20 CONTINUE
86 C
87  z(i,i)=1.0d0
88  30 CONTINUE
89 C
90  CALL tql2(nm,n,w,fv1,z,ierr,eps)
91  IF (ierr .NE. 0) GO TO 40
92  CALL trbak3(nm,n,nv,a,n,z)
93 C ********** LAST CARD OF RSP **********
94  40 RETURN
95  END

97  SUBROUTINE epseta(EPS,ETA)
98  IMPLICIT DOUBLE PRECISION (a-h,o-z)
99 C
100 C COMPUTE AND RETURN ETA, THE SMALLEST REPRESENTABLE NUMBER,
101 C AND EPS IS THE SMALLEST NUMBER FOR WHICH 1+EPS.NE.1.
102 C
103 C
104  eta = 1.d0
105  10 IF((eta/2.d0).EQ.0.d0) GOTO 20
106  IF(eta.LT.1.d-38) GOTO 20
107  eta = eta / 2.d0
108  GOTO 10
109  20 eps = 1.d0
110  30 IF((1.d0+(eps/2.d0)).EQ.1.d0) GOTO 40
111  IF(eps.LT.1.d-17) GOTO 40
112  eps = eps / 2.d0
113  GOTO 30
114  40 RETURN
115  END

117  SUBROUTINE tql2(NM,N,D,E,Z,IERR,EPS)
118  IMPLICIT DOUBLE PRECISION (a-h,o-z)
119 C ===== PROCESSED BY AUGMENT, VERSION 4N =====
120 C APPROVED FOR VAX 11/780 ON MAY 6,1980. J.D.NEECE
121 C ----- LOCAL VARIABLES -----
122 C ----- GLOBAL VARIABLES -----
123  dimension d(*), e(*), z(nm,*)
124 C ----- SUPPORTING PACKAGE FUNCTIONS -----
125 C ===== TRANSLATED PROGRAM =====
126 C
127 C
128 C THIS SUBROUTINE IS A TRANSLATION OF THE ALGOL PROCEDURE TQL2,
129 C NUM. MATH. 11, 293-306(1968) BY BOWDLER, MARTIN, REINSCH, AND
130 C WILKINSON.
131 C HANDBOOK FOR AUTO. COMP., VOL.II-LINEAR ALGEBRA, 227-240(1971).
132 C
133 C THIS SUBROUTINE FINDS THE EIGENVALUES AND EIGENVECTORS
134 C OF A SYMMETRIC TRIDIAGONAL MATRIX BY THE QL METHOD.
135 C THE EIGENVECTORS OF A FULL SYMMETRIC MATRIX CAN ALSO
136 C BE FOUND IF TRED2 HAS BEEN USED TO REDUCE THIS
137 C FULL MATRIX TO TRIDIAGONAL FORM.
138 C
139 C ON INPUT-
140 C
141 C NM MUST BE SET TO THE ROW DIMENSION OF TWO-DIMENSIONAL
142 C ARRAY PARAMETERS AS DECLARED IN THE CALLING PROGRAM
143 C DIMENSION STATEMENT,
144 C
145 C N IS THE ORDER OF THE MATRIX,
146 C
147 C D CONTAINS THE DIAGONAL ELEMENTS OF THE INPUT MATRIX,
148 C
149 C E CONTAINS THE SUBDIAGONAL ELEMENTS OF THE INPUT MATRIX
150 C IN ITS LAST N-1 POSITIONS. E(1) IS ARBITRARY,
151 C
152 C Z CONTAINS THE TRANSFORMATION MATRIX PRODUCED IN THE
153 C REDUCTION BY TRED2, IF PERFORMED. IF THE EIGENVECTORS
154 C OF THE TRIDIAGONAL MATRIX ARE DESIRED, Z MUST CONTAIN
155 C THE IDENTITY MATRIX.
156 C
157 C ON OUTPUT-
158 C
159 C D CONTAINS THE EIGENVALUES IN ASCENDING ORDER. IF AN
160 C ERROR EXIT IS MADE, THE EIGENVALUES ARE CORRECT BUT
161 C UNORDERED FOR INDICES 1,2,...,IERR-1,
162 C
163 C E HAS BEEN DESTROYED,
164 C
165 C Z CONTAINS ORTHONORMAL EIGENVECTORS OF THE SYMMETRIC
166 C TRIDIAGONAL (OR FULL) MATRIX. IF AN ERROR EXIT IS MADE,
167 C Z CONTAINS THE EIGENVECTORS ASSOCIATED WITH THE STORED
168 C EIGENVALUES,
169 C
170 C IERR IS SET TO
171 C ZERO FOR NORMAL RETURN,
172 C J IF THE J-TH EIGENVALUE HAS NOT BEEN
173 C DETERMINED AFTER 30 ITERATIONS.
174 C
175 C QUESTIONS AND COMMENTS SHOULD BE DIRECTED TO B. S. GARBOW,
176 C APPLIED MATHEMATICS DIVISION, ARGONNE NATIONAL LABORATORY
177 C
178 C ------------------------------------------------------------------
179 C
180 C
181  ierr = 0
182  IF (n .EQ. 1) GO TO 160
183 C
184  DO 10 i = 2, n
185  10 e(i-1) = e(i)
186 C
187  f=0.0d0
188  b=0.0d0
189  e(n)=0.0d0
190 C
191  DO 110 l = 1, n
192  j = 0
193  h=eps*(abs(d(l))+abs(e(l)))
194  IF (b .LT. h) b=h
195 C ********** LOOK FOR SMALL SUB-DIAGONAL ELEMENT **********
196  DO 20 m = l, n
197  IF (abs(e(m)).LE.b) GO TO 30
198 C ********** E(N) IS ALWAYS ZERO, SO THERE IS NO EXIT
199 C THROUGH THE BOTTOM OF THE LOOP **********
200  20 CONTINUE
201 C
202  30 IF (m .EQ. l) GO TO 100
203  40 IF (j .EQ. 30) GO TO 150
204  j = j + 1
205 C ********** FORM SHIFT **********
206  l1 = l + 1
207  g = d(l)
208  p=(d(l1)-g)/(2.0d0*e(l))
209  r=sqrt(p*p+1.0d0)
210  d(l)=e(l)/(p+sign(r,p))
211  h = g - d(l)
212 C
213  DO 50 i = l1, n
214  50 d(i) = d(i) - h
215 C
216  f = f + h
217 C ********** QL TRANSFORMATION **********
218  p = d(m)
219  c=1.0d0
220  s=0.0d0
221  mml = m - l
222 C ********** FOR I=M-1 STEP -1 UNTIL L DO -- **********
223  DO 90 ii = 1, mml
224  i = m - ii
225  g = c * e(i)
226  h = c * p
227  IF (abs(p).LT.abs(e(i))) GO TO 60
228  c = e(i) / p
229  r=sqrt(c*c+1.0d0)
230  e(i+1) = s * p * r
231  s = c / r
232  c=1.0d0/r
233  GO TO 70
234  60 c = p / e(i)
235  r=sqrt(c*c+1.0d0)
236  e(i+1) = s * e(i) * r
237  s=1.0d0/r
238  c = c * s
239  70 p = c * d(i) - s * g
240  d(i+1) = h + s * (c * g + s * d(i))
241 C ********** FORM VECTOR **********
242  DO 80 k = 1, n
243  h = z(k,i+1)
244  z(k,i+1) = s * z(k,i) + c * h
245  z(k,i) = c * z(k,i) - s * h
246  80 CONTINUE
247 C
248  90 CONTINUE
249 C
250  e(l) = s * p
251  d(l) = c * p
252  IF (abs(e(l)).GT.b) GO TO 40
253  100 d(l) = d(l) + f
254  110 CONTINUE
255 C ********** ORDER EIGENVALUES AND EIGENVECTORS **********
256  DO 140 ii = 2, n
257  i = ii - 1
258  k = i
259  p = d(i)
260 C
261  DO 120 j = ii, n
262  IF (d(j) .GE. p) GO TO 120
263  k = j
264  p = d(j)
265  120 CONTINUE
266 C
267  IF (k .EQ. i) GO TO 140
268  d(k) = d(i)
269  d(i) = p
270 C
271  DO 130 j = 1, n
272  p = z(j,i)
273  z(j,i) = z(j,k)
274  z(j,k) = p
275  130 CONTINUE
276 C
277  140 CONTINUE
278 C
279  GO TO 160
280 C ********** SET ERROR -- NO CONVERGENCE TO AN
281 C EIGENVALUE AFTER 30 ITERATIONS **********
282  150 ierr = l
283  160 RETURN
284 C ********** LAST CARD OF TQL2 **********
285  END
286  SUBROUTINE tqlrat(N,D,E2,IERR,EPS)
287  IMPLICIT DOUBLE PRECISION (a-h,o-z)
288 C ===== PROCESSED BY AUGMENT, VERSION 4N =====
289 C APPROVED FOR VAX 11/780 ON MAY 6,1980. J.D.NEECE
290 C ----- LOCAL VARIABLES -----
291 C ----- GLOBAL VARIABLES -----
292  dimension d(*), e2(*)
293 C ----- SUPPORTING PACKAGE FUNCTIONS -----
294 C ===== TRANSLATED PROGRAM =====
295 C
296 C
297 C THIS SUBROUTINE IS A TRANSLATION OF THE ALGOL PROCEDURE TQLRAT,
298 C ALGORITHM 464, COMM. ACM 16, 689(1973) BY REINSCH.
299 C
300 C THIS SUBROUTINE FINDS THE EIGENVALUES OF A SYMMETRIC
301 C TRIDIAGONAL MATRIX BY THE RATIONAL QL METHOD.
302 C
303 C ON INPUT-
304 C
305 C N IS THE ORDER OF THE MATRIX,
306 C
307 C D CONTAINS THE DIAGONAL ELEMENTS OF THE INPUT MATRIX,
308 C
309 C E2 CONTAINS THE SQUARES OF THE SUBDIAGONAL ELEMENTS OF THE
310 C INPUT MATRIX IN ITS LAST N-1 POSITIONS. E2(1) IS ARBITRARY.
311 C
312 C ON OUTPUT-
313 C
314 C D CONTAINS THE EIGENVALUES IN ASCENDING ORDER. IF AN
315 C ERROR EXIT IS MADE, THE EIGENVALUES ARE CORRECT AND
316 C ORDERED FOR INDICES 1,2,...IERR-1, BUT MAY NOT BE
317 C THE SMALLEST EIGENVALUES,
318 C
319 C E2 HAS BEEN DESTROYED,
320 C
321 C IERR IS SET TO
322 C ZERO FOR NORMAL RETURN,
323 C J IF THE J-TH EIGENVALUE HAS NOT BEEN
324 C DETERMINED AFTER 30 ITERATIONS.
325 C
326 C QUESTIONS AND COMMENTS SHOULD BE DIRECTED TO B. S. GARBOW,
327 C APPLIED MATHEMATICS DIVISION, ARGONNE NATIONAL LABORATORY
328 C
329 C ------------------------------------------------------------------
330 C
331 C
332  ierr = 0
333  IF (n .EQ. 1) GO TO 140
334 C
335  DO 10 i = 2, n
336  10 e2(i-1) = e2(i)
337 C
338  f=0.0d0
339  b=0.0d0
340  e2(n)=0.0d0
341 C
342  DO 120 l = 1, n
343  j = 0
344  h=eps*(abs(d(l))+sqrt(e2(l)))
345  IF (b .GT. h) GO TO 20
346  b = h
347  c = b * b
348 C ********** LOOK FOR SMALL SQUARED SUB-DIAGONAL ELEMENT **********
349  20 DO 30 m = l, n
350  IF (e2(m) .LE. c) GO TO 40
351 C ********** E2(N) IS ALWAYS ZERO, SO THERE IS NO EXIT
352 C THROUGH THE BOTTOM OF THE LOOP **********
353  30 CONTINUE
354 C
355  40 IF (m .EQ. l) GO TO 80
356  50 IF (j .EQ. 30) GO TO 130
357  j = j + 1
358 C ********** FORM SHIFT **********
359  l1 = l + 1
360  s=sqrt(e2(l))
361  g = d(l)
362  p=(d(l1)-g)/(2.0d0*s)
363  r=sqrt(p*p+1.0d0)
364  d(l)=s/(p+sign(r,p))
365  h = g - d(l)
366 C
367  DO 60 i = l1, n
368  60 d(i) = d(i) - h
369 C
370  f = f + h
371 C ********** RATIONAL QL TRANSFORMATION **********
372  g = d(m)
373  IF (g.EQ.0.0d0) g=b
374  h = g
375  s=0.0d0
376  mml = m - l
377 C ********** FOR I=M-1 STEP -1 UNTIL L DO -- **********
378  DO 70 ii = 1, mml
379  i = m - ii
380  p = g * h
381  r = p + e2(i)
382  e2(i+1) = s * r
383  s = e2(i) / r
384  d(i+1) = h + s * (h + d(i))
385  g = d(i) - e2(i) / g
386  IF (g.EQ.0.0d0) g=b
387  h = g * p / r
388  70 CONTINUE
389 C
390  e2(l) = s * g
391  d(l) = h
392 C ********** GUARD AGAINST UNDERFLOW IN CONVERGENCE TEST **********
393  IF (h.EQ.0.0d0) GO TO 80
394  IF (abs(e2(l)).LE.abs(c/h)) GO TO 80
395  e2(l) = h * e2(l)
396  IF (e2(l).NE.0.0d0) GO TO 50
397  80 p = d(l) + f
398 C ********** ORDER EIGENVALUES **********
399  IF (l .EQ. 1) GO TO 100
400 C ********** FOR I=L STEP -1 UNTIL 2 DO -- **********
401  DO 90 ii = 2, l
402  i = l + 2 - ii
403  IF (p .GE. d(i-1)) GO TO 110
404  d(i) = d(i-1)
405  90 CONTINUE
406 C
407  100 i = 1
408  110 d(i) = p
409  120 CONTINUE
410 C
411  GO TO 140
412 C ********** SET ERROR -- NO CONVERGENCE TO AN
413 C EIGENVALUE AFTER 30 ITERATIONS **********
414  130 ierr = l
415  140 RETURN
416 C ********** LAST CARD OF TQLRAT **********
417  END
418  SUBROUTINE trbak3(NM,N,NV,A,M,Z)
419  IMPLICIT DOUBLE PRECISION (a-h,o-z)
420 C ===== PROCESSED BY AUGMENT, VERSION 4N =====
421 C APPROVED FOR VAX 11/780 ON MAY 6,1980. J.D.NEECE
422 C ----- LOCAL VARIABLES -----
423 C ----- GLOBAL VARIABLES -----
424  dimension a(*), z(nm,*)
425 C ===== TRANSLATED PROGRAM =====
426 C
427 C
428 C THIS SUBROUTINE IS A TRANSLATION OF THE ALGOL PROCEDURE TRBAK3,
429 C NUM. MATH. 11, 181-195(1968) BY MARTIN, REINSCH, AND WILKINSON.
430 C HANDBOOK FOR AUTO. COMP., VOL.II-LINEAR ALGEBRA, 212-226(1971).
431 C
432 C THIS SUBROUTINE FORMS THE EIGENVECTORS OF A REAL SYMMETRIC
433 C MATRIX BY BACK TRANSFORMING THOSE OF THE CORRESPONDING
434 C SYMMETRIC TRIDIAGONAL MATRIX DETERMINED BY TRED3.
435 C
436 C ON INPUT-
437 C
438 C NM MUST BE SET TO THE ROW DIMENSION OF TWO-DIMENSIONAL
439 C ARRAY PARAMETERS AS DECLARED IN THE CALLING PROGRAM
440 C DIMENSION STATEMENT,
441 C
442 C N IS THE ORDER OF THE MATRIX,
443 C
444 C NV MUST BE SET TO THE DIMENSION OF THE ARRAY PARAMETER A
445 C AS DECLARED IN THE CALLING PROGRAM DIMENSION STATEMENT,
446 C
447 C A CONTAINS INFORMATION ABOUT THE ORTHOGONAL TRANSFORMATIONS
448 C USED IN THE REDUCTION BY TRED3 IN ITS FIRST
449 C N*(N+1)/2 POSITIONS,
450 C
451 C M IS THE NUMBER OF EIGENVECTORS TO BE BACK TRANSFORMED,
452 C
453 C Z CONTAINS THE EIGENVECTORS TO BE BACK TRANSFORMED
454 C IN ITS FIRST M COLUMNS.
455 C
456 C ON OUTPUT-
457 C
458 C Z CONTAINS THE TRANSFORMED EIGENVECTORS
459 C IN ITS FIRST M COLUMNS.
460 C
461 C NOTE THAT TRBAK3 PRESERVES VECTOR EUCLIDEAN NORMS.
462 C
463 C QUESTIONS AND COMMENTS SHOULD BE DIRECTED TO B. S. GARBOW,
464 C APPLIED MATHEMATICS DIVISION, ARGONNE NATIONAL LABORATORY
465 C
466 C ------------------------------------------------------------------
467 C
468  IF (m .EQ. 0) GO TO 50
469  IF (n .EQ. 1) GO TO 50
470 C
471  DO 40 i = 2, n
472  l = i - 1
473  iz = (i * l) / 2
474  ik = iz + i
475  h = a(ik)
476  IF (h.EQ.0.0d0) GO TO 40
477 C
478  DO 30 j = 1, m
479  s=0.0d0
480  ik = iz
481 C
482  DO 10 k = 1, l
483  ik = ik + 1
484  s = s + a(ik) * z(k,j)
485  10 CONTINUE
486 C ********** DOUBLE DIVISION AVOIDS POSSIBLE UNDERFLOW **********
487  s = (s / h) / h
488  ik = iz
489 C
490  DO 20 k = 1, l
491  ik = ik + 1
492  z(k,j) = z(k,j) - s * a(ik)
493  20 CONTINUE
494 C
495  30 CONTINUE
496 C
497  40 CONTINUE
498 C
499  50 RETURN
500 C ********** LAST CARD OF TRBAK3 **********
501  END
502  SUBROUTINE tred3(N,NV,A,D,E,E2,EPS,ETA)
503  IMPLICIT DOUBLE PRECISION (a-h,o-z)
504 C ===== PROCESSED BY AUGMENT, VERSION 4N =====
505 C APPROVED FOR VAX 11/780 ON MAY 6,1980. J.D.NEECE
506 C ----- LOCAL VARIABLES -----
507 C ----- GLOBAL VARIABLES -----
508  dimension a(*), d(*), e(*), e2(*)
509 C ----- SUPPORTING PACKAGE FUNCTIONS -----
510 C ===== TRANSLATED PROGRAM =====
511 C
512 C
513 C THIS SUBROUTINE IS A TRANSLATION OF THE ALGOL PROCEDURE TRED3,
514 C NUM. MATH. 11, 181-195(1968) BY MARTIN, REINSCH, AND WILKINSON.
515 C HANDBOOK FOR AUTO. COMP., VOL.II-LINEAR ALGEBRA, 212-226(1971).
516 C
517 C THIS SUBROUTINE REDUCES A REAL SYMMETRIC MATRIX, STORED AS
518 C A ONE-DIMENSIONAL ARRAY, TO A SYMMETRIC TRIDIAGONAL MATRIX
519 C USING ORTHOGONAL SIMILARITY TRANSFORMATIONS.
520 C
521 C ON INPUT-
522 C
523 C N IS THE ORDER OF THE MATRIX,
524 C
525 C NV MUST BE SET TO THE DIMENSION OF THE ARRAY PARAMETER A
526 C AS DECLARED IN THE CALLING PROGRAM DIMENSION STATEMENT,
527 C
528 C A CONTAINS THE LOWER TRIANGLE OF THE REAL SYMMETRIC
529 C INPUT MATRIX, STORED ROW-WISE AS A ONE-DIMENSIONAL
530 C ARRAY, IN ITS FIRST N*(N+1)/2 POSITIONS.
531 C
532 C ON OUTPUT-
533 C
534 C A CONTAINS INFORMATION ABOUT THE ORTHOGONAL
535 C TRANSFORMATIONS USED IN THE REDUCTION,
536 C
537 C D CONTAINS THE DIAGONAL ELEMENTS OF THE TRIDIAGONAL MATRIX,
538 C
539 C E CONTAINS THE SUBDIAGONAL ELEMENTS OF THE TRIDIAGONAL
540 C MATRIX IN ITS LAST N-1 POSITIONS. E(1) IS SET TO ZERO,
541 C
542 C E2 CONTAINS THE SQUARES OF THE CORRESPONDING ELEMENTS OF E.
543 C E2 MAY COINCIDE WITH E IF THE SQUARES ARE NOT NEEDED.
544 C
545 C QUESTIONS AND COMMENTS SHOULD BE DIRECTED TO B. S. GARBOW,
546 C APPLIED MATHEMATICS DIVISION, ARGONNE NATIONAL LABORATORY
547 C
548 C ------------------------------------------------------------------
549 C
550 C ********** FOR I=N STEP -1 UNTIL 1 DO -- **********
551  DO 100 ii = 1, n
552  i = n + 1 - ii
553  l = i - 1
554  iz = ( i * l ) / 2
555  h=0.0d0
556  scale=0.0d0
557  DO 10 k = 1, l
558  iz = iz + 1
559  d(k) = a(iz)
560  scale=scale+abs( d(k) )
561  10 CONTINUE
562 C
563  IF ( scale.NE.0.d0 ) GO TO 20
564  e(i)=0.0d0
565  e2(i)=0.0d0
566  GO TO 90
567 C
568  20 DO 30 k = 1, l
569  d(k) = d(k) / scale
570  h = h + d(k) * d(k)
571  30 CONTINUE
572 C
573  e2(i) = scale * scale * h
574  f = d(l)
575  g=-sign(sqrt(h),f)
576  e(i) = scale * g
577  h = h - f * g
578  d(l) = f - g
579  a(iz) = scale * d(l)
580  IF (l .EQ. 1) GO TO 90
581  f=0.0d0
582 C
583  DO 70 j = 1, l
584  g=0.0d0
