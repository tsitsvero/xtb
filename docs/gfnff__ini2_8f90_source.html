292  endif

335  enddo

346  subroutine getnb(n,at,rad,r,mchar,icase,f,f2,nbf,nb,param)
347  implicit none
348  type(tgffdata), intent(in) :: param
349  integer n,at(n),nbf(20,n),nb(20,n)
350  real*8 rad(n*(n+1)/2),r(n*(n+1)/2),mchar(n),f,f2
351 
352  integer i,j,k,nn,icase,hc_crit,nnfi,nnfj,lin
353  integer tag(n*(n+1)/2)
354  real*8 rco,fm
355 
356  nb = 0 ! resulting array (nbf is full from first call)
357  tag = 0
358  do i=1,n
359  nnfi=nbf(20,i) ! full CN of i, only valid for icase > 1
360  do j=1,i-1
361  nnfj=nbf(20,j) ! full CN of i
362  fm=1.0d0
363 ! full case
364  if(icase.eq.1)then
365  if(param%metal(at(i)).eq.2) fm=fm*f2 !change radius for metal atoms
366  if(param%metal(at(j)).eq.2) fm=fm*f2
367  if(param%metal(at(i)).eq.1) fm=fm*(f2+0.025)
368  if(param%metal(at(j)).eq.1) fm=fm*(f2+0.025)
369  endif
370 ! no HC atoms
371  if(icase.eq.2)then
372  hc_crit = 6
373  if(param%group(at(i)).le.2) hc_crit = 4
374  if(nnfi.gt.hc_crit) cycle
375  hc_crit = 6
376  if(param%group(at(j)).le.2) hc_crit = 4
377  if(nnfj.gt.hc_crit) cycle
378  endif
379 ! no metals and unusually coordinated stuff
380  if(icase.eq.3)then
381  if(mchar(i).gt.0.25 .or. param%metal(at(i)).gt.0) cycle ! metal case TMonly ?? TODO
382  if(mchar(j).gt.0.25 .or. param%metal(at(j)).gt.0) cycle ! metal case
383  if(nnfi.gt.param%normcn(at(i)).and.at(i).gt.10) cycle ! HC case
384  if(nnfj.gt.param%normcn(at(j)).and.at(j).gt.10) cycle ! HC case
385  endif
386  k=lin(j,i)
387  rco=rad(k) !(rad(i)+rad(j))/0.5291670d0
388 ! R est. R0
389  if(r(k).lt. fm * f * rco) tag(k)=1
390  enddo
391  enddo
392 
393  do i=1,n
394  nn = 0
395  do j=1,n
396  if(tag(lin(j,i)).eq.1.and.i.ne.j) then
397  nn=nn+1
398  nb(nn,i)=j
399  endif
400  enddo
401  nb(20,i) = nn
402  enddo
403 
404  end subroutine getnb
