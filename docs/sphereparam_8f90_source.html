155  implicit none
156  integer, intent(in) :: n
157  integer, intent(in) :: at(n)

159 
160  real(wp) :: dum(3),rx,ry,rz
161  real(wp) :: x,y,z,f,rr,r
162  integer :: i,j
163 
164 ! call axis3(0,n,at,xyz,xyz2,dum)
165 
166 ! xyz = xyz2
167 
168  r=0.0_wp
169  do i=1,n-1
170  do j=i+1,n
171  rx=xyz(1,i)-xyz(1,j)
172  ry=xyz(2,i)-xyz(2,j)
173  rz=xyz(3,i)-xyz(3,j)
174  rr=sqrt(rx**2+ry**2+rz**2)
175  if(rr.gt.r) r=rr
176  enddo
177  enddo
178 
179  boxr=0.5*r+3.5d0 ! assume some vdW radius

181 ! but here it is treated as an absolute measure, using this routine will
182 ! artifically compress larger molecules. I should fix this, dunno how.

338 subroutine logfermi_cavity_list(nat,at,xyz,nlist,list,temp,alpha,center,radius,&
339  & efix,gfix)
340  use xtb_mctc_constants, only : kb
341  implicit none
342  integer, intent(in) :: nat
343  integer, intent(in) :: at(nat)
344  real(wp),intent(in) :: xyz(3,nat)
345  integer, intent(in) :: nlist
346  integer, intent(in) :: list(nlist)
347 
348  real(wp),intent(in) :: temp ! temperature
349  real(wp),intent(in) :: alpha ! potential steepness
350  real(wp),intent(in) :: center(3) ! aufpunkt of wall potential
351  real(wp),intent(in) :: radius(3) ! radius of the sphere (anisotropic)
352 
353  integer :: i,iat
354  real(wp) :: r(3),w(3),dist,R0,expterm,fermi
355 
356  real(wp),intent(inout) :: efix
357  real(wp),intent(inout) :: gfix(3,nat)
358 
359  r0 = maxval(radius)
360  w = r0/radius ! for anisotropy
361 
362  do i = 1, nlist
363  iat = list(i)
364  r = w*(xyz(:,iat) - center)
365  dist = sqrt(sum(r**2))
366  expterm = exp(alpha*(dist-r0))
367  fermi = 1.0_wp/(1.0_wp+expterm)
368  efix = efix + kb*temp * log( 1.0_wp+expterm )
369  gfix(:,iat) = gfix(:,iat) + kb*temp * alpha*expterm*fermi * (r*w)/(dist+1.0e-14_wp)
370  enddo
371 
372 end subroutine logfermi_cavity_list
