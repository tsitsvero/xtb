22  Subroutine dthetadr(nat,xyz,j,k,i,theta,kijk,dei,dej,dek)
23  use xtb_mctc_accuracy, only : wp
24 
25 ! .....................................................................
26 !
27 ! Input
28 ! -----
29 ! __ __
30 ! ra,rb : distance ij and distnace ik
31 !
32 ! sinth : sinus of angle j-i-k
33 !
34 ! costh : cosinus of angle j-i-k
35 !
36 !
37 ! Output
38 ! ------
39 !
40 ! dei : derivatives of the atom i
41 !
42 ! dej : derivatives of the atom j
43 !
44 ! dek : derivatives of the atom k
45 !
46 ! .....................................................................
47 
48  implicit none
49 
50  external vecnorm
51 
52  integer ic,nat,j,k,i
53 
54  real(wp) ran,rbn,theta,xyz(3,nat)
55  real(wp) ra(3),rb(3),sinth,costh
56  real(wp) dei(3),dej(3),dek(3),kijk
57  real(wp) fac,vecnorm,eps
58 
59  parameter(eps=1.d-14)
60 
61 ! ... dtheta/dri, dtheta/drj and dtheta/drk
62 
63  sinth=sin(theta)
64  costh=cos(theta)
65 
66  do ic=1,3
67  ra(ic)=xyz(ic,j)-xyz(ic,i)
68  rb(ic)=xyz(ic,k)-xyz(ic,i)
69  end do
70 
71  ran=vecnorm(ra,3,0)
72  rbn=vecnorm(rb,3,0)
73  if (abs(sinth).le.eps) then
74  fac=sign(kijk,costh)/(ran*rbn)
75  else
76  fac=1.d0/(sinth*ran*rbn)
77  end if
78 
79  do ic=1,3
80 ! ... dedri
81  dei(ic)=fac*( ra(ic)+rb(ic)-(rbn*rbn*ra(ic)+ran*ran*rb(ic))* &
82  costh/(ran*rbn))
83 ! ... dedrj
84  dej(ic)=fac*(ra(ic)*costh*rbn/ran-rb(ic))
85 ! ... dedrk
86  dek(ic)=fac*(rb(ic)*costh*ran/rbn-ra(ic))
87  end do
88 
89  End Subroutine dthetadr

91  Subroutine dphidr(nat,xyz,i,j,k,l,phi, &
92  dphidri,dphidrj,dphidrk,dphidrl)

94 ! the torsion derivatives
95 
96  implicit none
97 
98  external vecnorm
99 
100  integer ic,i,j,k,l,nat
101 
102  real(wp) sinphi,cosphi,onenner,thab,thbc
103  real(wp) ra(3),rb(3),rc(3),rab(3),rac(3),rbc(3),rbb(3)
104  real(wp) raa(3),rba(3),rapba(3),rapbb(3),rbpca(3),rbpcb(3)
105  real(wp) rapb(3),rbpc(3),na(3),nb(3),nan,nbn
106  real(wp) dphidri(3),dphidrj(3),dphidrk(3),dphidrl(3)
107  real(wp) xyz(3,nat),phi,vecnorm,nenner,eps,vz
108 
109  parameter(eps=1.d-14)
110 
111  cosphi=cos(phi)
112  sinphi=sin(phi)
113  do ic=1,3
114  ra(ic)=xyz(ic,j)-xyz(ic,i)
115  rb(ic)=xyz(ic,k)-xyz(ic,j)
116  rc(ic)=xyz(ic,l)-xyz(ic,k)
117 
118  rapb(ic)=ra(ic)+rb(ic)
119  rbpc(ic)=rb(ic)+rc(ic)
120  end do
121 
122  call crossprod(ra,rb,na)
123  call crossprod(rb,rc,nb)
124  nan=vecnorm(na,3,0)
125  nbn=vecnorm(nb,3,0)
126 
127  nenner=nan*nbn*sinphi
128  if (abs(nenner).lt.eps) then
129  dphidri=0
130  dphidrj=0
131  dphidrk=0
132  dphidrl=0
133  onenner=1.0d0/(nan*nbn)
134  else
135  onenner=1.d0/nenner
136  endif
137 
138  call crossprod(na,rb,rab)
139  call crossprod(nb,ra,rba)
140  call crossprod(na,rc,rac)
141  call crossprod(nb,rb,rbb)
142  call crossprod(nb,rc,rbc)
143  call crossprod(na,ra,raa)
144 
145  call crossprod(rapb,na,rapba)
146  call crossprod(rapb,nb,rapbb)
147  call crossprod(rbpc,na,rbpca)
148  call crossprod(rbpc,nb,rbpcb)
149 
150 ! ... dphidri
151  do ic=1,3
152  dphidri(ic)=onenner*(cosphi*nbn/nan*rab(ic)-rbb(ic))
153 
154 ! ... dphidrj
155  dphidrj(ic)=onenner*(cosphi*(nbn/nan*rapba(ic) &
156  +nan/nbn*rbc(ic)) &
157  -(rac(ic)+rapbb(ic)))
158 ! ... dphidrk
159  dphidrk(ic)=onenner*(cosphi*(nbn/nan*raa(ic) &
160  +nan/nbn*rbpcb(ic)) &
161  -(rba(ic)+rbpca(ic)))
162 ! ... dphidrl
163  dphidrl(ic)=onenner*(cosphi*nan/nbn*rbb(ic)-rab(ic))
164  end do
165 
166  End Subroutine dphidr

228

231  end do
232  else
233  do ic=1,3
234  domegadri(ic)=0.d0
235  domegadrj(ic)=0.d0
236  domegadrk(ic)=0.d0
237  domegadrl(ic)=0.d0
238  end do

241  End Subroutine domegadr

274  rbabs=rbabs+rb(ic)*rb(ic)

276  end do

288  -ra(2)*(rb(1)*rc(3)-rb(3)*rc(1)) &
289  +ra(3)*(rb(1)*rc(2)-rb(2)*rc(1))
290 
291  call crossprod(-ra,rc,na)
292  test=0.0d0
293  do ic=1,3
294  test=test+na(ic)*rb(ic)
295  end do
296 
297  thab=valijk(nat,xyz,i,k,j)
298  thbc=valijk(nat,xyz,j,l,k)
299  call crossprod(ra,rb,na)
300  call crossprod(rb,rc,nb)
301  nan=vecnorm(na,3,1)
302  nbn=vecnorm(nb,3,1)
303 
304  snanb=0.0d0
305  do ic=1,3
306  snanb=snanb+na(ic)*nb(ic)
307  end do
308  if (abs(abs(snanb)-1.d0).lt.eps) then
309  snanb=sign(1.d0,snanb)
310  end if
311 
312  valijkl=acos(snanb)
313  if(test.lt.0.0d0) valijkl=-acos(snanb)
