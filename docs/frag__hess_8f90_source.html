320  subroutine frag_hess_diag( nat,hess,eig_calc,ispinsyst,nspinsyst,nsystem )
321  !---------------------------------------------------------------------------------------------
322  ! Purpose:
323  ! Subroutine performs diagonalization of fragmented hessian.
324  ! fragmentize must be called beforehand, fragments are available via this module.
325  !
326  ! How does it work?
327  ! This subroutine uses masked arrays: an array (hess_mask) consists only of logicals and has
328  ! the same size as the hessian matrix of the entire system. For all entries of the hessian
329  ! matrix that belong to one fragment, the corresponding position in the masked array is set
330  ! true. Using the intrinsic "pack and unpack" functions in combination with mask, the hessian
331  ! entries for only one fragment are written into a smaller array (mini_hess) which is faster
332  ! diagonalized due to the reduzed size (# of atoms in fragment < # of atoms in the entire system).
333  !
334  ! Input:
335  ! nat - Number of atoms of the entire system
336  ! hess - (Lindh) Hessian of the entire system
337  !
338  ! Output:
339  ! hess - diagonalized hessian (eigenvectors), overwritten
340  ! eig_calc - diagonalized hessian (eigenvalues)
341  !---------------------------------------------------------------------------------------------
342  use xtb_mctc_accuracy, only : wp, sp
343  implicit none
344  !Dummy Arguments
345  integer, intent(in) :: nat ! # of atoms
346  real(sp), intent(inout) :: hess(3*nat,3*nat) ! input hessian
347  real(sp), intent(out) :: eig_calc(3*nat) ! eigenvectors od entire system
348  integer, intent(in) :: ispinsyst(:,:) ! array with list of atoms of each fragment
349  integer, intent(in) :: nspinsyst(:) ! array with # of atoms for each fragment
350  integer, intent(in) :: nsystem ! # of fragments
351  !Stack
352  integer :: isystem
353  integer :: i,j,ii,jj,k
354  integer :: nat3 ! # of atoms in the system * 3
355  integer :: nat_cur ! # of atoms in fragment
356  integer :: nat3_cur ! nat in fragments * 3
357  integer :: info ! for ssyev
358  integer :: lwork ! for ssyev
359  logical :: hess_mask(3*nat,3*nat) ! masked hessian array
360  real(sp) :: ev_calc(3*nat,3*nat) ! eigenvectors of entire system
361  real(sp), allocatable :: mini_hess(:,:) ! eigenvectors of fragment
362  real(sp), allocatable :: eig(:) ! eigenvalues of fragment
363  real(sp), allocatable :: aux(:) ! for ssyev
364 
365 
366  ev_calc = 0.0e0_sp
367  eig_calc = 0.0e0_sp
368  nat3 = 3 * nat
369 
370 !$omp parallel default(none) &
371 !$omp private(isystem,i,ii,j,jj,nat_cur,nat3_cur,mini_hess,hess_mask,eig,lwork,aux,info) &
372 !$omp shared(nsystem,ev_calc,eig_calc,hess,nspinsyst,ispinsyst)
373 !!$omp do schedule(static)
374 !$omp do reduction(+:ev_calc,eig_calc)
375  do isystem = 1 , nsystem
376  hess_mask = .false.
377  do i = 1,nspinsyst(isystem)
378  do j = 1,i
379 
380  nat_cur = nspinsyst(isystem)
381  nat3_cur = 3 * nat_cur
382  ii = 3*ispinsyst(i,isystem)
383  jj = 3*ispinsyst(j,isystem)
384  hess_mask(ii-2:ii,jj-2:jj) = .true.
385  hess_mask(jj-2:jj,ii-2:ii) = .true.
386 
387  end do
388  end do
389 
390  allocate( mini_hess(nat3_cur,nat3_cur), source = 0.0e0_sp )
391  allocate( eig(nat3_cur), source = 0.0e0_sp )
392 
393  mini_hess = reshape( pack( hess, mask = hess_mask ), shape( mini_hess ) )
394  lwork = 1 + 6*nat3_cur + 2*nat3_cur**2
395  allocate(aux(lwork))
396  call ssyev ('V','U',nat3_cur,mini_hess,nat3_cur,eig,aux,lwork,info)
397  deallocate(aux)
398 !!$omp critical
399  ev_calc = unpack( reshape( mini_hess, [ nat3_cur*nat3_cur ] ), mask = hess_mask, field = ev_calc )
400  eig_calc = unpack( eig, mask = any(hess_mask,1), field = eig_calc )
401 !!$omp end critical
402 
403  deallocate( mini_hess,eig )
404 
405  end do
406 !$omp end do
407 !$omp end parallel
408 
409  do i = 1,nat3
410  ev_calc(:,i) = ev_calc(:,i) - ( sum( ev_calc(:,i) ) / nat3 )
411  end do
412 
413  hess = ev_calc
414 
415  call eigsort4(nat3,hess,eig_calc)
416 
417  end subroutine frag_hess_diag

422 !calculates the shortest path between two atoms
423 !start and goal are integers, determining the index in xyz
424 !https://de.wikipedia.org/wiki/Dijkstra-Algorithmus
425 !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
426 function shortest_distance(nspin, start, goal, neighbours, input_distances, visited, precessor)
427  use xtb_mctc_accuracy, only : wp, sp
428  implicit none
429  !Dummy Arguments:
430  real(wp) :: shortest_distance
431  integer, intent(in) :: nspin
432  integer, intent(in) :: start
433  integer, intent(in) :: goal
434  integer, intent(in) :: neighbours(20, nspin)
435  real(wp), intent(in) :: input_distances(nspin, nspin)
436  logical, intent(out) :: visited(nspin)
437  integer, intent(out) :: precessor(nspin)
438 
439  !Stack:
440  integer :: current
441  integer :: neighbour
442  integer :: i_neighbours
443  integer :: bonds
444  real(wp) :: alt_dist
445  real(wp) :: distance(nspin)
446 
447  !logical:: visited(nspin)
448 
449  bonds = 0
450  precessor = 0
451  distance = huge(distance)
452  distance(start) = 0
453  visited = .false.
454  do while (.not.all(visited)) !as long there are unvisited nodes
455  current = minloc(distance, 1,.not. visited)
456 
457  !Abort if Fragments are not connected:
458  if (distance(current) == huge(distance)) then
459  shortest_distance = huge(distance)
460  return
461  end if ! distance(current) == huge(distance)
462 
463  visited(current) = .true.
464  if (current == goal) then
465  shortest_distance = distance(goal)
466  !route to target found
467  do while (precessor(current) /= 0)
468  bonds = bonds + 1
469  current = precessor(current)
470  end do ! End loop: while precessor(current) /= 0
471  exit
472  else
473  !loop over all neighbours of current atom
474  do i_neighbours = 1, neighbours(20, current)
475  neighbour = neighbours(i_neighbours, current)
476  if (.not.visited(neighbour)) then
477  !distanzupdate
478  alt_dist = distance(current) + input_distances(current, neighbour)
479  !write( *, * ) alt_dist, distance(current),
480  if ((alt_dist < distance(neighbour))) then
481  distance(neighbour) = alt_dist
482  precessor(neighbour) = current
483  end if ! (alt_dist < distance(neighbour))
484  endif !(visited(neighbour))
485  end do ! End Loop over i_neighbours from 1 to all_Atoms(current)%n_neighbours
486  endif
487  end do ! End loop: while sum(visited) /= 0
488  !initialize
489  shortest_distance = distance(goal)
490 end function shortest_distance
