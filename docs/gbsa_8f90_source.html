395 subroutine update(self, env, num, xyz)
396 
398  class(tborn), intent(inout) :: self
399 
401  type(tenvironment), intent(inout) :: env
402 
404  integer, intent(in) :: num(:)
405 
407  real(wp), intent(in) :: xyz(:, :)
408 
409  ! initialize the neighbor list
410  call update_nnlist_gbsa(self%nat, self%ntpair, self%ppind, xyz, &
411  & self%lrcut, self%srcut, self%nnsas, self%nnlists, self%nnrad, &
412  & self%nnlistr, self%ddpair, .false.)
413 
414  call compute_bornr(self%nat, self%nnrad, self%nnlistr, self%ddpair, &
415  & self%vdwr, self%rho, self%svdw, self%bornScale, self%brad, self%brdr)
416 
417  ! compute solvent accessible surface and its derivatives
418  call compute_numsa(self%nat, self%nnsas, self%nnlists, xyz, self%vdwsa, &
419  & self%wrp, self%trj2, self%angWeight, self%angGrid, &
420  & self%sasa, self%dsdrt)
421 
422  ! contract surface gradient
423  call mctc_gemv(self%dsdrt, self%gamsasa, self%dsdr)
424  self%gsasa = mctc_dot(self%sasa, self%gamsasa)
425 
426  ! compute the Debye-Hueckel ion exclusion term
427  if (self%lsalt) then
428  call getdebyehueckel(self%nat, self%dielectricConst, self%kappa, &
429  & self%ionRad, self%brad, self%ionscr, self%discr)
430  end if
431 
432  ! compute the HB term
433  if (self%lhb) then
434  call compute_fhb(self%nat, self%hbmag, self%vdwsa, self%sasa, &
435  & self%hbw, self%dhbdw)
436  end if
437 
438  if (self%alpbet > 0.0_wp) then
439  call getadet(self%nat, xyz, self%vdwr, self%aDet)
440  end if
441 
442  self%bornMat(:, :) = 0.0_wp
443  call self%addBornMatrix(env, num, xyz, self%bornMat)
444

680  real(wp), parameter :: tof = 2.0_wp/5.0_wp, unity(3, 3) = reshape(&
681  & [1.0_wp, 0.0_wp, 0.0_wp, 0.0_wp, 1.0_wp, 0.0_wp, 0.0_wp, 0.0_wp, 1.0_wp], &
682  & [3, 3])
683 
684  totrad3 = 0.0_wp
685  center(:) = 0.0_wp
686  do iat = 1, natom
687  rad2 = rad(iat) * rad(iat)
688  rad3 = rad2 * rad(iat)
689  totrad3 = totrad3 + rad3
690  center(:) = center + xyz(:, iat) * rad3
691  end do
692  center = center / totrad3
693 
694  inertia(:, :) = 0.0_wp
695  do iat = 1, natom
696  rad2 = rad(iat) * rad(iat)
697  rad3 = rad2 * rad(iat)
698  vec(:) = xyz(:, iat) - center
699  r2 = sum(vec**2)
700  inertia(:, :) = inertia + rad3 * ((r2 + tof*rad2) * unity &
701  & - spread(vec, 1, 3) * spread(vec, 2, 3))
702  end do
703 
704  adet = sqrt(matdet3x3(inertia)**(1.0_wp/3.0_wp)/(tof*totrad3))
705
