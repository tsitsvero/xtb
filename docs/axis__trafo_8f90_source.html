94  ! xsum=evec(1,1)*(evec(2,2)*evec(3,3)-evec(3,2)*evec(2,3)) +
95  ! 1 evec(1,2)*(evec(2,3)*evec(3,1)-evec(2,1)*evec(3,3)) +
96  ! 2 evec(1,3)*(evec(2,1)*evec(3,2)-evec(2,2)*evec(3,1))
97  do i=1,3

99  eig(i)=0.d0

108  bb=rot(2)/2.9979245d+4
109  cc=rot(1)/2.9979245d+4

305  subroutine axis3(mode,numat,nat,coord,coordout,eig)
306  use xtb_splitparam
307  implicit none
308  integer, intent(in) :: numat,nat(numat),mode
309  real(wp),intent(in) :: coord(3,numat)
310  real(wp),intent(out) :: coordout(3,numat),eig(3)
311 
312  real(wp) :: sumw,sumwx,sumwy,sumwz,xsum,eps,ams
313  real(wp) :: t(6), evec(3,3)
314  real(wp) :: x(numat),y(numat),z(numat),coordtmp(3,numat)
315  real(wp) :: coord1(3,numat)
316  integer :: i,j,k
317  t = (/(0.0_wp,i=1,6)/)
318 
319  sumw=1.e-20_wp
320  sumwx=0.0_wp
321  sumwy=0.0_wp
322  sumwz=0.0_wp
323  do i=1,numat
324  if(mode.eq.0)then
325  ams = atmass(i)
326  else
327  ams = 1._wp/atmass(i)
328  endif
329  sumw=sumw+ams
330  sumwx=sumwx+ams*coord(1,i)
331  sumwy=sumwy+ams*coord(2,i)
332  sumwz=sumwz+ams*coord(3,i)
333  enddo
334 
335  eps=1.e-3_wp
336  sumwx=sumwx/sumw
337  sumwy=sumwy/sumw
338  sumwz=sumwz/sumw
339  do i=1,numat
340  x(i)=coord(1,i)-sumwx
341  y(i)=coord(2,i)-sumwy
342  z(i)=coord(3,i)-sumwz
343  coordtmp(1,i)=x(i)
344  coordtmp(2,i)=y(i)
345  coordtmp(3,i)=z(i)
346  enddo
347 
348  do i=1,6
349  t(i)=real(i,wp)*1.0e-10_wp
350  enddo
351 
352  do i=1,numat
353  t(1)=t(1)+atmass(i)*(y(i)**2+z(i)**2)+eps
354  t(2)=t(2)-atmass(i)*x(i)*y(i)
355  t(3)=t(3)+atmass(i)*(z(i)**2+x(i)**2)+eps
356  t(4)=t(4)-atmass(i)*z(i)*x(i)
357  t(5)=t(5)-atmass(i)*y(i)*z(i)
358  t(6)=t(6)+atmass(i)*(x(i)**2+y(i)**2)+eps
359  enddo
360 
361  call rsp(t,3,3,eig,evec)
362 
363 ! now to orient the molecule so the chirality is preserved
364  xsum=evec(1,1)*(evec(2,2)*evec(3,3)-evec(3,2)*evec(2,3)) + &
365  & evec(1,2)*(evec(2,3)*evec(3,1)-evec(2,1)*evec(3,3)) + &
366  & evec(1,3)*(evec(2,1)*evec(3,2)-evec(2,2)*evec(3,1))
367  if( xsum .lt. 0) then
368  do j=1,3
369  80 evec(j,1)=-evec(j,1)
370  enddo
371  endif
372 ! call prmat(6,evec,3,3,'Rmat')
373 
374  do i=1,numat
375  do j=1,3
376  xsum=0.d0
377  do k=1,3
378  xsum=xsum+coordtmp(k,i)*evec(k,j)
379  enddo
380  coord1(j,i)=xsum
381  enddo
382  enddo
383 
384  do i=1,numat
385  coordout(1,i)=coord1(1,i)
386  coordout(2,i)=coord1(2,i)
387  coordout(3,i)=coord1(3,i)
388  enddo
389 
390  return
391  end subroutine axis3
