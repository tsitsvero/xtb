96 subroutine singlepoint(self, env, mol, chk, printlevel, restart, &
97  & energy, gradient, sigma, hlgap, results)
98 
100  character(len=*), parameter :: source = 'xtb_calculator_singlepoint'
101 
103  class(txtbcalculator), intent(inout) :: self
104 
106  type(tenvironment), intent(inout) :: env
107 
109  type(tmolecule), intent(inout) :: mol
110 
112  type(trestart), intent(inout) :: chk
113 
115  integer, intent(in) :: printlevel
116 
118  logical, intent(in) :: restart
119 
121  real(wp), intent(out) :: energy
122 
124  real(wp), intent(out) :: gradient(:, :)
125 
127  real(wp), intent(out) :: sigma(:, :)
128 
130  real(wp), intent(out) :: hlgap
131 
133  type(scc_results), intent(out) :: results
134 
135  class(tsolvation), allocatable :: solvation
136  type(tborn), allocatable :: gbsa
137  integer :: i,ich
138  integer :: mode_sp_run = 1
139  real(wp) :: efix
140  logical :: inmol
141  logical, parameter :: ccm = .true.
142  logical :: exitRun
143 
144  call mol%update
145  if (mol%npbc > 0) call generate_wsc(mol,mol%wsc)
146 
147  energy = 0.0_wp
148  gradient(:, :) = 0.0_wp
149  sigma(:, :) = 0.0_wp
150  hlgap = 0.0_wp
151  efix = 0.0_wp
152 
153  ! ------------------------------------------------------------------------
154  ! actual calculation
155  select case(self%xtbData%level)
156  case(1, 2)
157  if (allocated(self%solvation)) then
158  call newsolvationmodel(self%solvation, env, solvation, mol%at)
159  end if
160  call scf(env,mol,chk%wfn,self%basis,self%pcem,self%xtbData,solvation, &
161  & hlgap,self%etemp,self%maxiter,printlevel,restart,.true., &
162  & self%accuracy,energy,gradient,results)
163 
164  case(0)
165  if (allocated(self%solvation)) then
166  allocate(gbsa)
167  call newbornmodel(self%solvation, env, gbsa, mol%at)
168  end if
169  call peeq &
170  & (env,mol,chk%wfn,self%basis,self%xtbData,gbsa,hlgap,self%etemp, &
171  & printlevel,.true.,ccm,self%accuracy,energy,gradient,sigma,results)
172 
173  end select
174 
175  call env%check(exitrun)
176  if (exitrun) then
177  call env%error("Electronic structure method terminated", source)
178  return
179  end if
180 
181  ! ------------------------------------------------------------------------
182  ! post processing of gradient and energy
183 
184  ! point charge embedding gradient file
185  if (allocated(pcem_grad) .and. self%pcem%n > 0) then
186  call open_file(ich,pcem_grad,'w')
187  do i=1,self%pcem%n
188  write(ich,'(3f12.8)')self%pcem%grd(1:3,i)
189  enddo
190  call close_file(ich)
191  endif
192 
193  ! ------------------------------------------------------------------------
194  ! various external potentials
195  call constrain_pot(potset,mol%n,mol%at,mol%xyz,gradient,efix)
196  call constrpot (mol%n,mol%at,mol%xyz,gradient,efix)
197  call cavity_egrad(mol%n,mol%at,mol%xyz,efix,gradient)
198 
199  if (prob_flag) then
200  call prob_metadynamic (metaset,mol%n,mol%at,mol%xyz,efix,gradient)
201  call prob_metadynamic (rmsdset,mol%n,mol%at,mol%xyz,efix,gradient)
202  else
203  call metadynamic (metaset,mol%n,mol%at,mol%xyz,efix,gradient)
204  call metadynamic (rmsdset,mol%n,mol%at,mol%xyz,efix,gradient)
205  end if
206 
207  ! ------------------------------------------------------------------------
208  ! fixing of certain atoms
209  ! print*,abs(efix/etot)
210  energy = energy + efix
211  results%e_total = energy
212  results%gnorm = norm2(gradient)
213  if (fixset%n.gt.0) then
214  do i=1, fixset%n
215  !print*,i,fixset%atoms(i)
216  gradient(1:3,fixset%atoms(i))=0
217  enddo
218  endif
219 
220  if (printlevel.ge.2) then
221  ! start with summary header
222  if (.not.silent) then
223  write(env%unit,'(9x,53(":"))')
224  write(env%unit,'(9x,"::",21x,a,21x,"::")') "SUMMARY"
225  endif
226  write(env%unit,'(9x,53(":"))')
227  write(env%unit,outfmt) "total energy ", results%e_total,"Eh "
228  if (.not.silent.and.allocated(self%solvation)) then
229  write(env%unit,outfmt) "total w/o Gsasa/hb", &
230  & results%e_total-results%g_sasa-results%g_hb-results%g_shift, "Eh "
231  endif
232  write(env%unit,outfmt) "gradient norm ", results%gnorm, "Eh/a0"
233  write(env%unit,outfmt) "HOMO-LUMO gap ", results%hl_gap, "eV "
234  if (.not.silent) then
235  if (verbose) then
236  write(env%unit,'(9x,"::",49("."),"::")')
237  write(env%unit,outfmt) "HOMO orbital eigv.", chk%wfn%emo(chk%wfn%ihomo), "eV "
238  write(env%unit,outfmt) "LUMO orbital eigv.", chk%wfn%emo(chk%wfn%ihomo+1),"eV "
239  endif
240  write(env%unit,'(9x,"::",49("."),"::")')
241  if (self%xtbData%level.eq.2) call print_gfn2_results(env%unit,results,verbose,allocated(self%solvation))
242  if (self%xtbData%level.eq.1) call print_gfn1_results(env%unit,results,verbose,allocated(self%solvation))
243  if (self%xtbData%level.eq.0) call print_gfn0_results(env%unit,results,verbose,allocated(self%solvation))
244  write(env%unit,outfmt) "add. restraining ", efix, "Eh "
245  write(env%unit,outfmt) "total charge ", sum(chk%wfn%q), "e "
246  if (verbose) then
247  write(env%unit,'(9x,"::",49("."),"::")')
248  write(env%unit,outfmt) "atomisation energy", results%e_atom, "Eh "
249  endif
250  endif
251  write(env%unit,'(9x,53(":"))')
252  write(env%unit,'(a)')
253  endif
254 
255 end subroutine singlepoint
