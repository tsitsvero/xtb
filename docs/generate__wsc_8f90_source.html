19 subroutine generate_wsc(mol,wsc)
20  use xtb_mctc_accuracy, only : wp
21  use xtb_type_molecule
22  use xtb_type_wsc
23  implicit none
25  type(tmolecule),intent(inout) :: mol
27  type(tb_wsc), intent(inout) :: wsc
28 ! ------------------------------------------------------------------------
29 ! Variables
30 ! ------------------------------------------------------------------------
31  integer :: rep(3)
32  integer :: ii,jj,ich
33  integer :: aa,bb,cc
34  integer :: c,wc
35  integer :: minpos
36  integer :: nminpos
37  real(wp) :: t(3),rw(3)
38  real(wp) :: mindist
39  real(wp) :: nmindist
41  real(wp),parameter :: tol = 0.01_wp
42  integer, allocatable,dimension(:,:) :: lattr
43  real(wp),allocatable,dimension(:) :: dist
44  logical, allocatable,dimension(:) :: trans
45 
46  where(mol%pbc)
47  rep = 1
48  elsewhere
49  rep = 0
50  endwhere
51 
52 ! ------------------------------------------------------------------------
53 ! allocate space for the WSC first
54  call wsc%allocate(mol%n,rep)
55 
56 ! ------------------------------------------------------------------------
57 ! initialize
58 ! ------------------------------------------------------------------------
59  allocate( lattr(3,wsc%cells), source = 0 )
60  allocate( dist(wsc%cells), source = 0.0_wp )
61  allocate( trans(wsc%cells), source = .true. )
62 ! ------------------------------------------------------------------------
63 ! Create the Wigner-Seitz Cell (WSC)
64 ! ------------------------------------------------------------------------
65  wsc%at = 0
66  wsc%itbl= 0
67 !$omp parallel default(none) &
68 !$omp private(ii,jj,wc,c,dist,trans,t,lattr,rw) &
69 !$omp shared(mol,wsc,rep) &
70 !$omp shared(mindist,minpos,nmindist,nminpos)
71 !$omp do schedule(dynamic)
72  ! Each WSC of one atom consists of n atoms
73  do ii=1,mol%n
74  do jj=1,mol%n
75  !if (ii.eq.jj) cycle
76  ! find according neighbours
77  c=0
78  dist = 0.0_wp
79  lattr = 0
80  do aa=-rep(1),rep(1),1
81  do bb=-rep(2),rep(2),1
82  do cc=-rep(3),rep(3),1
83  if ((aa.eq.0 .and. bb.eq.0 .and. cc.eq.0).and.ii.eq.jj) cycle
84  t = [aa,bb,cc]
85  c=c+1
86  lattr(:,c) = [aa,bb,cc]
87  rw = mol%xyz(:,jj) + matmul(mol%lattice,t)
88  dist(c)=sqrt(sum((mol%xyz(:,ii)-rw)**2))
89  end do
90  end do
91  end do
92  ! sanity check; otherwise code below crashes sometimes
93  if (c .eq. 0) cycle
94  ! get first image with same dist
95  ! find minimum in dist-array and assign it to minpos = minimum position
96  trans=.true.
97  minpos=minloc(dist(:c),dim=1)
98  ! minimum distance saved in mindist
99  mindist=dist(minpos)
100  trans(minpos)=.false.
101  wc=1
102  wsc%lattr(:,wc,jj,ii)=lattr(:,minpos)
103  ! get other images with same distance
104  if (c > 1) then
105  find_images : do
106  nminpos=minloc(dist(:c),dim=1,mask=trans(:c))
107  nmindist=dist(nminpos)
108  if(abs(mindist-nmindist).lt.tol)then
109  trans(nminpos)=.false.
110  wc=wc+1
111  wsc%lattr(:,wc,jj,ii)=lattr(:,nminpos)
112  else
113  wsc%w(jj,ii)=1.0_wp/real(wc,wp)
114  wsc%itbl(jj,ii)=wc
115  wsc%at(jj,ii)=jj
116  exit find_images
117  end if
118  end do find_images
119  else
120  wsc%w(jj,ii) = 1.0_wp
121  wsc%itbl(jj,ii) = 1
122  wsc%at(jj,ii)=jj
123  endif
124  end do
125  end do
126 !$omp enddo
127 !$omp endparallel
128 
129 end subroutine generate_wsc
