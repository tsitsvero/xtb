19 ! module xtb_lineardep
20 !
21 ! Contains some routines related to problems that can arise due to
22 ! linear dependencies in the basis set, i.e., if any of the eigenvalues
23 ! of the diagonalized overlap matrix approach zero.
24 ! With these routines we can get cut-off eigenvalues of the overlap that are too small
25 ! and generate a transformation matrix X. With this matrix X we
26 ! can then transform our Hamiltionian to H'=X**(T)*H*X, which can then be diagonalized
27 ! without need for the Overlap.
28 !
29 ! For more information see Chapter 3.4.5 "Orthogonalization of the Basis" in Szabo/Ostlund
30 !
31 ! P.Pracht, May 2019
32 !----------------------------------------------------------------------------------------------

45 ! subroutine cholesky
46 !
47 ! Try to perform a Cholesky-factorization A = U**T*U, where
48 ! A has to be a symmetric positive definite matrix and U is
49 ! the normalized upper triangular matrix.
50 ! The Cholesky decomposition is used here ONLY to test if
51 ! A is positive definite.
52 ! This check is usefull to pre-test if the overlap will
53 ! have near linear dependencies, and is faster than checking the
54 ! eigenvalues of the overlap.
55 !
56 ! INPUT - ndim dimension of the matrix A
57 ! - A symmetric matrix with the dimension A(ndim,ndim)
58 ! A is not modified and will be the same on In- and Output
59 !
60 ! OUTPUT - fail logical to indicate whether the Cholesky-factorization
61 ! was successful
62 !
63 !----------------------------------------------------------------------------------------------
64 subroutine cholesky(iunit,printl,ndim,A,fail)
65  implicit none
66  !---
67  integer,intent(in) :: iunit

70  logical,intent(out) :: fail
71  logical,intent(in) :: printl
72  !---
73  integer :: ierr
74  real(wp),allocatable :: U(:,:)
75 
76  if(printl) &
77  & write(iunit,'(2x,a)',advance='no')'Checking positiv definite overlap ...'
78 
79  !---
80  allocate(u(ndim,ndim))
81  fail=.false.
82  u=a
83  call dpotrf('U',ndim,u,ndim,ierr) !LAPACK routine A = U**T*U
84  !on input U=A, retruns U
85 
86  if(printl) then
87  write(iunit,'(2x,a)')'done.'
88  write(iunit,*)
89  endif
90 
91  if(ierr /= 0) then
92  fail=.true.
93  if(printl)then
94  write(iunit,'(a)') '**** WARNING ****'
95  write(iunit,'(a)') ' Cholesky factorization of the overlap failed!'
96  write(iunit,'(a)') ' It is possible that there are near linear dependecies in the basis.'
97  write(iunit,'(a)') ' Therefore PEEQ will work with an orthogonal basis by using'
98  write(iunit,'(a)') ' canonical orthogonalization.'
99  write(iunit,'(a)') '**** WARNING ****'
100  write(iunit,'(a)')
101  endif
102  endif

140 ! subroutine canorthog
141 !
142 ! In order to get the orthogonal basis we need a transformation
143 ! matrix X. In practice, this X can be modified to circumvent
144 ! near linear dependencies. This can hoever only be done if the
145 ! so-called 'canonical orthogonalization' is used, i.e.,
146 ! X is initially determined as X = U*s**(-1/2), where
147 ! U is a unitary matix and s**(-1/2) is the inverse square root
148 ! of the diagonalized overlap matrix S.
149 ! As a sanity check, X**T*S*X = 1 has to be valid.
150 ! See Szabo/Ostlund, equation 3.169 and following.
151 ! This routine determines X = U*s**(-1/2) from the overlap S.
152 !
153 ! INPUT - ndim dimension of the matrices S and X
154 ! - S overlap matrix with the dimension S(ndim,ndim)
155 !
156 ! OUTPUT - X transformation matrix X, with the dimension X(ndim,newdim)
157 ! - newdim new dimension of trafo X
158 ! - fail logical to indicate whether the setup of matrix X
159 ! was successful
160 !
161 !----------------------------------------------------------------------------------------------
162 subroutine canorthog(iunit,ndim,S,X,newdim,printl,fail)
163  implicit none
164  !---
165  integer,intent(in) :: iunit
166  integer,intent(in) :: ndim
167  real(wp),intent(in) :: S(ndim,ndim)
168  real(wp),intent(out) :: X(ndim,ndim)
169  logical,intent(out) :: fail
170  integer,intent(out) :: newdim
171  logical,intent(in) :: printl
172  !---
173  integer :: i
174  real(wp),allocatable :: U(:,:)
175  real(wp),allocatable :: ssq(:)
176  real(wp),allocatable :: sisq(:)
177  real(wp),allocatable :: P(:,:),P2(:,:)
178  !---
179  allocate(u(ndim,ndim),ssq(ndim),sisq(ndim))
180  fail=.false.
181  x=0.0d0
182  !--- diagonalize overlap and get eigenvectors U and eigenvalues s
183  if(printl) &
184  & write(iunit,'(2x,a)',advance='no')'Diagonalization of the Overlap ...'
185  call canorthog2(ndim,s,u,ssq,fail)
186  if(fail)then
187  if(printl) write(iunit,'(2x,a)')'failed.'
188  return
189  else
190  if(printl) write(iunit,'(2x,a)')'done.'
191  endif
192 
193  !--- restructure U and ssq
194  call sorteigen(ndim,ssq,u)
195 
196  !--- cut-off small eigenvalues and build the inverse square root s**(-1/2)
197  newdim=ndim
198  call lidepcut(iunit,ndim,ssq,u,sisq,newdim,printl)
199  !call prmat(6,e,nbf,1,"eigenvalues")
200  !call prmat(6,U,ndim,ndim,"U")
201  !--- sanity check stuff
202  !newdim=ndim
203  !do i=1,ndim
204  ! sisq(i)=1.0d0 / sqrt(ssq(i))
205  !enddo
206 
207  !--- get X and its new dimensions
208  if(printl) &
209  & write(iunit,'(2x,a)',advance='no')'Building transformation matrix X ...'
210  call buildtrafox(ndim,ssq,u,sisq,newdim,x)
211  !call prmat(6,X,ndim,ndim,"X")
212  if(printl) then
213  write(iunit,'(2x,a)')'done.'
214  write(iunit,'(a)')
215  endif
216  !--- more sanity check stuff
217  !call prmat(6,X,ndim,ndim,"Trafo X")
218  !allocate(P(ndim,ndim),P2(ndim,ndim))
219  !call gemm('N','N',ndim,newdim,ndim,1.0d0,S,ndim,X,ndim,0.0d0,P,ndim)
220  !call gemm('T','N',ndim,ndim,newdim,1.0d0,X,ndim,P,ndim,0.0d0,P2,ndim)
221  !call prmat(6,P2,ndim,ndim,"X**T*S*X")
222  !deallocate(P2,P)
223  !stop
224 
225  deallocate(sisq,ssq,u)
226  return
227 end subroutine canorthog

230 ! subroutine canorthog2
231 !
232 ! Diagonalizes the overlap matrix S and returns the unitary transformation
233 ! matrix U and the eigenvalues s, i.e., U**T*S*U = s is performed.
234 !
235 !
236 ! INPUT - ndim dimension of the matrices S and X
237 ! - S overlap matrix with the dimension S(ndim,ndim)
238 !
239 ! OUTPUT - X transformation matrix U, with the dimension U(ndim,ndim)
240 ! - ssq eigenvalues s of matrix S
241 ! - fail logical to indicate whether the setup of U and
242 ! was successful
243 !
244 !----------------------------------------------------------------------------------------------
245 subroutine canorthog2(ndim,S,U,ssq,fail)
246  implicit none
247  !---
248  integer,intent(in) :: ndim
249  real(wp),intent(in) :: S(ndim,ndim)
250  real(wp),intent(out) :: U(ndim,ndim)
251  real(wp),intent(out) :: ssq(ndim)
252  logical,intent(out) :: fail
253  !---
254  integer :: ierr
255  integer :: lwork
256  real(wp),dimension(1) :: wdim
257  real(wp),allocatable :: work(:)
258 
259  !---
260  fail=.false.
261  u=s
262  ssq=0.0d0
263  !--- determine optimal work space
264  !--- only calculates the optimal size of the 'work' array
265  call lapack_syev('v','l',ndim,u,ndim,ssq,wdim,-1,ierr)
266  if(ierr /= 0)then
267  fail=.true.
268  return
269  endif
270  lwork=int(wdim(1))
271  allocate(work(lwork))
272 
273  !--- diagonalize overlap s = U**T*S*U
274  call lapack_syev('v','l',ndim,u,ndim,ssq,work,lwork,ierr)
275  if(ierr /= 0)then
276  fail=.true.
277  return
278  endif
279  !!write(iunit,*) ssq
280  return
281 end subroutine canorthog2

285 ! subroutine sorteigen
286 !
287 ! We are allowed to order the eigenvalues in any way in the diagonal matrix s,
288 ! provided we order the columns of U in the same way.
289 ! Sort the overlap eigenvalues form highest to lowest and also
290 ! sort the eigenvectors accordingly.
291 ! s is typically structured in ascending order from DSYEV,
292 ! hence, this order only has to be reversed.
293 !
294 !
295 ! INPUT - ndim dimension of eigenvalue array s and matrix U
296 !
297 ! IN/OUTPUT - s eigenvalues of the overlap with the dimension s(ndim)
298 ! - U eigenvectors of the overlap with the dimension U(ndim,ndim)
299 !
300 !----------------------------------------------------------------------------------------------
301 subroutine sorteigen(ndim,s,U)
302  implicit none
303  !---
304  integer,intent(in) :: ndim
305  real(wp),intent(inout) :: s(ndim)
306  real(wp),intent(inout) :: U(ndim,ndim)
307  !---
308  integer :: i,j,k,l,imax
309  real(wp) :: smax
310  real(wp) :: Umax
311  !---
312  imax=ndim-1
313  do i=1,imax
314  smax=s(i)
315  k=i
316  l=i+1
317  do j=l,ndim
318  if(s(j).le.smax) cycle
319  k=j
320  smax=s(j)
321  enddo
322  umax=s(i)
323  s(i)=s(k)
324  s(k)=umax
325  do j=1,ndim
326  umax=u(j,i)
327  u(j,i)=u(j,k)
328  u(j,k)=umax
329  enddo
330  enddo
331  !---
332  return
333 end subroutine sorteigen

338 ! subroutine lidepcut
339 !
340 ! Cut-off small eigenvalues, which can be necessary to prevent linear dependencies
341 !
342 !
343 ! INPUT - ndim dimension of eigenvalue array s and matrix U
344 ! - printl print some output?
345 !
346 ! IN/OUTPUT - s eigenvalues of the overlap with the dimension s(ndim)
347 ! on OUTPUT this is s**(1/2)
348 ! - U eigenvectors of the overlap with the dimension U(ndim,ndim)
349 !
350 ! OUTPUT - sisq array that contains inverse square root of the eigenvalues s**(-1/2)
351 !
352 !----------------------------------------------------------------------------------------------
353 subroutine lidepcut(iunit,ndim,s,U,sisq,newdim,printl)
354  implicit none
355  !---
356  integer,intent(in) :: iunit
357  integer,intent(in) :: ndim
358  real(wp),intent(inout) :: s(ndim)
359  real(wp),intent(inout) :: U(ndim,ndim)
360  real(wp),intent(out) :: sisq(ndim)
361  integer,intent(out) :: newdim
362  logical,intent(in) :: printl
363  !---
364  integer :: i,j,imax
365  real(wp) :: eigen
366  real(wp) :: ehig,elow
367  !---
368  !lidethr=1.0d-4
369 
370  sisq=0.0d0
371  newdim=ndim
372  !--- min and max eigenvalues of the overlap
373  ehig = s(1)
374  elow = s(ndim)
375  !--- cut eigenvalues and eigenvectors that are below the predefined threshold
376  if(printl) &
377  & write(iunit,'(2x,a)',advance='no')'Cutting off small eigenvalues ...'
378  do i=1,ndim
379  eigen = s(i)
380  if(eigen > lidethr) then
381  eigen = sqrt(eigen)
382  s(i) = eigen
383  sisq(i) = 1.0d0 / eigen
384  else
385  s(i) = 0.0d0
386  sisq(i) = 0.0d0
387  u(1:ndim,i) = 0.0d0
388  newdim=newdim-1
389  endif
390  enddo
391  if(printl)then
392  write(iunit,'(2x,a)')'done.'
393  !--- some printout
394  write(iunit,'(2x,a)')'Maximum eigenvalues of the overlap:'
395  write(iunit,'(4x,a,f10.4)')'Largest eigenvalue : ',ehig
396  write(iunit,'(4x,a,f10.4)')'Smallest eigenvalue : ',elow
397  write(iunit,'(2x,a,e14.4)')'Eigenvalue cut-off threshold : ',lidethr
398  write(iunit,'(2x,a,i6)') 'Initial number of eigenvectors : ',ndim
399  write(iunit,'(2x,a,i6)') 'Removed eigenvectors : ',ndim-newdim
400  write(iunit,'(2x,a,i6)') 'Number of remaining eigenvectors : ',newdim
401  write(iunit,'(4x,a,f10.4)')'New smallest eigenvalue : ',s(newdim)**2
402  endif
403  !---
404  return
405 end subroutine lidepcut

448 ! subroutine orthgsolve
449 !
450 ! Modification of the solv routine.
451 ! Diagonalization of H is done with the trafo matrix X = U*s**(-1/2), i.e.,
452 ! in the orthogonal basis.
453 ! See Szabo/Ostlund equation 3.176:
454 ! (X**T*F*X)*C' = (X**T*S*X)*C'*e
455 ! which is possible since X**T*S*X=1
456 !
457 ! INPUT - full some logical to indicate was should be done in this routine
458 ! - ndim original dimension of the matrices H,S,X,P and vector e
459 ! - cutdim new second dimension of the rectangular trafo matrix X
460 ! - ihomo
461 ! - acc accuracy setting
462 ! - H Hamiltonian matrix
463 ! - X trafo matrix with dimensions X(ndim,cutdim)
464 ! - S overlap matrix with the dimension S(ndim,ndim)
465 ! - e eigenvalues
466 !
467 !
468 ! OUTPUT - fail logical to indicate whether the diagonalization
469 ! was successful
470 ! -
471 !
472 !----------------------------------------------------------------------------------------------
473 subroutine orthgsolve(full,ndim,cutdim,ihomo,acc,H,S,X,P,e,fail)
474  implicit none
475  integer, intent(in) :: ndim
476  integer, intent(in) :: cutdim
477  logical, intent(in) :: full
478  integer, intent(in) :: ihomo
479  real(wp),intent(inout):: H(ndim,ndim)
480  real(wp),intent(in) :: S(ndim,ndim)
481  real(wp),intent(inout):: X(ndim,ndim)
482  real(wp),intent(out) :: P(ndim,ndim)
483  real(wp),intent(out) :: e(ndim)
484  real(wp),intent(in) :: acc
485  logical, intent(out) :: fail
486 
487  integer :: i,j,info,lwork,liwork,nfound,iu
488  integer :: nbf,xbf
489  integer, allocatable :: iwork(:),ifail(:)
490  real(wp),allocatable :: aux (:)
491  real(wp),allocatable :: Haux(:,:),Paux(:,:)
492  fail =.false.
493 
494 
495  !--- DIAG IN ORTHOGONAL BASIS WITH X=U*s^-1/2 TRAFO
496  nbf = ndim
497  xbf = cutdim
498  lwork = 1 + 6*nbf + 2*nbf**2
499  allocate (aux(lwork))
500  !--- calculate F'=X**T*F*X
501 
502  call blas_gemm('n','n',nbf,xbf,nbf,1.0d0,h,nbf,x,nbf,0.0d0,p,nbf)
503  call blas_gemm('t','n',xbf,xbf,nbf,1.0d0,x,nbf,p,nbf,0.0d0,h,xbf)
504 
505  !--- Caluclate C' and ε from F'*C' = C'*ε
506  call lapack_syev('v','u',xbf,h,xbf,e,aux,lwork,info)
507  if(info.ne.0)then
508  fail=.true.
509  return
510  endif
511  !--- calculate C = X*C'
512  p=0.0d0
513  call blas_gemm('N','N',nbf,xbf,xbf,1.0d0,x,nbf,h,xbf,0.0d0,p,nbf)
514  h=p
515  deallocate(aux)
516  return
517 end subroutine orthgsolve
