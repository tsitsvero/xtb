170 
171  bonds = 0
172 
173  do i = 1, n
174  bonds(i,i) = ceiling(cn(i))
175  do j = 1, i-1
176  rij = xyz(:,i) - xyz(:,j)
177  r2 = sum(rij**2)
178  if (r2.gt.200.0_wp) cycle
179  ati = at(i)
180  atj = at(j)
181  ir = itr(ati)
182  jr = itr(atj)
183  ra=r0(ati)+cnfak(ati)*cn(i)+shift
184  rb=r0(atj)+cnfak(atj)*cn(j)+shift
185  den=abs(en(ati) - en(atj))
186  k1=0.005_wp*(p(ir,1) + p(jr,1))
187  k2=0.005_wp*(p(ir,2) + p(jr,2))
188  ff=1.0_wp - k1*den - k2*den**2
189  ! save distances => rab
190  rab = ((ra + rb)*ff)**2
191  if (r2 < rab) then
192  bonds(i,j) = 1
193  bonds(j,i) = 1
194  endif
195  enddo ! k
