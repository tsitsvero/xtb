124  integer :: ig
125  real(wp) :: rik2, rik(3), dtmp, expterm, arg
126  real(wp) :: fqpc, falp, ds(3, 3)
127  real(wp), parameter :: unity(3, 3) = reshape(&
128  & [1.0_wp, 0.0_wp, 0.0_wp, 0.0_wp, 1.0_wp, 0.0_wp, 0.0_wp, 0.0_wp, 1.0_wp], &
129  & [3, 3])
130 
131  damat = 0.0_wp
132  sigma = 0.0_wp
133  fqpc = 2*qpc**2
134  falp = 2.0_wp/3.0_wp/(4.0_wp*alpha**2)
135  do ig = 1, size(gtrans, dim=2)
136  rik = gtrans(:, ig)
137  rik2 = dot_product(rik,rik)
138  expterm = exp(-rik2/(4.0_wp*alpha**2))/rik2
139  arg = dot_product(rik,vec)
140  dtmp = -sin(arg) * expterm
141  damat = damat + rik*dtmp
142  ds = spread(rik,1,3)*spread(rik,2,3)
143  sigma = sigma + expterm * cos(arg) * ( &
144  & - unity * (1.0_wp + rik2*falp + rik2*fqpc) &
145  & + (2.0_wp/rik2 + 0.5_wp/alpha**2 + 0.5_wp*fqpc) * ds)
146  end do
147  damat = damat * 4.0_wp*pi/volume * scale
148  sigma = sigma * 4.0_wp*pi/volume * scale

234  real(wp), parameter :: alphainit = 1.0e-8_wp
235 
236  real(wp) :: ming, minr, diff
237  integer :: iiter
238  integer :: ierror
239  character(len=100) :: errorstring
240 
241  ming = sqrt(minval(sum(recvec(:,:)**2, dim=1)))
242  minr = sqrt(minval(sum(latvec(:,:)**2, dim=1)))
243 
244  ierror = 0
245  alpha = alphainit
246  diff = diffrecreal(alpha, getgtermpbc3d, ming, minr, volume)
247  do while (diff < -tolerance .and. alpha <= huge(1.0_wp))
248  alpha = 2.0_wp * alpha
249  diff = diffrecreal(alpha, getgtermpbc3d, ming, minr, volume)
250  end do
251  if (alpha > huge(1.0_wp)) then
252  ierror = 1
253  elseif (alpha == alphainit) then
254  ierror = 2
255  end if
256 
257  if (ierror == 0) then
258  alphaleft = 0.5_wp * alpha
259  do while (diff < tolerance .and. alpha <= huge(1.0_wp))
260  alpha = 2.0_wp * alpha
261  diff = diffrecreal(alpha, getgtermpbc3d, ming, minr, volume)
262  end do
263  if (alpha > huge(1.0_wp)) then
264  ierror = 3
265  end if
266  end if
267 
268  if (ierror == 0) then
269  alpharight = alpha
270  alpha = (alphaleft + alpharight) / 2.0
271  iiter = 0
272  diff = diffrecreal(alpha, getgtermpbc3d, ming, minr, volume)
273  do while (abs(diff) > tolerance .and. iiter <= nsearchiter)
274  if (diff < 0) then
275  alphaleft = alpha
276  else
277  alpharight = alpha
278  end if
279  alpha = (alphaleft + alpharight) / 2.0
280  diff = diffrecreal(alpha, getgtermpbc3d, ming, minr, volume)
281  iiter = iiter + 1
282  end do
283  if (iiter > nsearchiter) then
284  ierror = 4
285  end if
286  end if
287 
288  if (ierror /= 0) then
289  !alpha = exp(-0.310104 * log(volume) + 0.786382) / 2.0
290  call env%error('Failed to determine the optimal alpha for Ewald sum', source)
291  end if

293 end subroutine getoptimalalpha
