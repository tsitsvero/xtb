35 subroutine readmoleculepdb(mol, unit, status, iomsg)

52 ! ATOM 2461 HA3 GLY A 153 -10.977 -7.661 2.011 1.00 0.00 H
53 ! TER 2462 GLY A 153
54 ! a6----i5---xa4--aa3-xai4--axxxf8.3----f8.3----f8.3----f6.2--f6.2--xxxxxxa4--a2a2
55 ! HETATM 2463 CHA HEM A 154 9.596 -13.100 10.368 1.00 0.00 C
56  character(len=*), parameter :: pdb_format = &
57  & '(6x,i5,1x,a4,a1,a3,1x,a1,i4,a1,3x,3f8.3,2f6.2,6x,a4,a2,a2)'
58  integer, parameter :: p_initial_size = 1000 ! this is going to be a proteine
59 
60  status = .false.
61 
62  allocate(list(p_initial_size), source=0)
63  allocate(sym(p_initial_size), source=' ')
64  allocate(xyz(3, p_initial_size), source=0.0_wp)
65  allocate(pdb(p_initial_size), source=pdb_data())
66 
67  iatom = 0
68  iresidue = 0
69 
70  error = 0
71  do while(error == 0)
72  call getline(unit, line, error)
73  if (index(line, 'END') == 1) exit
74  if (index(line, 'ATOM') == 1 .or. index(line, 'HETATM') == 1) then
75  if (iatom >= size(xyz, 2)) call resize(xyz)
76  if (iatom >= size(sym)) call resize(sym)
77  if (iatom >= size(list)) call resize(list)
78  if (iatom >= size(pdb)) call resize(pdb)
79  iatom = iatom + 1
80  pdb(iatom)%het = index(line, 'HETATM') == 1
81  read(line, pdb_format) &
82  & jatom, pdb(iatom)%name, pdb(iatom)%loc, pdb(iatom)%residue, &
83  & pdb(iatom)%chains, pdb(iatom)%residue_number, pdb(iatom)%code, &
84  & coords, occ, temp, pdb(iatom)%segid, sym(iatom), a_charge
85  xyz(:,iatom) = coords * aatoau
86  atom_type = tonumber(sym(iatom))
87  if (atom_type == 0) then
88  try = scan(pdb(iatom)%name, 'HCNOSPF')
89  if (try > 0) sym(iatom) = pdb(iatom)%name(try:try)//' '
90  endif
91  !read(a_charge, *, iostat=try) pdb(iatom)%charge
92  call pdbchrg2int(a_charge,pdb(iatom)%charge)
93  !if (try /= 0) pdb(iatom)%charge = 0
94  endif
95  enddo
96  if (error /= 0) then
97  iomsg = "could not read in coordinates, last line was: '"//line//"'"
98  return
99  endif
100  j=0
101  associate( rn => pdb(:iatom)%residue_number)
102  do i = minval(rn),maxval(rn)
103  if (any(i .eq. rn)) then
104  j=j+1
105  where (i .eq. rn) list = j
106  endif
107  enddo
108  end associate
109 
110  call init(mol, sym(:iatom), xyz(:, :iatom))
111  call mol%frag%allocate(list(:iatom))
112  mol%pdb = pdb(:iatom)
113  mol%chrg = sum(mol%pdb%charge)
114 
115  if (.not.all(mol%at > 0)) then
116  iomsg = "invalid atom type found"
117  return
118  endif
119 
120  ! since PDB is used for biomolecules, this is a sensible check (prevents GIGO)
121  if (.not.any(mol%at == 1)) then
122  iomsg = "You get no calculation today, please add hydrogen atoms first"
123  return
124  endif
125 
126  status = .true.
127 
128 end subroutine readmoleculepdb
