368 subroutine writeinfo(self, unit, num)
369 
371  class(txtbdata), intent(in) :: self
372 
374  integer, intent(in) :: unit
375 
377  integer, intent(in), optional :: num(:)
378 
379  character(len=*), parameter :: rnum = '(f12.6)'
380  character(len=*), parameter :: offset = '(8x)'
381  character(len=*), parameter :: head = '(6x,"*",1x,a,":")'
382  character(len=*), parameter :: rfmt = '('//offset//',a,t36,'//rnum//')'
383  character(len=*), parameter :: afmt = '('//offset//',a,t36,4x,a)'
384  character(len=:), allocatable :: name
385  integer :: ii, jj
386 
387  write(unit, '(a)')
388  if (allocated(self%name)) then
389  allocate(character(len=2*len(self%name)-1) :: name)
390  name = repeat(' ', len(name))
391  do ii = 1, len(self%name)
392  jj = 2*ii-1
393  name(jj:jj) = self%name(ii:ii)
394  end do
395  else
396  name = repeat(' ', 10)
397  write(name, '(i0)') self%level
398  name = 'xTB level '//trim(name)
399  end if
400  call generic_header(unit, name, 49, 10)
401 
402  write(unit, '(a)')
403  if (allocated(self%doi)) then
404  write(unit, afmt) "Reference", self%doi
405  end if
406 
407  write(unit, head) "Hamiltonian"
408  write(unit, rfmt, advance='no') "H0-scaling (s, p, d)"
409  do ii = 0, 2
410  write(unit, rnum, advance='no') self%hamiltonian%kScale(ii, ii)
411  end do
412  write(unit, '(a)')
413  write(unit, rfmt) "zeta-weighting", self%hamiltonian%wExp
414 
415  write(unit, head) "Dispersion"
416  write(unit, rfmt) "s8", self%dispersion%dpar%s8
417  write(unit, rfmt) "a1", self%dispersion%dpar%a1
418  write(unit, rfmt) "a2", self%dispersion%dpar%a2
419  write(unit, rfmt) "s9", self%dispersion%dpar%s9
420 
421  write(unit, head) "Repulsion"
422  write(unit, rfmt, advance='no') "kExp", self%repulsion%kExp
423  if (self%repulsion%kExpLight /= self%repulsion%kExp) then
424  write(unit, rnum, advance='no') self%repulsion%kExpLight
425  end if
426  write(unit, '(a)')
427  write(unit, rfmt) "rExp", self%repulsion%rExp
428 
429  write(unit, head) "Coulomb"
430  write(unit, rfmt) "alpha", self%coulomb%gExp
431  if (allocated(self%coulomb%thirdOrderShell)) then
432  write(unit, afmt) "third order", "shell-resolved"
433  else if (allocated(self%coulomb%thirdOrderAtom)) then
434  write(unit, afmt) "third order", "atomic"
435  else
436  write(unit, afmt) "third order", "false"
437  end if
438 
439  if (allocated(self%multipole)) then
440  write(unit, afmt) "anisotropic", "true"
441  write(unit, rfmt) "a3", self%multipole%dipDamp
442  write(unit, rfmt) "a5", self%multipole%quadDamp
443  write(unit, rfmt) "cn-shift", self%multipole%cnShift
444  write(unit, rfmt) "cn-exp", self%multipole%cnExp
445  write(unit, rfmt) "max-rad", self%multipole%cnRMax
446  else
447  write(unit, afmt) "anisotropic", "false"
448  end if
449 
450  if (allocated(self%halogen)) then
451  write(unit, head) "Halogen bond correction"
452  write(unit, rfmt) "rad-scale", self%halogen%radScale
453  write(unit, rfmt) "damping", self%halogen%dampingPar
454  end if
455 
456  if (allocated(self%srb)) then
457  write(unit, head) "Polar bond correction"
458  write(unit, rfmt) "rad-shift", self%srb%shift
459  write(unit, rfmt) "strength", self%srb%prefactor
460  write(unit, rfmt) "en-exp", self%srb%steepness
461  write(unit, rfmt) "en-scale", self%srb%enScale
462  end if
463  write(unit, '(a)')
464 
465 end subroutine writeinfo
