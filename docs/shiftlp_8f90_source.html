18 ! shift the LP=protonation position to an "empty" region
19 subroutine shiftlp(n,at,ia1,xyz,ex,ey,ez)
20  use xtb_mctc_accuracy, only : wp
21  implicit none
22  integer n,at(n),ia1
23  real(wp) xyz(3,n),ex,ey,ez
24 
25  real(wp) r0,r1,r2,exmin,eymin,ezmin,rmin,xsave,ysave,zsave,rmax
26  real x,f
27  integer icount,j
28 
29  r0=2.5d0
30  rmin=1.d+42
31  rmax=0
32  icount=0
33 
34  10 f=1.0
35  call random_number(x)
36  if(x.lt.0.5)f=-1.0
37  call random_number(x)
38  ex=xyz(1,ia1)+x*f*r0
39  call random_number(x)
40  ey=xyz(2,ia1)+x*f*r0
41  call random_number(x)
42  ez=xyz(3,ia1)+x*f*r0
43  r1=sqrt((xyz(1,ia1)-ex)**2+(xyz(2,ia1)-ey)**2+(xyz(3,ia1)-ez)**2)
44  if(abs(r0-r1).gt.0.1) goto 10
45  rmin=1.d+42
46  do j=1,n
47  r2=sqrt((xyz(1,j)-ex)**2+(xyz(2,j)-ey)**2+(xyz(3,j)-ez)**2)
48  if(r2.lt.rmin.and.j.ne.ia1) then
49  rmin=r2
50  exmin=ex
51  eymin=ey
52  ezmin=ez
53  endif
54  enddo
55  if(rmin.gt.rmax)then
56  rmax=rmin
57  xsave=exmin
58  ysave=eymin
59  zsave=ezmin
60  endif
61  icount=icount+1
62  if(icount.lt.100) goto 10
63 
64  ex=xsave
65  ey=ysave
66  ez=zsave
67 
68 end subroutine
