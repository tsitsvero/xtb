100 subroutine resize_pdb_data(var, n)
101  type(pdb_data), allocatable, intent(inout) :: var(:)
102  integer, intent(in), optional :: n
103  type(pdb_data), allocatable :: tmp(:)
104  integer :: length, current_length
105  current_length = size(var)
106  if (current_length > 0) then
107  if (present(n)) then
108  if (n <= current_length) return
109  length = n
110  else
111  length = current_length + current_length/2 + 1
112  endif
113  allocate(tmp(length), source=pdb_data())
114  tmp(:current_length) = var(:current_length)
115  deallocate(var)
116  call move_alloc(tmp, var)
117  else
118  if (present(n)) then
119  length = n
120  else
121  length = 64
122  endif
123  allocate(var(length), source=pdb_data())
124  endif
125 end subroutine resize_pdb_data
