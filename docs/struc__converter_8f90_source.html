72 ! loop geoopt -----------------------------------------------------------------

75  real(wp), allocatable :: shift(:)

95 !===============================
96 ! Set Block
97  time_in = time_md

99  temp_in = temp_md

111  else

113  endif
114  opt_logfile = 'convert.log'
115 !------------------------------------------------------------------------------
116 ! force field geometry optimization
117  ! loop runs 3 geoopt with different shifts in the new 3rd coordinate
118  ! and then keeps the mol%xyz with lowest etot for md
119  mol_shifted = mol
120  do i=1, num_shift_runs
121  if (i.ne.42) then ! if in case you want an opt with the 2D init -> e.g. use if (i.ne.1)

123  ! create array with random shifts: 0<= shift <1
124  call random_number(shift)
125  ! change signs of shifts randomly
126  do j=1, size(shift)

129  enddo

131  endif
132 
133  call geometry_optimization &

135  & egap,etemp,maxiter,maxcycle,etot,g,sigma,p_olev_crude,.false.,.true.,fail)
136  mol_xyz_arr(:,:,i) = mol_shifted%xyz ! store optimized xyz
137  etot_arr(i) = etot ! store energy etot
138  enddo
139 
140  mol%xyz(:,:) = mol_xyz_arr(:,:,minloc(etot_arr, dim=1)) ! keep xyz with lowest etot
141 
142  if (allocated(fnv)) then
143  opt_logfile = fnv
144  else
145  deallocate(opt_logfile)
146  endif
147  write(*,*)
148 !------------------------------------------------------------------------------
149 ! force field md simulation
150  idum = 0
151  call md &
152  & (env,mol,chk,calc, &
153  & egap,etemp,maxiter,etot,g,sigma,0,temp_md,idum)
154 !------------------------------------------------------------------------------
155 ! set all back to input
156  time_md = time_in
157  temp_md = temp_in
158  tstep_md = step_in
159  dump_md2 = dump_in
160  md_hmass = hmass_in
161  if (allocated(opt_in)) then
162  opt_engine = opt_in
163  else
164  deallocate(opt_engine)
165  end if
166 !------------------------------------------------------------------------------
167  write(*,*)
168  write(*,'(10x," ------------------------------------------------- ")')
169  write(*,'(10x,"| 2D => 3D conversion done! |")')
170  write(*,'(10x," ------------------------------------------------- ")')
171  write(*,*)
172  mode_extrun = mode_input
173  mol%struc%two_dimensional=.false.
174  call gfnff_param_dealloc(calc%topo)
175
