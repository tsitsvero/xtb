102  lcsymbol = ' '
103 
104  k = 0
105  do j = 1, len_trim(symbol)
106  if (k > 2) exit
107  l = iachar(symbol(j:j))
108  if (k >= 1 .and. l == iachar(' ')) exit
109  if (k >= 1 .and. l == 9) exit
110  if (l >= iachar('A') .and. l <= iachar('Z')) l = l + offset

115  enddo

121  endif

123
