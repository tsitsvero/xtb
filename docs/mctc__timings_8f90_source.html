47 subroutine prtiming(i,inmsg)
48  implicit none
49  integer,intent(in) :: i
50  character(len=*),intent(in),optional :: inmsg
51  character(len=:),allocatable :: msg
52  real(wp) :: cputime,walltime
53  integer(int64) :: cpudays, cpuhours, cpumins
54  integer(int64) :: walldays,wallhours,wallmins
55 ! <7-------> <5---------> <7-----------> <10------------>
56 ! '(1x,a,1x,"time:",1x,5i,1x,"d",1x,2i,1x,"h",1x,2i,1x,"min",1x,f5.2,1x,"sec")'
57 ! '(5i,1x,"d")'
58 ! '(a,1x,2i,1x,"h")'
59 ! '(1x,2i,1x,"min")'
60 ! '(1x,f5.2,1x,"sec")'
61 ! '(1x,a,1x,"time:",1x,a)'
62  if (present(inmsg)) then
63  msg = inmsg
64  else
65  msg = '*'
66  endif
67  ! DAYS HOURS MINUTES SECONDS
68  ! DAYS 1 1/24 1/1440 1/86400
69  ! HOURS 24 1 1/60 1/3600
70  ! MINUTES 1440 60 1 1/60
71  ! SECONDS 86400 3600 60 1
72  cputime = timing_cpu(i)
73 ! if (cputime.ge.86400._wp) then
74  cpudays = int(cputime/86400._wp)
75  cputime = cputime - cpudays*86400._wp
76 ! endif
77 ! if (cputime.ge.3600._wp) then
78  cpuhours = int(cputime/3600._wp)
79  cputime = cputime - cpuhours*3600._wp
80 ! endif
81 ! if (cputime.ge.60._wp) then
82  cpumins = int(cputime/60._wp)
83  cputime = cputime - cpumins*60._wp
84 ! endif
85  walltime = timing_wall(i)
86 ! if (walltime.ge.86400._wp) then
87  walldays = int(walltime/86400._wp)
88  walltime = walltime - walldays*86400._wp
89 ! endif
90 ! if (walltime.ge.3600._wp) then
91  wallhours = int(walltime/3600._wp)
92  walltime = walltime - wallhours*3600._wp
93 ! endif
94 ! if (walltime.ge.60._wp) then
95  wallmins = int(walltime/60._wp)
96  walltime = walltime - wallmins*60._wp
97 ! endif
98 
99  if (verbose) then
100  write(output_unit,'(1x,a,":")') msg
101  write(output_unit,'(" * wall-time: ",i5," d, ",i2," h, ",i2," min, ",f6.3," sec")') walldays,wallhours,wallmins,walltime
102  write(output_unit,'(" * cpu-time: ",i5," d, ",i2," h, ",i2," min, ",f6.3," sec")') cpudays, cpuhours, cpumins, cputime
103  write(output_unit,'(1x,"*",1x,"ratio c/w:",1x,f9.3,1x,"speedup")') timing_cpu(i)/timing_wall(i)
104  else
105  write(output_unit,'(1x,a,1x,"time: ",i5," d, ",i2," h, ",i2," min, ",f6.3," sec")') msg,walldays,wallhours,wallmins,walltime
106  endif
107 end subroutine prtiming
