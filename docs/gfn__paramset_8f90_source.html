88  integer i,j,ii,jj

90  real*8 kparam

93 
94  if(gfn_method.eq.1)then
95  kp(1)=1.1 ! 3d
96  kp(2)=1.2 ! 4d

99  kp(1)=1.10

103  tmgroup=(/29,47,79/)
104  do i=1,3
105  do j=1,3
106  ii=tmgroup(i)
107  jj=tmgroup(j)
108  pairparam(ii,jj)=kparam
109  pairparam(jj,ii)=kparam
110  enddo
111  enddo
112  elseif(gfn_method.gt.1)then
113  kp(1)=1. ! 3d
114  kp(2)=1. ! 4d
115  kp(3)=1. ! 5d or 4f
116  ! write(*,'(''KAB for pair M(3d)-M(3d) :'',f8.4)')kp(1)
117  ! write(*,'(''KAB for pair M(4d)-M(4d) :'',f8.4)')kp(2)
118  ! write(*,'(''KAB for pair M(5d)-M(5d) :'',f8.4)')kp(3)
119  endif
120 
121  do i=21,79
122  do j=21,i
123  ii=tmmetal(i)
124  jj=tmmetal(j)
125  ! metal-metal interaction
126  notset=abs(pairparam(i,j)-1.0d0).lt.1.d-6 .and. &
127  & abs(pairparam(j,i)-1.0d0).lt.1.d-6
128  if(ii.gt.0.and.jj.gt.0.and.notset) then
129  pairparam(i,j)=0.5*(kp(ii)+kp(jj))
130  pairparam(j,i)=0.5*(kp(ii)+kp(jj))
131  endif
132  enddo
133  enddo
134 
135 end subroutine setpair
