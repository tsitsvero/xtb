19  subroutine ispline(u, x, y, b, c, d, n, isp, isp1, isp2)
20 !======================================================================
21 ! function ispline evaluates the cubic spline interpolation at point z
22 ! ispline = y(i)+b(i)*(u-x(i))+c(i)*(u-x(i))**2+d(i)*(u-x(i))**3
23 ! where x(i) <= u <= x(i+1)
24 !----------------------------------------------------------------------
25 ! input..
26 ! u = the abscissa at which the spline is to be evaluated
27 ! x, y = the arrays of given data points
28 ! b, c, d = arrays of spline coefficients computed by spline
29 ! n = the number of data points
30 ! output:
31 ! isp = interpolated value at point u
32 ! isp1 = first derivative at point u
33 ! isp2 = second derivative at point u
34 !=======================================================================
35 implicit none
36 double precision isp,isp1,isp2
37 integer n
38 double precision u, x(n), y(n), b(n), c(n), d(n)
39 integer i, j, k
40 double precision dx
41 
42 isp1=0
43 isp2=0
44 ! if u is ouside the x() interval take a boundary value (left or right)
45 if(u <= x(1)) then
46  isp = y(1)
47  return
48 end if
49 if(u >= x(n)) then
50  isp = y(n)
51  return
52 end if
53 
54 !*
55 ! binary search for for i, such that x(i) <= u <= x(i+1)
56 !*
57 i = 1
58 j = n+1
59 do while (j > i+1)
60  k = (i+j)/2
61  if(u < x(k)) then
62  j=k
63  else
64  i=k
65  end if
66 end do
67 !*
68 ! evaluate spline interpolation and derivatives
69 !*
70 dx = u - x(i)
71 isp = y(i) + dx*(b(i) + dx*(c(i) + dx*d(i)))
72 isp1= (b(i) + dx*(c(i)*2. + 3.*dx*d(i)))
73 isp2= + (c(i)*2. + 6.*dx*d(i))
74 end

76 subroutine spline2(x, y, b, c, d, n)
77 !======================================================================
78 ! Calculate the coefficients b(i), c(i), and d(i), i=1,2,...,n
79 ! for cubic spline interpolation
80 ! s(x) = y(i) + b(i)*(x-x(i)) + c(i)*(x-x(i))**2 + d(i)*(x-x(i))**3
81 ! for x(i) <= x <= x(i+1)
82 !----------------------------------------------------------------------
83 ! input..
84 ! x = the arrays of data abscissas (in strictly increasing order)
85 ! y = the arrays of data ordinates
86 ! n = size of the arrays xi() and yi() (n>=2)
87 ! output..
88 ! b, c, d = arrays of spline coefficients
89 ! comments ...
90 ! spline.f90 program is based on fortran version of program spline.f
91 ! the accompanying function fspline can be used for interpolation
92 !======================================================================
93 implicit none
94 integer n
95 double precision x(n), y(n), b(n), c(n), d(n)
96 integer i, j, gap
97 double precision h
98 
99 gap = n-1
100 ! check input
101 if ( n < 2 ) return
102 if ( n < 3 ) then
103 b(1) = (y(2)-y(1))/(x(2)-x(1)) ! linear interpolation
104 c(1) = 0.
105 d(1) = 0.
106 b(2) = b(1)
107 c(2) = 0.
108 d(2) = 0.
109 return
110 end if
111 !
112 ! step 1: preparation
113 !
114 d(1) = x(2) - x(1)
115 c(2) = (y(2) - y(1))/d(1)
116 do i = 2, gap
117 d(i) = x(i+1) - x(i)
118 b(i) = 2.0*(d(i-1) + d(i))
119 c(i+1) = (y(i+1) - y(i))/d(i)
120 c(i) = c(i+1) - c(i)
121 end do
122 !
123 ! step 2: end conditions
124 !
125 b(1) = -d(1)
126 b(n) = -d(n-1)
127 c(1) = 0.0
128 c(n) = 0.0
129 if(n /= 3) then
130 c(1) = c(3)/(x(4)-x(2)) - c(2)/(x(3)-x(1))
131 c(n) = c(n-1)/(x(n)-x(n-2)) - c(n-2)/(x(n-1)-x(n-3))
132 c(1) = c(1)*d(1)**2/(x(4)-x(1))
133 c(n) = -c(n)*d(n-1)**2/(x(n)-x(n-3))
134 end if
135 !
136 ! step 3: forward elimination
137 !
138 do i = 2, n
139 h = d(i-1)/b(i-1)
140 b(i) = b(i) - h*d(i-1)
141 c(i) = c(i) - h*c(i-1)
142 end do
143 !
144 ! step 4: back substitution
145 !
146 c(n) = c(n)/b(n)
147 do j = 1, gap
148 i = n-j
149 c(i) = (c(i) - d(i)*c(i+1))/b(i)
150 end do
151 !
152 ! step 5: compute spline coefficients
153 !
154 b(n) = (y(n) - y(gap))/d(gap) + d(gap)*(c(gap) + 2.0*c(n))
155 do i = 1, gap
156 b(i) = (y(i+1) - y(i))/d(i) - d(i)*(c(i+1) + 2.0*c(i))
157 d(i) = (c(i+1) - c(i))/d(i)
158 c(i) = 3.*c(i)
159 end do
160 c(n) = 3.0*c(n)
161 d(n) = d(n-1)
162 end subroutine spline2
