158 subroutine constrpot(nat,at,xyz,g,e)
159  use xtb_mctc_accuracy, only : wp
160  use xtb_mctc_constants
161  use xtb_scanparam
162  use xtb_splitparam
163  implicit none
164  integer nat,at(nat)
165  real(wp) e
166  real(wp) xyz(3,nat)
167  real(wp) g(3,nat)
168 
169  integer i,j,k,l,kk
170  real(wp)rij(3),dum,r0,ffc,vp(3),ra(3),rb(3)
171  real(wp)va(3),vb(3),vc(3),vab(3),vcb(3),deda(3),dedc(3),dedb(3)
172  real(wp)dda(3),ddb(3),ddc(3),ddd(3)
173  real(wp)c0,rab2,rcb2,theta,dt,rmul1,rmul2,deddt,rp,cosa,ea,vlen,d
174  real(wp)c1,phi0,phi,dphi1,x1cos,x1sin,dij,valijkl,ff,dum1,dum2
175  real(wp)dx,dy,dz,termx,termy,termz,abcx,abcy,abcz,a,z0,expo,expom1
176 
177  if(nconstr.eq.0) return
178 
179  ffc=fcconstr
180 
181  do kk=1,nconstr
182  ! CMA distance to valconstr (usually=0) for atoms in fragment1
183  if(atconstr(1,kk).eq.-2)then
184  rb=0
185  do i=1,iatf1
186  a=atmass(i)
187  rb(1)=rb(1)+a*xyz(1,i)
188  rb(2)=rb(2)+a*xyz(2,i)
189  rb(3)=rb(3)+a*xyz(3,i)
190  enddo
191  z0=valconstr(kk)
192  ra(1)=rb(1)/massf1 - z0
193  ra(2)=rb(2)/massf1 - z0
194  ra(3)=rb(3)/massf1 - z0
195  d=ra(1)**2+ra(2)**2+ra(3)**2
196  ffc=fcconstr/dble(iatf1)
197  e=e+ffc*d
198  do i=1,iatf1
199  ff=2.0d0*ffc*atmass(i)/massf1
200  g(1,i)=g(1,i)+ra(1)*ff
201  g(2,i)=g(2,i)+ra(2)*ff
202  g(3,i)=g(3,i)+ra(3)*ff
203  enddo
204  endif
205  ! Z distance for atoms in fragment2
206  if(atconstr(1,kk).eq.-1)then
207  ffc=0.1*fcconstr/dble(iatf2)
208  z0=valconstr(kk)
209  do i=1,nat
210  if(splitlist(i).eq.2)then
211  dz=xyz(3,i)-z0
212  e=e+ffc*dz**2
213  g(3,i)=g(3,i)+2.0d0*ffc*dz
214  endif
215  enddo
216  endif
217  ! CMA distance (rcma in common block for average in MD)
218  if(atconstr(1,kk).eq.0)then
219  expo=6.0d0
220  expom1=5.0d0
221  ffc=0.1*fcconstr
222  r0=valconstr(kk)
223  call cmafrag(nat,at,xyz,ra,rb) ! rcma in common
224  d=rcma-r0
225  e=e+(1./expo)*ffc*d**expo
226  do i=1,nat
227  dx = (ra(2)-rb(2))**2+(ra(3)-rb(3))**2
228  dy = (ra(1)-rb(1))**2+(ra(3)-rb(3))**2
229  dz = (ra(2)-rb(2))**2+(ra(1)-rb(1))**2
230  if(splitlist(i).eq.1)then ! fragment 1
231  a = ffc*atmass(i)/massf1
232  abcx=ra(1)-rb(1)
233  abcy=ra(2)-rb(2)
234  abcz=ra(3)-rb(3)
235  termx=sqrt(abcx**2+dx)
236  termy=sqrt(abcy**2+dy)
237  termz=sqrt(abcz**2+dz)
238  g(1,i)=g(1,i)+a*abcx*(termx-r0)**expom1/termx
239  g(2,i)=g(2,i)+a*abcy*(termy-r0)**expom1/termy
240  g(3,i)=g(3,i)+a*abcz*(termz-r0)**expom1/termz
241  else ! fragment 2
242  a = ffc*atmass(i)/massf2
243  abcx=rb(1)-ra(1)
244  abcy=rb(2)-ra(2)
245  abcz=rb(3)-ra(3)
246  termx=sqrt(abcx**2+dx)
247  termy=sqrt(abcy**2+dy)
248  termz=sqrt(abcz**2+dz)
249  g(1,i)=g(1,i)+a*abcx*(termx-r0)**expom1/termx
250  g(2,i)=g(2,i)+a*abcy*(termy-r0)**expom1/termy
251  g(3,i)=g(3,i)+a*abcz*(termz-r0)**expom1/termz
252  endif
253  enddo
254  endif
255  ! length (rcma in common block for average in MD)
256  if(atconstr(3,kk).eq.0.and.atconstr(1,kk).ne.0&
257  &.and.atconstr(2,kk).ne.0)then
258  i= atconstr(1,kk)
259  j= atconstr(2,kk)
260  r0=valconstr(kk)
261  rij=xyz(:,j)-xyz(:,i)
262  rcma=sqrt(sum(rij*rij))
263  d=rcma-r0
264  ! e=e+ffc*d*d
265  ! ff=ffc*2.0d0*d
266  dum= d**springexpo
267  dum2=d**(springexpo-1.)
268  e=e+ffc*dum
269  ff=ffc*springexpo*dum2
270  if(kk.eq.1) pmf=ff
271  dum=ff/rcma
272  g(1,j)=g(1,j)+dum*rij(1)
273  g(1,i)=g(1,i)-dum*rij(1)
274  g(2,j)=g(2,j)+dum*rij(2)
275  g(2,i)=g(2,i)-dum*rij(2)
276  g(3,j)=g(3,j)+dum*rij(3)
277  g(3,i)=g(3,i)-dum*rij(3)
278  endif
279  ! bend
280  if (atconstr(4,kk).eq.0.and.atconstr(3,kk).gt.0) then
281  i= atconstr(1,kk)
282  j= atconstr(2,kk)
283  k= atconstr(3,kk)
284  c0 =valconstr(kk)
285  va = xyz(1:3,i)
286  vb = xyz(1:3,j)
287  vc = xyz(1:3,k)
288  vab = va-vb
289  vcb = vc-vb
290  rab2 = vab(1)*vab(1) + vab(2)*vab(2) + vab(3)*vab(3)
291  rcb2 = vcb(1)*vcb(1) + vcb(2)*vcb(2) + vcb(3)*vcb(3)
292  call crprod(vcb,vab,vp)
293  rp = norm2(vp)+1.d-14
294  call impsc(vab,vcb,cosa)
295  cosa = dble(min(1.0d0,max(-1.0d0,cosa)))
296  theta= dacos(cosa)
297  dt = theta - c0
298  ea = ffc* dt**2
299  deddt = 2.d0 * ffc * dt
300  e = e + ea
301  call crprod(vab,vp,deda)
302  rmul1 = -deddt / (rab2*rp)
303  deda = deda*rmul1
304  call crprod(vcb,vp,dedc)
305  rmul2 = deddt / (rcb2*rp)
306  dedc = dedc*rmul2
307  dedb = deda+dedc
308  g(1:3,i) = g(1:3,i) + deda(1:3)
309  g(1:3,j) = g(1:3,j) - dedb(1:3)
310  g(1:3,k) = g(1:3,k) + dedc(1:3)
311  endif
312  ! dihed
313  if(atconstr(4,kk).ne.0)then
314  i= atconstr(1,kk)
315  j= atconstr(2,kk)
316  k= atconstr(3,kk)
317  l= atconstr(4,kk)
318  phi0 =valconstr(kk)
319  phi=valijkl(nat,xyz,i,j,k,l)
320  if(abs(phi-pi).lt.1.d-8.or.abs(phi).lt.1.d-8)phi=phi+1.d-8
321  call dphidr(nat,xyz,i,j,k,l,phi,dda,ddb,ddc,ddd)
322  dphi1=phi0-phi+pi
323  x1cos=cos(dphi1)
324  x1sin=sin(dphi1)
325  dij =ffc*x1sin
326  g(1:3,i)=g(1:3,i)+dij*dda(1:3)
327  g(1:3,j)=g(1:3,j)+dij*ddb(1:3)
328  g(1:3,k)=g(1:3,k)+dij*ddc(1:3)
329  g(1:3,l)=g(1:3,l)+dij*ddd(1:3)
330  e=e+ffc*(1.+x1cos)
331  endif
332 
333  enddo
334 
335 end subroutine constrpot

424 Subroutine dthetadr(nat,xyz,j,k,i,theta,kijk,dei,dej,dek)
425  use xtb_mctc_accuracy, only : wp
426 
427  ! .....................................................................
428  !
429  ! Input
430  ! -----
431  ! __ __
432  ! ra,rb : distance ij and distnace ik
433  !
434  ! sinth : sinus of angle j-i-k
435  !
436  ! costh : cosinus of angle j-i-k
437  !
438  !
439  ! Output
440  ! ------
441  !
442  ! dei : derivatives of the atom i
443  !
444  ! dej : derivatives of the atom j
445  !
446  ! dek : derivatives of the atom k
447  !
448  ! .....................................................................
449 
450  implicit none
451 
452  external vecnorm
453 
454  integer ic,nat,j,k,i
455 
456  real(wp) :: &
457  & ran,rbn,theta,xyz(3,nat),&
458  & ra(3),rb(3),sinth,costh,&
459  & dei(3),dej(3),dek(3),kijk,&
460  & fac,vecnorm,eps
461 
462  parameter(eps=1.d-14)
463 
464  ! ... dtheta/dri, dtheta/drj and dtheta/drk
465 
466  sinth=sin(theta)
467  costh=cos(theta)
468 
469  do ic=1,3
470  ra(ic)=xyz(ic,j)-xyz(ic,i)
471  rb(ic)=xyz(ic,k)-xyz(ic,i)
472  end do
473 
474  ran=vecnorm(ra,3,0)
475  rbn=vecnorm(rb,3,0)
476  if (abs(sinth).le.eps) then
477  fac=sign(kijk,costh)/(ran*rbn)
478  else
479  fac=1.d0/(sinth*ran*rbn)
480  end if
481 
482  do ic=1,3
483  ! ... dedri
484  dei(ic)=fac*( ra(ic)+rb(ic)-(rbn*rbn*ra(ic)+ran*ran*rb(ic))*&
485  & costh/(ran*rbn))
486  ! ... dedrj
487  dej(ic)=fac*(ra(ic)*costh*rbn/ran-rb(ic))
488  ! ... dedrk
489  dek(ic)=fac*(rb(ic)*costh*ran/rbn-ra(ic))
490  end do
491 
492 End subroutine

498 Subroutine dphidr(nat,xyz,i,j,k,l,phi,&
499  & dphidri,dphidrj,dphidrk,dphidrl)
500  use xtb_mctc_accuracy, only : wp
501  ! the torsion derivatives
502 
503  implicit none
504 
505  external vecnorm
506 
507  integer ic,i,j,k,l,nat
508 
509  real(wp)&
510  & sinphi,cosphi,onenner,thab,thbc,&
511  & ra(3),rb(3),rc(3),rab(3),rac(3),rbc(3),rbb(3),&
512  & raa(3),rba(3),rapba(3),rapbb(3),rbpca(3),rbpcb(3),&
513  & rapb(3),rbpc(3),na(3),nb(3),nan,nbn,&
514  & dphidri(3),dphidrj(3),dphidrk(3),dphidrl(3),&
515  & xyz(3,nat),phi,vecnorm,nenner,eps,vz
516 
517  parameter(eps=1.d-14)
518 
519  cosphi=cos(phi)
520  sinphi=sin(phi)
521  do ic=1,3
522  ra(ic)=xyz(ic,j)-xyz(ic,i)
523  rb(ic)=xyz(ic,k)-xyz(ic,j)
524  rc(ic)=xyz(ic,l)-xyz(ic,k)
525 
526  rapb(ic)=ra(ic)+rb(ic)
527  rbpc(ic)=rb(ic)+rc(ic)
528  end do
529 
530  call crossprod(ra,rb,na)
531  call crossprod(rb,rc,nb)
532  nan=vecnorm(na,3,0)
533  nbn=vecnorm(nb,3,0)
534 
535  nenner=nan*nbn*sinphi
536  if (abs(nenner).lt.eps) then
537  dphidri=0
538  dphidrj=0
539  dphidrk=0
540  dphidrl=0
541  onenner=1.0d0/(nan*nbn)
542  else
543  onenner=1.d0/nenner
544  endif
545 
546  call crossprod(na,rb,rab)
547  call crossprod(nb,ra,rba)
548  call crossprod(na,rc,rac)
549  call crossprod(nb,rb,rbb)
550  call crossprod(nb,rc,rbc)
551  call crossprod(na,ra,raa)
552 
553  call crossprod(rapb,na,rapba)
554  call crossprod(rapb,nb,rapbb)
555  call crossprod(rbpc,na,rbpca)
556  call crossprod(rbpc,nb,rbpcb)
557 
558  ! ... dphidri
559  do ic=1,3
560  dphidri(ic)=onenner*(cosphi*nbn/nan*rab(ic)-rbb(ic))
561 
562  ! ... dphidrj
563  dphidrj(ic)=onenner*(cosphi*(nbn/nan*rapba(ic)&
564  & +nan/nbn*rbc(ic))&
565  & -(rac(ic)+rapbb(ic)))
566  ! ... dphidrk
567  dphidrk(ic)=onenner*(cosphi*(nbn/nan*raa(ic)&
568  & +nan/nbn*rbpcb(ic))&
569  & -(rba(ic)+rbpca(ic)))
570  ! ... dphidrl
571  dphidrl(ic)=onenner*(cosphi*nan/nbn*rbb(ic)-rab(ic))
572  end do
573 
574 End subroutine
