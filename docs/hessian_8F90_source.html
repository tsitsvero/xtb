169 
170  h = 0.0_wp
171  htb = 0.0_wp
172  hbias = 0.0_wp

174 !! ========================================================================
175 ! Hessian part -----------------------------------------------------------
176 
177  parallize = .true.
178  select type(calc)
179  type is (tgffcalculator)
180  calc%update = .false.
181  parallize = .false.
182  end select
183 
184  if(freezeset%n.gt.0) then
185  ! for frozfc of about 10 the frozen modes
186  ! approach 5000 cm-1, i.e., come too close to
187  ! the real ones
188  nonfrozh=mol%n-freezeset%n
189  do a = 1,mol%n
190  res%freq(a)=float(a)
191  enddo
192  do a=1,freezeset%n
193  res%freq(freezeset%atoms(a))=freezeset%atoms(a)*100000_wp
194  enddo
195  call sortind(mol%n,res%freq)
196  do a=1,nonfrozh
197  indx(a)=idint(res%freq(a))
198  enddo
199  do a=nonfrozh+1,mol%n
200  indx(a)=idint(res%freq(a)/100000_wp)
201  enddo
202  write(*,'(''atoms frozen in Hessian calc.:'',10i4)') &
203  & indx(nonfrozh+1:mol%n)
204 
205  h = 0.0_wp
206  dipd = 0.0_wp
207  pold = 0.0_wp
208 ! This OpenMP statements lead to invalid LLVM-IRwith NVHPC (20.7 to 20.11)
209 #ifndef __PGIC__
210  !$omp parallel if(parallize) default(shared)
211  call numdiff2(env, mol, chk0, calc, indx(:nonfrozh), step, h, dipd)
212  !$omp end parallel
213 #else
214  call numdiff2(env, mol, chk0, calc, indx(:nonfrozh), step, h, dipd)
215 #endif
216 
217  else
218 !! ------------------------------------------------------------------------
219 ! normal case
220 !! ------------------------------------------------------------------------
221  h = 0.0_wp
222  dipd = 0.0_wp
223  pold = 0.0_wp
224 ! This OpenMP statements lead to invalid LLVM-IR with NVHPC (20.7 to 20.11)
225 #ifndef __PGIC__
226  !$omp parallel if(parallize) default(shared)
227  call numdiff2(env, mol, chk0, calc, step, h, dipd)
228  !$omp end parallel
229 #else
230  call numdiff2(env, mol, chk0, calc, step, h, dipd)
231 #endif
232  endif
233 
234 ! Hessian done -----------------------------------------------------------
235 !! ========================================================================
236 
237  if (runtyp.eq.p_run_bhess) call numhess_rmsd(env,mol,hbias)
238 
239  if(freezeset%n.gt.0)then
240  ! inverse mass array
241  do a = 1, mol%n
242  ia = indx(a)
243  do ic = 1, 3
244  ii = (ia-1)*3+ic
245  isqm(ii)=1.0_wp/sqrt(atmass(ia))
246  amass(ii)=isqm(ii)/sqrt(amutoau)
247  enddo
248  enddo
249  do a = 1, nonfrozh
250  ia = indx(a)
251  do ic = 1, 3
252  ii = (ia-1)*3+ic
253  do b = 1, nonfrozh
254  ja = indx(b)
255  do jc = 1, 3
256  jj = (ja-1)*3+jc
257  if(abs(h(ii,jj)-h(jj,ii)).gt.1.d-2) then
258  write(errstr,'(a,1x,i0,1x,i0,1x,a,1x,es14.6,1x,es14.6)') &
259  & 'Hessian element ',i,j,' is not symmetric:',h(i,j),h(j,i)
260  call env%warning(trim(errstr), source)
261  endif
262  h(jj,ii) = 0.5_wp*(h(ii,jj)+h(jj,ii))
263  enddo
264  enddo
265  enddo
266  enddo
267  ! at this point the H matrix has zeros in the frozen atom block
268  do a = nonfrozh+1,mol%n
269  ia = indx(a)
270  do ic = 1, 3
271  ii = (ia-1)*3+ic
272  h(:, ii) = 0.0_wp
273  h(ii, :) = 0.0_wp
274  h(ii,ii)=freezeset%fc ! fill frozen diagonal block only
275  enddo
276  enddo
277  res%hess = h ! copy
278  else
279  ! symmetrize
280  do i=1,n3
281  do j=1,n3
282  res%hess(j,i)=(h(i,j)+h(j,i))*0.5_wp
283  if(abs(h(i,j)-h(j,i)).gt.1.d-2) then
284  write(errstr,'(a,1x,i0,1x,i0,1x,a,1x,es14.6,1x,es14.6)') &
285  & 'Hessian element ',i,j,' is not symmetric:',h(i,j),h(j,i)
286  call env%warning(trim(errstr), source)
287  endif
288  enddo
289  enddo
290  ! inverse mass array
291  do ia = 1, mol%n
292  do ic = 1, 3
293  ii = (ia-1)*3+ic
294  isqm(ii)=1.0_wp/sqrt(atmass(ia))
295  amass(ii)=isqm(ii)/sqrt(amutoau)
296  enddo
297  enddo
298  endif
299 
300  if (pr_dftbp_hessian_out) then
301  call writehessianout('hessian.out', res%hess)
302  write(env%unit, '(A)') "DFTB+ style hessian.out written"
303  end if
304 
305  ! prepare all for diag
306  ! copy
307  k=0
308  do i=1,n3
309  do j=1,i
310  k=k+1
311  hss(k)=res%hess(j,i)
312  enddo
313  enddo
314  ! same for bhess run
315  if (runtyp.eq.p_run_bhess) then
316  k=0
317  do i=1,n3
318  do j=1,i
319  k=k+1
320  hsb(k)=hbias(j,i)
321  enddo
322  enddo
323  end if
324  ! project
325  if(.not.res%linear)then ! projection does not work for linear mol.
326  if (runtyp.eq.p_run_bhess) then
327  call trproj(mol%n,n3,mol%xyz,hsb,.false.,0,res%freq,1) ! freq is dummy
328  end if
329  call trproj(mol%n,n3,mol%xyz,hss,.false.,0,res%freq,1) ! freq is dummy
330  endif
331  ! non mass weigthed Hessian in hss
332  hname = 'hessian'
333  write(env%unit,'(a)')
334  write(env%unit,'("writing file <",a,">.")') hname
335  call wrhess(n3,hss,hname)
336 
337  ! include masses
338  k=0
339  do i=1,n3
340  do j=1,i
341  k=k+1
342  res%hess(j,i)=hss(k)*isqm(i)*isqm(j)*scalh
343  res%hess(i,j)=res%hess(j,i)
344  enddo
345  enddo
346  ! same for bhess run
347  if (runtyp.eq.p_run_bhess) then
348  k=0
349  do i=1,n3
350  do j=1,i
351  k=k+1
352  hbias(j,i)=hsb(k)*isqm(i)*isqm(j)*scalh
353  hbias(i,j)=hbias(j,i)
354  enddo
355  enddo
356  end if
357  ! calcualte htb without RMSD bias
358  if (runtyp.eq.p_run_bhess) htb=res%hess-hbias
359  ! diag
360  lwork = 1 + 6*n3 + 2*n3**2
361  allocate(aux(lwork))
362  call dsyev ('V','U',n3,res%hess,n3,res%freq,aux,lwork,info)
363  if(info.ne.0) then
364  call env%error('Diagonalization of hessian failed', source)
365  return
366  end if
367 
368  ! calculate fc_tb and fc_bias
369  alp1=1.27_wp
370  alp2=1.5d-4
371  if (runtyp.eq.p_run_bhess) then
372  do j=1,n3
373  v(1:n3) = res%hess(1:n3,j) ! modes
374  call mctc_gemv(htb,v,fc_tmp)
375  fc_tb(j) = mctc_dot(v,fc_tmp)
376  call mctc_gemv(hbias,v,fc_tmp)
377  fc_bias(j) = mctc_dot(v,fc_tmp)
378  if (abs(res%freq(j)).gt.1.0d-6) then
379  freq_scal(j) = sqrt( (fc_tb(j)+alp2) / ( (fc_tb(j)+alp2) + alp1*fc_bias(j) ) )
380  if (fc_tb(j).lt.0.and.fc_bias(j).ne.0) then
381  freq_scal(j) = -sqrt( (abs(fc_tb(j))+alp2) / ( (abs(fc_tb(j))+alp2) + alp1*fc_bias(j) ) )
382  end if
383  else
384  freq_scal(j) = 1.0_wp
385  end if
386  end do
387  end if
388 
389  write(env%unit,'(a)')
390  if(res%linear)then
391  write(env%unit,'(1x,a)') ⁻¹'vibrational frequencies (cm)'
392  else
393  write(env%unit,'(1x,a)') ⁻¹'projected vibrational frequencies (cm)'
394  endif
395  k=0
396  do i=1,n3
397  ! Eigenvalues in atomic units, convert to wavenumbers
398  res%freq(i)=autorcm*sign(sqrt(abs(res%freq(i))),res%freq(i))/sqrt(amutoau)
399  if(abs(res%freq(i)).lt.0.01_wp) then
400  k=k+1
401  izero(k)=i
402  endif
403  enddo
404 
405  ! scale frequencies
406  if (runtyp.eq.p_run_bhess) then
407  do j=1,n3
408  res%freq(j)=freq_scal(j)*res%freq(j)
409  end do
410  end if
411 
412  if (verbose.and.runtyp.eq.p_run_bhess) then
413  write(env%unit,'(4x,"freq fc_tb fc_bias scal")')
414  do i=1,n3
415  write(env%unit,'(f8.2,2x,f9.6,2x,f9.6,2x,f7.4)') &
416  res%freq(i),fc_tb(i),fc_bias(i),freq_scal(i)
417  end do
418  write(env%unit,*)
419  end if
420 
421  ! sort such that rot/trans are modes 1:6, H/isqm are scratch
422  kend=6
423  if(res%linear)then
424  kend=5
425  do i=1,kend
426  izero(i)=i
427  enddo
428  res%freq(1:5)=0
429  endif
430  do k=1,kend
431  h(1:n3,k)=res%hess(1:n3,izero(k))
432  isqm( k)=res%freq(izero(k))
433  enddo
434  j=kend
435  do k=1,n3
436  if(abs(res%freq(k)).gt.0.01_wp)then
437  j=j+1
438  if(j.gt.n3) then
439  call env%error('internal error while sorting hessian', source)
440  return
441  end if
442  h(1:n3,j)=res%hess(1:n3,k)
443  isqm( j)=res%freq( k)
444  endif
445  enddo
446  res%hess = h
447  res%freq = isqm
448  call preigf(env%unit,res%freq,res%n3true)
449 
450  ! reduced mass
451  res%lowmode=1
452  k=0
453  do i=1,n3
454  if(res%freq(i).lt.mode_vthr) res%lowmode=i
455  xsum=0
456  k=k+1
457  do ia=1,mol%n
458  do ic=1,3
459  ii = (ia-1)*3+ic
460  xsum=xsum+atmass(ia)*res%hess(ii,i)**2
461  enddo
462  enddo
463  res%rmass(i)=xsum
464  enddo
465 
466  !--- IR intensity ---!
467  ! 1. res%hess corresponds to the orthonormal eigenvectors of the hessian
468  ! matrix (-> normal modes of vibration). Mass-weighting is introduced
469  ! back again via multiplying with amass(j).
470  !
471  ! 2. res%hess(j,i) is the matrix which transforms a derivative with
472  ! respect to the j-th cartesian coordinate ("dipd") into a derivative with
473  ! respect to the i-th (mass-weighted) normal coordinate.
474  !
475  ! 3. amass(j) = 1/sqrt(m(j)); m(j) is given in atomic units (a.u.).
476  !
477  ! 4. matmul(D x H) = U
478  !
479  ! 5. D = dipd(3,n3); H = res%hess(n3:n3); U = Matrix with dipol derivatives
480  ! in x, y and z direction per mode
481 
482  do i = 1, n3
483  do k = 1, 3
484  sum2 = 0.0_wp
485  do j = 1, n3
486  sum2 = sum2 + dipd(k,j)*(res%hess(j,i)*amass(j))
487  end do
488  trdip(k) = sum2
489  end do
490  res%dipt(i) = autokmmol*(trdip(1)**2+trdip(2)**2+trdip(3)**2)
491  end do
492  ! Raman intensity
493  do i = 1, n3
494  sum2 = 0.0_wp
495  do j = 1, n3
496  sum2 = sum2 + pold(j)*(res%hess(j,i)*amass(j))
497  end do
498  res%polt(i) = abs(sum2)
499  end do
500 
501 end subroutine numhess

503 subroutine numhess_rmsd( &
504  & env,mol,hbias)
505  use xtb_mctc_accuracy, only : wp
506  use xtb_mctc_convert
507 
508 !! ========================================================================
509 ! type definitions
510  use xtb_type_environment
511  use xtb_type_molecule
512  use xtb_type_restart
513  use xtb_type_calculator
514  use xtb_type_data
515 
516  use xtb_setparam

518  use xtb_fixparam
519  use xtb_metadynamic
520 
521  use prob_module ! prob_module code
522 
523  implicit none
525  type(tenvironment), intent(inout) :: env
526  type(tmolecule), intent(inout) :: mol
527  real(wp),intent(inout) :: hbias(mol%n*3,mol%n*3)
529  type(tmolecule) :: tmol
530  real(wp) :: ebias
531  real(wp) :: step,step2
532  integer :: n3,i,j,k,ic,jc,ia,ja,ii,jj,a,b
533  real(wp),allocatable :: gr(:,:)
534  real(wp),allocatable :: gl(:,:)
535  real(wp),allocatable :: xyzsave(:,:)
536 
537  n3=3*mol%n
538 
539  allocate(gr(3,mol%n),gl(3,mol%n),xyzsave(3,mol%n))
540 
541  xyzsave = mol%xyz
542 
543  ! step length
544  step=0.0001_wp
545  step=step_hess
546  step2=0.5_wp/step
547 
548 !! ========================================================================
549 ! RMSD part -----------------------------------------------------------
550 
551  do ia = 1, mol%n
552  do ic = 1, 3
553  ii = (ia-1)*3+ic
554 
555  tmol=mol
556  tmol%xyz(ic,ia)=xyzsave(ic,ia)+step
557 
558  gr = 0.0_wp
559  ebias = 0.0_wp
560 
561  if (prob_flag) then ! prob_module code
562  call prob_metadynamic(metaset,tmol%n,tmol%at,tmol%xyz,ebias,gr)
563  else
564  call metadynamic(metaset,tmol%n,tmol%at,tmol%xyz,ebias,gr)
565  end if
566 
567  ! call metadynamic(metaset,tmol%n,tmol%at,tmol%xyz,ebias,gr)
568 
569  tmol=mol
570  tmol%xyz(ic,ia)=xyzsave(ic,ia)-step
571 
572  gl = 0.0_wp
573  ebias = 0.0_wp
574 
575  if (prob_flag) then ! prob_module code
576  call prob_metadynamic(metaset,tmol%n,tmol%at,tmol%xyz,ebias,gl)
577  else
578  call metadynamic(metaset,tmol%n,tmol%at,tmol%xyz,ebias,gl)
579  end if
580 
581  ! call metadynamic(metaset,tmol%n,tmol%at,tmol%xyz,ebias,gl)
582 
583  tmol%xyz(ic,ia)=xyzsave(ic,ia)
584 
585  do ja= 1, mol%n
586  do jc = 1, 3
587  jj = (ja-1)*3 + jc
588  hbias(ii,jj) =(gr(jc,ja) - gl(jc,ja)) * step2
589  enddo
590  enddo
591 
592  call tmol%deallocate
593  enddo
594 
595  end do
596 
597 ! RMSD done -----------------------------------------------------------
598 !! ========================================================================
599 
600 end subroutine numhess_rmsd
