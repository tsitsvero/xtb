71  return
72  end if
73  end if
74  ii = il
75  ij = 2 * il
76  do while (ij <= ir)
77  if (ij < ir) then
78  if (array(indx(ij)) < array(indx(ij+1)) - tol) then
79  ij = ij + 1
80  end if
81  end if

83  indx(ii) = indx(ij)

88  end if
