20 subroutine lopt(init, n, no, accr, op, d)
21  !
22  ! subroutine for boys localization r.ahlrichs 2/85
23  ! to determine vectors |i>, i=1,n
24  ! which maximize
25  ! f = sum(i,j=1,n)sum(k=1,no) (<i|op(k)|i>-<j|op(k)|j>)**2
26  ! equivalent to maximization of
27  ! g = sum(i=1,n)sum(k=1,no) <i|op(k)|i>**2 + <j|op(k)|j>**2
28  ! equivalent to minimization of
29  ! h = sum(i<j)sum(k) <i|op(k)|j>**2
30  ! stationarity condition
31  ! sum(k) op(ij,k)*(op(ii,k)-op(jj,k)) = 0 all i.ne.j
32  !
33  use xtb_mctc_accuracy, only : wp
34  implicit real(wp)(a-h,o-z)
35 
36  logical init
37  integer :: auxi
38  real(wp) :: d(n,n)
39  real(wp) :: op(n*(n+1)/2,no)
40  real(wp) :: auxvec(n)
41  real(wp) :: accr
42  !DIR$ ASSUME_ALIGNED D: 64
43  !DIR$ ASSUME_ALIGNED OP: 64
44  !DIR$ ASSUME_ALIGNED auxvec: 64
45 
46 
47 
48 
49 
50 
51  !
52  ! input
53  ! n = dimension of lin. space
54  ! no = # of operators in f or g
55  ! op(ij,k) matrix representation of k'th operator
56  ! symmetric and stored triangular,nn1=n(n+1)/2
57  ! thi = rotation threshold (1.-6 to 1.-12 sngl or dbl prc)
58  ! nboys = max. nb. of sweeps
59  ! result
60  ! d contains the solution vectors as columns
61  ! op the transformed operators
62  !
63  a0=0.d0
64  a1=1.d0
65  a2=2.d0
66  a10=10.d0
67  aqu=0.25d0
68  ! accr=1.d-6 ! originally 10-14 but this setting saves a factor of 3 !
69  thi=1.d-10
70 
71  first=0
72 
73  th=dmax1(thi,accr)
74  !
75  ! set d to unit matrix
76  !
77  if(init)then
78  write(*,*) 'initialization of trafo matrix to unity'
79  do i=1,n
80  do j=1,n
81  10 d(j,i)=a0-accr*i+accr*i*j
82  end do
83  20 d(i,i)=a1
84  end do
85  endif
86 
87  if(n.eq.1) return
88 
89  nboys=20000
90  nn1=n*(n+1)/2
91 
92  !
93  ! opn = operator norm needed for cut off threshold
94  !
95  opn=a0
96  do k=1,no
97  do i=1,nn1
98  21 opn=opn+op(i,k)**2
99  end do
100  22 continue
101  end do
102  if(opn.eq.a0) return
103  opn=dsqrt(opn/dble(nn1*no))*accr
104 
105  ths=aqu
106  it=0
107  do isw=1,nboys
108  !
109  ! loop over sweeps
110  !
111  thsw=a0
112  ii=1
113  ij=2
114  do i=2,n
115  !
116  ii=ii+i
117  jj=0
118  im1=i-1
119  ip1=i+1
120  li=ii-i
121  do j=1,im1
122  !
123  ! loop over i,j rotations
124  ! maximize sum(k) <i|op(k)|i>**2 + <j|op(k)|j>**2
125  ! like in jacobi
126  !
127  jj=jj+j
128  !
129  ! get ax , b determining the tan of the rotation angle
130  !
131  ax=a0
132  b=a0
133  do k=1,no
134  aux1=op(jj,k)-op(ii,k)
135  aux2=op(ij,k)
136  ax=ax+aux1*aux2
137  30 b=b+aqu*aux1**2-aux2**2
138  end do
139  !
140  ! rotation angle x0
141  !
142  if (dabs(ax)+dabs(b).lt.opn) go to 100
143  x0=aqu*datan2(ax,b)
144  thsw=dmax1(thsw,dabs(x0))
145  if (dabs(x0).lt.ths) go to 100
146  !
147  ! rotation
148  !
149  it=it+1
150  ! c=dcos(x0)
151  s=dsin(x0)
152  s2=s*s
153  c2=a1-s2
154  c=dsqrt(c2)
155  do l=1,n
156  aux1=d(l,j)
157  d(l,j)=c*d(l,j)+s*d(l,i)
158  d(l,i)=c*d(l,i)-s*aux1
159  40 continue
160  end do
161  jm1=j-1
162  ! c2=c*c
163  ! s2=s*s
164  cs=c*s
165  c2ms2=c2-s2
166  lj=jj-j
167  jp1=j+1
168 
169  ! !$OMP PARALLEL PRIVATE (k,l,aux1,aux2,il,jl) &
170  ! !$OMP& SHARED(op) &
171  ! !$OMP& DEFAULT(SHARED)
172  ! !$OMP DO
173  do k=1,no
174  aux1=op(ii,k)
175  aux2=op(jj,k)
176  op(ii,k)=aux1*c2+aux2*s2-a2*op(ij,k)*cs
177  op(jj,k)=aux1*s2+aux2*c2+a2*op(ij,k)*cs
178  op(ij,k)=cs*(aux1-aux2)+op(ij,k)*c2ms2
179  if(jm1.le.0) go to 60
180 
181  ! do l=1,jm1
182  ! aux1=c*op(li+l,k)-s*op(lj+l,k)
183  ! op(lj+l,k)=s*op(li+l,k)+c*op(lj+l,k)
184  ! op(li+l,k)=aux1
185  ! end do
186  auxvec(1:jm1) = c*op(li+1:li+jm1,k)-s*op(lj+1:lj+jm1,k)
187  op(lj+1:lj+jm1,k) = op(lj+1:lj+jm1,k) * c
188  op(lj+1:lj+jm1,k) = op(lj+1:lj+jm1,k) + s*op(li+1:li+jm1,k)
189  op(li+1:li+jm1,k) = auxvec(1:jm1)
190 
191 
192 
193  60 if (im1.eq.j) go to 80
194  jl=jj+j
195 
196  do l=jp1,im1
197  aux1=c*op(li+l,k)-s*op(jl,k)
198  op(jl,k)=c*op(jl,k)+s*op(li+l,k)
199  op(li+l,k)=aux1
200  70 jl=jl+l
201  end do
202 
203 
204  80 if (i.eq.n) go to 95
205  jl=ii+j
206  il=ii+i
207 
208  do l=ip1,n
209  aux1=c*op(jl,k)+s*op(il,k)
210  op(il,k)=c*op(il,k)-s*op(jl,k)
211  op(jl,k)=aux1
212  il=il+l
213  90 jl=jl+l
214  end do
215 
216  95 continue
217  end do
218  ! !$OMP END DO
219  ! !$OMP END PARALLEL
220 
221  100 ij=ij+1
222  end do
223  110 ij=ij+1
224  end do
225  !
226  ! check convergence
227  !
228  if (thsw.le.th) go to 210
229  ths=dmin1(thsw**2,ths*aqu)
230  ths=dmax1(th,ths)
231  if(mod(isw,50).eq.0)&
232  &write(*,'(''iteration'',i7,2x,''convergence'',d16.8)')isw,thsw
233  200 continue
234  end do
235  write (*,300) isw,thsw
236  return
237  210 write (*,310) isw,thsw
238 
239  return
240  300 format (/' not converged in',i7,' iterations, threshold :',d16.8)
241  310 format (/' converged in',i7,' iterations, threshold : ',d16.8)
242 end subroutine
