101  "orca input file :",ext_orca%input_file,&
102  "orca input present :",bool2string(exist),&
103  "orca input override :",bool2string(.not.ext_orca%exist)
104  endif
105  ! sanity check
106  if (ext_orca%exist) then
107  call open_file(iorca,ext_orca%input_file,'r')

128  ext_orca%input_string = trim(adjustl(line(idx_spce:)))
129  endif
130  enddo
131  call close_file(iorca)
132  if (.not.(chk_engrad.and.chk_xyzfile)) then
133  call env%error("Please add '! ENGRAD' and/or '* xyzfile' to '"//&
134  & ext_orca%input_file //"'!",source)
135  return
136  endif
137 
138  else
139  ! check for the input line

142  ext_orca%input_string = ext_orca%input_string(2:)
143  endif
144  else

147  endif
148  endif
149  if (verbose) then
150  write(stdout,'(a,1x,a)') &
151  !$ "orca parallel :",bool2string(omp_get_num_threads() > 1),&
152  !$ "orca number of threads :",omp_get_num_threads(),&
153  "orca input line :",ext_orca%input_string
154  endif
