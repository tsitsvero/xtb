317 subroutine generic_header(iunit,string,width,offset)
318 implicit none
319 integer,intent(in) :: iunit
320 integer,intent(in) :: offset
321 integer,intent(in) :: width
322 character(len=*),intent(in) :: string
323 character(len=width) :: dum1,dum2
324 character(len=2*width) :: outstring
325 character(len=width) :: formatstr
326 integer :: strlen,ifront,iback
327 strlen = len(string)
328 ifront = (width - strlen)/2
329 iback = width - ifront - strlen
330 write(dum1,*) width
331 write(dum2,*) offset
332 write(formatstr,'(i0,"x,a,",i0,"x")') ifront,iback
333 write(outstring,'("|",'//formatstr//',"|")') string
334 write(iunit,'('//dum2//'x,1x,'//dum1//'("-"),1x)')
335 write(iunit,'('//dum2//'x,a)') trim(outstring)
336 write(iunit,'('//dum2//'x,1x,'//dum1//'("-"),1x)')
337 end subroutine generic_header
