31 !! ========================================================================
32 ! signal processing
33  external :: wSIGTERM
34  external :: wSIGINT
35 ! here two important signal handlers are installed, it seems that
36 ! FORTRAN by itself cannot handle signals in the way I expected it
37 ! to do, but this will force it to die when I hit CTRL^C.
38 ! UPDATE: Since the FORTRAN standard does not specify which signals
39 ! are handled by the language, some compilers may handle no
40 ! signals at all. This is *non-standard* so it will break
41 ! your program on some platforms with some compilers!
42  call signal(2,wsigint)
43  call signal(15,wsigterm)
44 
45 ! initialize the timing system
46  call start_timing_run
47  call init_timing(ntimer,verb=verbose) ! verbosity allows printing of cputime
48  call start_timing(1)
49 
50 ! initialize the messagebuffer for the error handler
51  call init_errorbuffer
52 
53 ! initialize the filelist
54  call init_filelist(20)
55 
56 ! set this for xtb_mctc_global
57  name = progname
58 
59  allocate(persistentenv)
60  call init(persistentenv)
61 
62 #ifdef USE_CUSOLVER
63  err = cusolverdncreate(cusolverdnh)
64  if (err /= 0) then
65  call persistentenv%error("failed to create cusolver handle", "mctc_init")
66  end if
67 #endif
68 end subroutine mctc_init
