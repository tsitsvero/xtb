386 subroutine readparamfile(self, env)

389  character(len=*), parameter :: source = 'solv_model_readParamFile'
390 
392  type(tenvironment), intent(inout) :: env
393 
395  type(tsolvmodel), intent(inout) :: self
396 
397  type(gbsa_parameter) :: param
398  integer :: unit, stat, statc, ii
399 
400  statc = 0
401  call open_file(unit, self%paramFile, 'r')
402  if (unit == -1) then
403  call env%error("Could not open parameter file '"//self%paramFile//"'", &
404  & source)
405  return
406  end if
407 
408  read(unit,*,iostat=stat) param%epsv
409  if (stat /= 0) statc = statc + 1
410  read(unit,*,iostat=stat) param%smass
411  if (stat /= 0) statc = statc + 1
412  read(unit,*,iostat=stat) param%rhos
413  if (stat /= 0) statc = statc + 1
414  read(unit,*,iostat=stat) param%c1
415  if (stat /= 0) statc = statc + 1
416  read(unit,*,iostat=stat) param%rprobe
417  if (stat /= 0) statc = statc + 1
418  read(unit,*,iostat=stat) param%gshift
419  if (stat /= 0) statc = statc + 1
420  read(unit,*,iostat=stat) param%soset
421  if (stat /= 0) statc = statc + 1
422  read(unit,*,iostat=stat) param%alpha
423  if (stat /= 0) statc = statc + 1
424  do ii = 1, 94
425  read(unit,*,iostat=stat) param%gamscale(ii), param%sx(ii), param%tmp(ii)
426  if (stat /= 0) statc = statc + 1
427  enddo
428 
429  if (statc /= 0) then
430  call env%error("Could not read parameters from '"//self%paramFile//"'", &
431  & source)
432  return
433  end if
434 
435  call paramtomodel(self, param)
436 
437 end subroutine readparamfile

440 subroutine paramtomodel(self, param)
441 
443  type(tsolvmodel), intent(inout) :: self
444 
445  type(gbsa_parameter), intent(in) :: param
446 
447  integer :: ii
448 
449  self%dielectricConst = param%epsv
450  self%molarMass = param%smass
451  self%density = param%rhos/param%smass * molcm3toau
452  self%freeEnergyShift = param%gshift * kcaltoau
453  self%bornScale = param%c1
454  self%bornOffset = param%soset * 0.1_wp*aatoau
455  self%probeRad = param%rprobe * aatoau
456 
457  self%vdwRad = vanderwaalsradd3
458  self%surfaceTension = param%gamscale * fourpi*surfacetension
459  self%descreening = param%sx
460  allocate(self%hBondStrength(size(param%tmp)))
461  if (any(abs(param%tmp).gt.1.e-3_wp)) then
462  do ii = 1, size(param%tmp)
463  self%hBondStrength(ii) = -(param%tmp(ii)**2) * kcaltoau
464  end do
465  else
466  self%hBondStrength(:) = 0.0_wp
467  end if
468 
469 end subroutine paramtomodel

498  case(1)

507

509  ! try "param_<model>_<solvent>.txt" next

511  call rdpath(env%xtbpath, fname, pfile)
512  end if
513 
514  ! finally try ".param_<model>_<solvent>" which is the old default
515  if (.not.allocated(pfile)) then
516  select case(level)
517  case default
518  fname = '.param_' // model // '_' // solvent
519  case(2)
520  fname = '.param_' // model // '2_' // solvent
521  case(0)
522  fname = '.param_' // model // '0_' // solvent
523  end select
524 
525  call rdpath(env%xtbpath, fname, pfile)
526  end if
527 
528 end subroutine getparamfile

532 subroutine info(self, unit)
533 
535  class(tsolvmodel), intent(in) :: self
536 
538  integer, intent(in) :: unit
539 
540  write(unit, '(6x, "*", 1x, a, ":", t40)', advance='no') "Solvation model"
541  if (self%alpb) then
542  write(unit, '(a)') "ALPB"
543  else
544  write(unit, '(a)') "GBSA"
545  end if
546 
547  write(unit, '(8x, a, t40, a)') "Solvent", self%solvent
548  write(unit, '(8x, a, t40, a)') "Parameter file", self%paramFile
549  write(unit, '(8x, a, t40, es14.4)') &
550  & "Dielectric constant", self%dielectricConst
551 
552  write(unit, '(8x, a, t40)', advance='no') "Reference state"
553  select case(self%state)
554  case default
555  write(unit, '(a)') 'gsolv [1 M gas/solution]'
556  case(solutionstate%reference)
557  write(unit, '(a)') 'gsolv=reference [X=1]'
558  case(solutionstate%mol1bar)
559  write(unit, '(a)') 'gsolv [1 bar gas/1 M solution]'
560  end select
561 
562  write(unit, '(8x, a, t40, es14.4, 1x, a, t60, es14.4, 1x, a)') &
563  & "Free energy shift", self%freeEnergyShift, "Eh", &
564  & self%freeEnergyShift * autokcal, "kcal/mol"
565  write(unit, '(8x, a, t40, es14.4, 1x, a)') &
566  & "Temperature", self%temperature, "K"
567  write(unit, '(8x, a, t40, es14.4, 1x, a)') &
568  & "Density", self%density / (molcm3toau/self%molarMass), "kg/L"
569  write(unit, '(8x, a, t40, es14.4, 1x, a)') &
570  & "Solvent mass", self%molarMass, "g/mol"
571 
572  write(unit, '(8x, a, t40)', advance='no') "Interaction kernel"
573  select case(self%kernel)
574  case default
575  write(unit, '(i0, 1x, a)') self%kernel, '(internal error)'
576  case(gbkernel%still)
577  write(unit, '(a)') 'Still'
578  case(gbkernel%p16)
579  write(unit, '(a)') 'P16'
580  end select
581 
582  write(unit, '(8x, a, t40, es14.4)') &
583  "Born radius scaling (c1)", self%bornScale
584  write(unit, '(8x, a, t40, a)') "Born radii integrator", "GBOBC"
585  write(unit, '(8x, a, t40, es14.4, 1x, a, t60, es14.4, 1x, a)') &
586  "Born offset", self%bornOffset, "a0", self%bornOffset/autoaa, "AA"
587 
588  write(unit, '(8x, a, t40)', advance='no') "H-bond correction"
589  if (any(self%hBondStrength < 0.0_wp)) then
590  write(unit, '(a)') "true"
591  else
592  write(unit, '(a)') "false"
593  end if
594 
595  write(unit, '(8x, a, t40)', advance='no') "Ion screening"
596  if (self%ionStrength > 0.0_wp) then
597  write(unit, '(a)') "true"
598  else
599  write(unit, '(a)') "false"
600  end if
601 
602  if (allocated(self%surfaceTension)) then
603  write(unit, '(8x, a, t40, es14.4, 1x, a, t60, es14.4, 1x, a)') &
604  "Surface tension", surfacetension, "Eh", surfacetension*automnm, "dyn/cm"
605  end if
606 
607  write(unit, '(8x, a, t40, i14, 1x, a)') &
608  "Grid points", self%nAng, "per atom"
609 
610 end subroutine info
