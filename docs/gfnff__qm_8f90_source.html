18 ! solve QM Hamiltonian A in overlap basis S (if ovlp=.true., ZDO otherwise)
19 ! and return density matrix in A (and energy weighted density in S if ovlp=.true.)
20 ! ndim is the dimension of the problem for nel electrons with nopen more alpha than beta
21 
22 subroutine gfnffqmsolve(pr,A,S,ovlp,et,ndim,nopen,nel,eel,focc,e)
23  use xtb_mctc_accuracy, only : wp
24  use xtb_scc_core, only : dmat, fermismear, occu
25  implicit none
26  integer ndim ! # basis
27  integer nopen ! # of open shells
28  integer nel ! # of electrons
29  logical ovlp ! in overlap basis?
30  logical pr !
31  real(wp) et ! electronic temp Fermi smear
32  real(wp) eel ! electronic energy = sum_occ nocc*eps
33  real(wp) focc(ndim)! occupations
34  real(wp) e(ndim) ! eigenvalues
35  real(wp) A(ndim,ndim)
36  real(wp) S(ndim,ndim)
37 
38  integer ihomoa,ihomob,i,liwork,info,lwork
39  real(wp) ga,gb,efa,efb,nfoda,nfodb
40  real(wp),allocatable:: X(:,:)
41  real(wp),allocatable:: focca(:),foccb(:),aux(:)
42 ! real*4 ,allocatable:: aux4(:),e4(:),A4(:,:)
43  integer,allocatable :: iwork(:),ifail(:)
44 
45 
46  allocate(focca(ndim),foccb(ndim))
47 
48 ! ZDO case
49  if(.not.ovlp) then
50 
51  lwork = 1 + 6*ndim + 2*ndim**2
52 ! allocate(aux4(lwork),e4(ndim),A4(ndim,ndim))
53 ! A4 = A
54 ! call ssyev ('V','U',ndim,A4,ndim,e4,aux4,lwork,info)
55 ! A = A4
56 ! e = e4
57 ! deallocate(aux4,A4,e4)
58  allocate(aux(lwork))
59  call dsyev ('V','U',ndim,a,ndim,e,aux,lwork,info)
60 
61  else
62 ! with overlap case
63 
64  allocate (aux(1),iwork(1),ifail(ndim))
65  call dsygvd(1,'V','U',ndim,a,ndim,s,ndim,e,aux, & !workspace query
66  & -1,iwork,liwork,info)
67  lwork=int(aux(1))
68  liwork=iwork(1)
69  deallocate(aux,iwork)
70 
71  allocate (aux(lwork),iwork(liwork)) !do it
72  call dsygvd(1,'V','U',ndim,a,ndim,s,ndim,e,aux, &
73  & lwork,iwork,liwork,info)
74 
75  endif
76 ! DONE
77 
78 ! scale energy so that gaps a roughly ok (in eV)
79 
80  e=e*0.1*27.2113957
81 
82  if(info.ne.0) then
83  write(*,*) 'INFO',info
84  stop 'diag error'
85  endif
86 
87  if(et.gt.1.d-3) then
88 ! Fermi smearing, convert restricted occ first to alpha/beta
89  call occu(ndim,nel,nopen,ihomoa,ihomob,focca,foccb)
90  if(ihomoa.le.ndim.and.ihomoa.gt.0) then
91  call fermismear(.false.,ndim,ihomoa,et,e,focca,nfoda,efa,ga)
92  else
93  focca=0
94  endif
95  if(ihomob.le.ndim.and.ihomob.gt.0) then
96  call fermismear(.false.,ndim,ihomob,et,e,foccb,nfodb,efb,gb)
97  else
98  foccb=0
99  endif
100  focc = focca + foccb
101  if(ihomoa+1.le.ndim)then
102  if(abs(focc(ihomoa)-focc(ihomoa+1)).lt.1.d-4) then ! a perfect birad is anit-aromatic
103  focc = 0 ! and hence we break the sym
104  do i=1,nel/2
105  focc(i)=2.0d0
106  enddo
107  write(*,*) 'perfect biradical detected at FT-HMO level. Breaking the symmetry'
108  write(*,*) 'because its assumed to be an anit-aromatic system like COT or CB.'
109  endif
110  endif
111  else
112  focc = 0
113  do i=1,nel/2
114  focc(i)=2.0d0
115  enddo
116  if(2*(nel/2).ne.nel) focc(nel/2+1)=1.0
117  endif
118 
119  focca = focc * e
120  eel = sum(focca)
121 ! density matrix
122  s = a
123  call dmat(ndim,focc,s,a)
124 
125  if(ovlp) then
126  allocate(x(ndim,ndim))
127 ! energy weighted density on S
128  call dmat(ndim,focca,s,x)
129  s = x
130  endif
131 
132  deallocate(focca,foccb)
133 
134 end subroutine gfnffqmsolve
