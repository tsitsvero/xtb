45  precessor = 0
46  distance = huge(distance)
47  distance(start) = 0
48  visited = .false.
49  do while (all(visited) == .false.) !as long there are unvisited nodes
50  current = minloc(distance,1, .not. visited)
51 
52  !Abort if Fragments are not connected:
53  if (distance(current) == huge(distance)) then
54  shortest_distance = huge(distance)
55  return
56  end if ! distance(current) == huge(distance)
57 
58  visited(current) = .true.
59  if (current == goal) then
60  shortest_distance = distance(goal)
61  !route to target found
62  do while (precessor(current) /= 0)
63  current = precessor(current)
64  end do ! End loop: while precessor(current) /= 0
65  exit
66  else
67  !loop over all neighbours of current atom
68  do i_neighbours = 1, neighbours(20,current)
69  neighbour = neighbours(i_neighbours,current)
70  if (visited(neighbour) == .false.) then
71  !distanzupdate
72  alt_dist = distance(current) + input_distances(lin(current,neighbour))
73  if ((alt_dist < distance(neighbour))) then
74  distance(neighbour) = alt_dist
75  precessor(neighbour) = current
76  end if ! (alt_dist < distance(neighbour))
77  endif !(visited(neighbour))
78  end do ! End Loop over i_neighbours from 1 to all_Atoms(current)%n_neighbours
79  endif
80  end do ! End loop: while sum(visited) /= 0
81  !initialize
82  shortest_distance = distance(goal)
83 end function shortest_distance
