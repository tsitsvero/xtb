610 pure subroutine drep_grad(repData,mol,trans,erep,gradient,sigma)
611  use xtb_type_molecule
612  use xtb_type_param
613  use xtb_pbc_tools
614  use xtb_pbc
615  implicit none

618  real(wp), intent(in) :: trans(:, :)
619  real(wp), intent(inout) :: gradient(:, :)
620  real(wp), intent(inout) :: sigma(:, :)

623  integer :: iat,jat,ati,atj
624  real(wp) :: ri(3),dr(3)
625  real(wp) :: rij(3)

627  real(wp) :: den2,den4
628  real(wp) :: alpha
629  real(wp) :: repab
630  real(wp) :: expterm
631  real(wp) :: dtmp

633  real(wp) :: w,t(3)
634  integer :: latrep(3),tx,ty,tz,itr
635  call get_realspace_cutoff(mol%lattice,rthr,latrep)
636  w = 1.0_wp

638  erep = 0.0_wp

640  ati=mol%at(i)
641  do j = 1, i
642  do itr = 1, size(trans, dim=2)
643  t = trans(:, itr)

645  r2 = sum(rij**2)
646  if(r2.gt.rthr .or. r2.lt.1.0e-6_wp) cycle
647  atj=mol%at(j)
648  r = sqrt(r2)
649  den2 = (repdata%electronegativity(ati) - repdata%electronegativity(atj))**2
650  den4 = den2**2
651  alpha=sqrt(repdata%alpha(ati)*repdata%alpha(atj))&
652  *(1.0_wp+(0.01_wp*den2+0.01_wp*den4)*repdata%enScale)
653  repab = repdata%zeff(ati)*repdata%zeff(atj)
654  r2top34 = r2**0.75_wp
655  r2top34top2 = r2top34**2
656  expterm = exp(-alpha*r2top34)*repab
657  ! save repulsion energy
658  erep = erep + expterm/r * w
659  ! save repulsion gradient
660  dtmp = expterm*(1.5_wp*alpha*r2top34 + 1)/r2top34top2 * w
661  gradient(:,i) = gradient(:,i) - dtmp*rij
662  gradient(:,j) = gradient(:,j) + dtmp*rij
663  sigma = sigma - dtmp*outer_prod_3x3(rij,rij)
664  enddo ! k WSC partner
665  enddo ! j atom
666  enddo ! i atom

670 ! short-ranged bond correction
671 subroutine dsrb_grad(mol,srb,cn,dcndr,dcndL,trans,esrb,gradient,sigma)
672 
673  use xtb_type_param
674  use xtb_type_molecule
675 
676  use xtb_approxrab
677  use xtb_disp_ncoord
678  use xtb_pbc_tools
679  use xtb_pbc
680 
681  implicit none
682 
683  type(tmolecule), intent(in) :: mol
684  type(tshortrangedata), intent(in) :: srb
685  real(wp), intent(in) :: trans(:, :)
686  real(wp), intent(inout) :: gradient(:, :)
687  real(wp), intent(inout) :: sigma(:, :)
688  real(wp), intent(out) :: esrb
689  real(wp), intent(in) :: cn(:)
690  real(wp), intent(in) :: dcndr(:,:,:)
691  real(wp), intent(in) :: dcndL(:,:,:)
692 
693  integer :: i,j,k,wsAt
694  integer :: iat,jat,ati,atj
695  integer :: nsrb,itr
696  real(wp) :: den
697  real(wp) :: expterm
698  ! distances
699  real(wp) :: rij(3)
700  real(wp) :: r2,r,dr,rab
701  real(wp) :: dtmp,pre
702  real(wp) :: w
703  ! allocatables
704  real(wp), allocatable :: dEdr0(:)
705  real(wp), allocatable :: drab0dr(:,:,:)
706  real(wp), allocatable :: drab0dL(:,:,:)
707  real(wp), allocatable :: rab0(:)
708  integer, allocatable :: srblist(:,:)
709 
710  ! initialize
711  esrb = 0.0_wp
712 
713  w = 1.0_wp
714 
715  call build_srblist(mol,nsrb,srblist)
716  if (nsrb.eq.0) return
717 
718  ! get memory
719  allocate( dedr0(nsrb), source = 0.0_wp )
720  allocate( drab0dr(3,mol%n,nsrb), source = 0.0_wp )
721  allocate( drab0dl(3,3,nsrb), source = 0.0_wp )
722  allocate( rab0(nsrb), source = 0.0_wp )
723  ! get approximated distances rab and gradients
724  call pbc_approx_rab(mol%n,mol%at,mol%xyz,cn,dcndr,dcndl,nsrb,srblist,srb%shift, &
725  & rab0,drab0dr,drab0dl)
726  do i = 1, nsrb
727  iat = srblist(1,i)
728  jat = srblist(2,i)
729  ati = mol%at(iat)
730  atj = mol%at(jat)
731  den = en(ati) - en(atj)
732  pre = srb%steepness*(1.0_wp + srb%enScale*den**2)
733  do itr = 1, size(trans, dim=2)
734  rij = mol%xyz(:,iat) - (mol%xyz(:,jat) + trans(:, itr))
735  rab = norm2(rij)
736  dr = rab - rab0(i)
737  expterm = srb%prefactor*exp(-pre*dr**2)
738  ! save SRB energy
739  esrb = esrb + expterm * w
740  dtmp = 2.0_wp*pre*dr*expterm * w
741  gradient(:,iat) = gradient(:,iat) - dtmp*rij/rab
742  gradient(:,jat) = gradient(:,jat) + dtmp*rij/rab
743  ! three body gradient
744  dedr0(i) = dedr0(i) + dtmp
745  sigma = sigma - dtmp*spread(rij, 1, 3)*spread(rij, 2, 3)/rab
746  enddo ! rep
747  enddo ! i
748 
749  call contract(drab0dr, dedr0, gradient, beta=1.0_wp)
750  call contract(drab0dl, dedr0, sigma, beta=1.0_wp)
751 
752 end subroutine dsrb_grad

754 pure subroutine build_srblist(mol,nsrb,srblist)
755  use xtb_type_molecule
756  implicit none
757  type(tmolecule), intent(in) :: mol
758  integer, intent(out) :: nsrb
759  integer, allocatable, intent(out) :: srblist(:,:)
760  integer :: i,j,k
761  real(wp) :: r2
762  ! cutoff
763  real(wp), parameter :: srb_cut = 200.0_wp
764  nsrb = 0
765  do i = 1, mol%n-1
766  if (srbatom(mol%at(i))) cycle ! i i case
767  do j = i+1, mol%n
768  if (srbatom(mol%at(j))) cycle ! j j case
769  if (mol%at(i).eq.mol%at(j)) cycle ! hetero only
770  r2 = mol%dist(j,i)**2 ! get minimum image distance
771  ! increment array size according to SRB cutoff criterium
772  ! default: srb_cut = 200.0_wp
773  if(r2.lt.srb_cut) nsrb=nsrb+1
774  enddo ! j
775  enddo ! i
776  ! nsrb = 0: no pairs for SRB correction, nothing to do here
777  if(nsrb.eq.0) return
778  ! get memory
779  allocate( srblist(2,nsrb), source = 0 )
780  ! setup srblist
781  k=0
782  do i = 1, mol%n-1
783  if (srbatom(mol%at(i))) cycle ! i i case
784  do j = i+1, mol%n
785  if (srbatom(mol%at(j))) cycle ! j j case
786  if (mol%at(i).eq.mol%at(j)) cycle ! hetero only
787  r2 = mol%dist(j,i)**2 ! get minimum image distance
788  if(r2.gt.srb_cut) cycle
789  k = k + 1
790  srblist(1,k) = i
791  srblist(2,k) = j
792  enddo ! j
793  enddo ! i
794  nsrb = k

809 subroutine ccm_build_sh0(nShell, hData, selfEnergy, nat, at, basis, nbf, nao, &
810  & xyz, lattice, intcut, sint, h0, wsc)
811 
812  use xtb_type_basisset
813  use xtb_type_wsc
814 
815  use xtb_lin, only : lin
816  use xtb_intgrad
817  use xtb_scc_core

821  integer, intent(in) :: nShell(:)
822  type(thamiltoniandata), intent(in) :: hData
823  type(tb_wsc), intent(in) :: wsc
824  type(tbasisset), intent(in) :: basis
825  real(wp), intent(in) :: selfEnergy(:, :)
826 
827  integer, intent(in) :: nat
828  integer, intent(in) :: nbf
829  integer, intent(in) :: nao
830  integer, intent(in) :: at(nat)
831  real(wp),intent(in) :: xyz(3,nat)
832  real(wp),intent(in) :: lattice(3,3)
833  real(wp),intent(in) :: intcut
834 
835  real(wp),intent(out) :: sint(nao,nao) ! overlap matrix S
836  real(wp),intent(out) :: h0(nao*(1+nao)/2)
837 
838  integer, parameter :: llao (0:3) = (/ 1, 3, 6,10/)
839  integer, parameter :: llao2(0:3) = (/ 1, 3, 5, 7/)
840 
841  integer i,j,k,l,m,ii,jj,ll,mm,kk,ki,kj,kl,mi,mj,ij
842  real(wp) tmp1,tmp2,tmp3,tmp4,step,step2
843  real(wp) dx,dy,dz,s00r,s00l,s00,alpj
844  real(wp) skj,r1,r2,tt,t1,t2,t3,t4,f,ci,cc,cj,alpi,rab2,ab,est
845 
846  real(wp) :: den,den2,den4,enpoly
847  real(wp) :: zi,zj,zetaij
848  real(wp) :: hii,hjj,hav
849  real(wp) :: shpoly,km
850  logical :: valaoi,valaoj
851 
852  real(wp) ri(3),rj(3),t(3),f1,f2
853  real(wp),parameter ::point(3) = 0.0_wp
854  real(wp) dtmp(3),qtmp(6),ss(6,6),dd(3,6,6),qq(6,6,6),tmp(6,6)
855  real(wp) sstmp(6,6) !TESTST
856  integer ip,jp,iat,jat,ati,atj,ish,jsh,icao,jcao,iao,jao,jshmax,il,jl
857  integer ishtyp,jshtyp,iptyp,jptyp,naoi,naoj,mli,mlj,iprim,jprim
858  integer itt(0:3)
859 
860  integer kat
861 
862  parameter(itt =(/0,1,4,10/))
863  real(wp) :: saw(10)
864 
865  sint = 0.0_wp

868  !$omp parallel default(none) &
869  !$omp private(iat,jat,ij,ati,cc,ci,rab2,atj,ish,ishtyp,valaoi,valaoj, &
870  !$omp& ri,rj,icao,naoi,iptyp,jsh,jshmax,jshtyp,jcao,naoj,jptyp, &
871  !$omp& ss,saw,est,alpi,alpj,ab,iprim,jprim,ip,jp,km,shpoly, &
872  !$omp& mli,mlj,tmp,zi,zj,zetaij,enpoly,iao,jao, &
873  !$omp& ii,jj,k,den,den2,den4,i,j,il,jl,hii,hjj,hav,t) &
874  !$omp reduction (+:sint,h0) &
875  !$omp shared(wsc,basis,at,nShell,hData,xyz,lattice,intcut,nat,selfEnergy)
876  !$omp do schedule(runtime)
877  do iat = 1, nat
878  ri = xyz(:,iat)
879  ati = at(iat)
880  do jat = 1, iat-1
881  atj = at(jat)
882 
883  den = hdata%electronegativity(ati) - hdata%electronegativity(atj)
884  den2 = den**2
885  den4 = den2**2
886 
887  ishells: do ish = 1, nshell(ati)
888  ishtyp = hdata%angShell(ish,ati)
889  icao = basis%caoshell(ish,iat)
890  naoi = llao(ishtyp)
891  iptyp = itt(ishtyp)
892  jshmax = nshell(atj)
893  if (iat == jat) jshmax = ish
894 
895  jshells: do jsh = 1, jshmax
896  jshtyp = hdata%angShell(jsh,atj)
897  jcao = basis%caoshell(jsh,jat)
898  naoj = llao(jshtyp)
899  jptyp = itt(jshtyp)
900 
901  il = ishtyp+1
902  jl = jshtyp+1
903  ! diagonals are the same for all H0 elements
904  hii = selfenergy(ish, iat)
905  hjj = selfenergy(jsh, jat)
906 
907  ! evaluate the EN polynom for this shells
908  enpoly = (1.0_wp + hdata%enScale(jl-1,il-1)*den2 &
909  & + hdata%enScale4*hdata%enScale(jl-1,il-1)*den4)
910 
911  ! we scale the two shells depending on their exponent
912  zi = hdata%slaterExponent(ish, ati)
913  zj = hdata%slaterExponent(jsh, atj)
914  zetaij = 2 * sqrt(zi*zj)/(zi+zj)
915 
916  ! now do the real magic (called EHT enhancement factor)
917  km = hdata%kScale(jl-1,il-1) * hdata%pairParam(ati,atj) * zetaij * enpoly
918 
919  ! check for valence orbitals
920  valaoi = hdata%valenceShell(ish, ati).eq.0
921  valaoj = hdata%valenceShell(jsh, atj).eq.0
922  ! and scale appropiately
923  if (valaoi) then
924  if (valaoj) then
925  km = 0.0_wp
926  else
927  km = km * hdata%kDiff
928  endif
929  else
930  if (valaoj) km = km * hdata%kDiff
931  endif
932 
933  ! averaged H0 element (without overlap contribution!)
934  hav = 0.5_wp * km * (hii + hjj)
935 
936  wscat: do kat = 1, wsc%itbl(jat,iat)
937  ss = 0.0_wp
938  t = wsc%lattr(:,kat,jat,iat)
939  rj = xyz(:,jat) + matmul(lattice,t)
940 
941  ! distance dependent polynomial
942  shpoly = shellpoly(hdata%shellPoly(il,ati),hdata%shellPoly(jl,atj),&
943  & hdata%atomicRad(ati),hdata%atomicRad(atj),ri,rj)
944 
945  ! get overlap integral
946  call get_overlap(icao,jcao,naoi,naoj,iptyp,jptyp,ri,rj,point, &
947  & intcut,basis%nprim,basis%primcount, &
948  & basis%alp,basis%cont,ss)
949 
950  !transform from CAO to SAO
951  call dtrf2(ss,ishtyp,jshtyp)
952 
953  do ii = 1, llao2(ishtyp)
954  iao = ii+basis%saoshell(ish,iat)
955  do jj = 1, llao2(jshtyp)
956  jao = jj+basis%saoshell(jsh,jat)
957  if (jao > iao) cycle
958  ij = lin(iao,jao)
959  ! add all WSC images
960  sint(iao,jao) = sint(iao,jao) + ss(jj,ii)*wsc%w(jat,iat)
961  sint(jao,iao) = sint(jao,iao) + ss(jj,ii)*wsc%w(jat,iat)
962  ! Hamiltonian
963  h0(ij) = h0(ij) + hav * shpoly * ss(jj,ii) * wsc%w(jat,iat)
964  enddo
965  enddo
966 
967  enddo wscat
968  enddo jshells
969  enddo ishells
970  enddo
971  enddo
972  !$OMP END DO
973  !$OMP END PARALLEL
974 
975  ! diagonal elements
976  do iat = 1, nat
977  ati = at(iat)
978  do ish = 1, nshell(ati)
979  iao = 1+basis%saoshell(ish,iat)
980  ishtyp = hdata%angShell(ish,ati)
981  il = ishtyp + 1
982  do iao = 1, llao2(ishtyp)
983  i = iao+basis%saoshell(ish,iat)
984  sint(i,i)=1.0_wp+sint(i,i)
985 
986  ii = i*(1+i)/2 ! H0 is packed, note i*(i-1)/2+i = i*(1+i)/2
987 
988  ! calculate environment dependent shift
989  hii = selfenergy(ish, iat)
990  h0(ii) = hii
991  end do
992  end do
993  end do
994 
995 end subroutine ccm_build_sh0
