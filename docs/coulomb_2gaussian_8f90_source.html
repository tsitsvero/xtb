331 pure function getrterm(vec, gam, rTrans, alpha, scale) result(rTerm)
332  real(wp),intent(in) :: vec(3)
333  real(wp),intent(in) :: gam
334  real(wp),intent(in) :: rtrans(:,:)
335  real(wp),intent(in) :: alpha
336  real(wp),intent(in) :: scale
337  real(wp) :: rterm
338  real(wp),parameter :: eps = 1.0e-9_wp
339  integer :: itr
340  real(wp) :: r1, rij(3)
341  rterm = 0.0_wp
342  do itr = 1, size(rtrans, dim=2)
343  rij = vec + rtrans(:, itr)
344  r1 = sqrt(sum(rij**2))
345  ! self-interaction correction
346  if(r1 < eps) then
347  rterm = rterm - 2.0_wp*alpha/sqrtpi
348  else
349  rterm = rterm + erf(gam*r1)/r1 - erf(alpha*r1)/r1
350  end if
351  end do
352  rterm = rterm * scale
353 end function getrterm

555 pure subroutine getrderiv(vec, gij, rTrans, alpha, scale, dG, dS)
556  real(wp), intent(in) :: vec(:)
557  real(wp), intent(in) :: gij
558  real(wp), intent(in) :: rtrans(:,:)
559  real(wp), intent(in) :: alpha
560  real(wp), intent(in) :: scale
561  real(wp), intent(out) :: dg(:)
562  real(wp), intent(out) :: ds(:,:)
563  real(wp), parameter :: eps = 1.0e-9_wp
564  integer :: itr
565  real(wp) :: r1, rij(3), arg, dd
566  real(wp), parameter :: unity(3, 3) = reshape(&
567  & [1.0_wp, 0.0_wp, 0.0_wp, 0.0_wp, 1.0_wp, 0.0_wp, 0.0_wp, 0.0_wp, 1.0_wp], &
568  & [3, 3])
569  dg = 0.0_wp
570  ds = 0.0_wp
571  do itr = 1, size(rtrans, dim=2)
572  ! real contributions
573  rij = vec + rtrans(:, itr)
574  r1 = sqrt(sum(rij**2))
575  if (r1 < eps) cycle
576  arg = alpha**2*r1**2
577  dd = + 2*gij*exp(-gij**2*r1**2)/(sqrtpi*r1**2) - erf(gij*r1)/(r1**3) &
578  & - 2*alpha*exp(-arg)/(sqrtpi*r1**2) + erf(alpha*r1)/(r1**3)
579  dg = dg + rij*dd
580  ds = ds + 0.5_wp * dd*spread(rij, 1, 3)*spread(rij, 2, 3)
581  enddo
582  dg = dg * scale
583  ds = ds * scale
584 
585 end subroutine getrderiv
