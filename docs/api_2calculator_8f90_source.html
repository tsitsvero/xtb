334 subroutine setsolvent_api(venv, vcalc, charptr, state, temperature, grid) &
335  & bind(C, name="xtb_setSolvent")
336  character(len=*), parameter :: source = 'xtb_api_setSolvent'
337  type(c_ptr), value :: venv
338  type(venvironment), pointer :: env
339  type(c_ptr), value :: vcalc
340  type(vcalculator), pointer :: calc
341  character(kind=c_char), intent(in) :: charptr(*)
342  integer(c_int), intent(in), optional :: state
343  real(c_double), intent(in), optional :: temperature
344  integer(c_int), intent(in), optional :: grid
345  character(len=:), allocatable :: solvent
346  type(tsolvinput) :: input
347  integer :: gsolvstate, nang
348  real(wp) :: temp
349  logical :: exitrun
350 
351  if (c_associated(venv)) then
352  call c_f_pointer(venv, env)
353  call checkglobalenv
354 
355  if (.not.c_associated(vcalc)) then
356  call env%ptr%error("Singlepoint calculator is not allocated", source)
357  return
358  end if
359  call c_f_pointer(vcalc, calc)
360 
361  if (.not.allocated(calc%ptr)) then
362  call env%ptr%error("No calculator loaded to add solvation model", source)
363  return
364  end if
365 
366  if (present(state)) then
367  gsolvstate = state
368  else
369  gsolvstate = solutionstate%gsolv
370  end if
371 
372  if (present(temperature)) then
373  temp = temperature
374  else
375  temp = 298.15_wp
376  end if
377 
378  if (present(grid)) then
379  nang = grid
380  else
381  nang = 230
382  end if
383 
384  call c_f_character(charptr, solvent)
385 
386  ! PGI 20.5 cannot use default constructor with deferred-length characters:
387  ! input = TSolvInput(solvent=solvent, temperature=temp, state=gsolvstate, &
388  ! & nang=nang)
389  input%solvent = solvent
390  input%temperature = temp
391  input%state = gsolvstate
392  input%nang = nang
393  input%alpb = .false.
394  input%kernel = gbkernel%still
395  call addsolvationmodel(env%ptr, calc%ptr, input)
396 
397  call env%ptr%check(exitrun)
398  if (exitrun) then
399  call env%ptr%error("Could not add solvation model for '"//solvent//"'", &
400  & source)
401  return
402  end if
403 
404  end if

441 subroutine setexternalcharges_api(venv, vcalc, npc, numbers, charges, positions) &
442  & bind(C, name="xtb_setExternalCharges")
443  character(len=*), parameter :: source = 'xtb_api_setExternalCharges'
444  type(c_ptr), value :: venv
445  type(venvironment), pointer :: env
446  type(c_ptr), value :: vcalc
447  type(vcalculator), pointer :: calc
448  integer(c_int), intent(in) :: npc
449  integer(c_int), intent(in) :: numbers(*)
450  real(c_double), intent(in) :: charges(*)
451  real(c_double), intent(in) :: positions(3, *)
452  integer :: ii
453 
454  if (c_associated(venv)) then
455  call c_f_pointer(venv, env)
456  call checkglobalenv
457 
458  if (npc <= 0) then
459  call env%ptr%error("Negative number of point charges provided", source)
460  return
461  end if
462 
463  if (.not.c_associated(vcalc)) then
464  call env%ptr%error("Singlepoint calculator is not allocated", source)
465  return
466  end if
467  call c_f_pointer(vcalc, calc)
468 
469  if (.not.allocated(calc%ptr)) then
470  call env%ptr%error("No calculator loaded to add external potential", &

472  return
473  end if
474 
475  select type(xtb => calc%ptr)
476  class default
477  call env%ptr%error("Calculator does not support external potentials", &

479  return

481  type is(txtbcalculator)

483  call env%ptr%error("GFN0-xTB does not support external potentials", &

485  return
486  end if
487 
488  call xtb%pcem%allocate(npc)
489  do ii = 1, npc
490  xtb%pcem%xyz(:, ii) = positions(:, ii)
491  xtb%pcem%gam(ii) = xtb%xtbData%coulomb%chemicalHardness(numbers(ii))
492  xtb%pcem%q(ii) = charges(ii)
493  end do

497  end if
