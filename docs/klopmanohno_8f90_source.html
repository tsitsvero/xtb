411 pure function getrterm(vec, gam, gExp, rTrans, alpha, scale) result(rTerm)
412  real(wp),intent(in) :: vec(3)
413  real(wp),intent(in) :: gam
414  real(wp),intent(in) :: gexp
415  real(wp),intent(in) :: rtrans(:,:)
416  real(wp),intent(in) :: alpha
417  real(wp),intent(in) :: scale
418  real(wp) :: rterm
419  real(wp),parameter :: eps = 1.0e-9_wp
420  integer :: itr
421  real(wp) :: r1, rij(3)
422  rterm = 0.0_wp
423  do itr = 1, size(rtrans, dim=2)
424  rij = vec + rtrans(:, itr)
425  r1 = sqrt(sum(rij**2))
426  ! self-interaction correction
427  if(r1 < eps) then
428  rterm = rterm - 2.0_wp*alpha/sqrtpi
429  else
430  rterm = rterm + 1.0_wp/(r1**gexp + gam**(-gexp))**(1.0_wp/gexp) &
431  & - erf(alpha*r1)/r1
432  end if
433  end do
434  rterm = rterm * scale
435 end function getrterm

509  !$omp shared(mol, itbl, qvec, gExp, hardness) &
510  !$omp private(iat, jat, ish, jsh, ii, jj, iid, jid, r1, g1, gij, vec, dG, dS)

512  ii = itbl(1, iat)

521  gij = gamaverage(hardness(ish, iid), hardness(jsh, jid))
522  g1 = 1.0_wp / (r1**gexp + gij**(-gexp))
523  dg(:) = -vec*r1**(gexp-2.0_wp) * g1 * g1**(1.0_wp/gexp)

527  djdtr(:, jj+jsh) = djdtr(:, jj+jsh) + dg*qvec(ii+ish)
528  djdtr(:, ii+ish) = djdtr(:, ii+ish) - dg*qvec(jj+jsh)
529  djdl(:, :, jj+jsh) = djdl(:, :, jj+jsh) + ds*qvec(ii+ish)
