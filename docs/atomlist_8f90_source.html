293 pure subroutine string_assign_atomlist(string, self)
294  character(len=:), allocatable, intent(out) :: string
295  class(tatomlist), intent(in) :: self
296  character(len=10) :: buffer
297  integer :: i, last
298  logical :: state, first
299  last = -1
300  first = .true.

302  do i = 1, size(self)

367 pure subroutine atomlist_parse_string(self, string)
368  class(tatomlist), intent(inout) :: self
369  character(len=*),intent(in) :: string
370  integer :: pos, item, begin, last, err
371 
372  pos = index(string,self%skip)
373  if (pos.eq.0) then
374  read(string,*,iostat=err) item
375  if (err.ne.0) then
376  self%error = .false.
377  return
378  endif
379  call self%add(item)
380  else
381  read(string(:pos-1),*,iostat=err) begin
382  if (err.ne.0) then
383  self%error = .false.
384  return
385  endif
386  read(string(pos+1:),*,iostat=err) last
387  if (err.ne.0) then
388  self%error = .false.
389  return
390  endif
391  if (last.lt.begin) then
392  self%error = .false.
393  return
394  endif
395  do item = begin, last
396  call self%add(item)
397  enddo
398  endif
399

406  integer :: length, current_length
407  current_length = size(self)
408  if (current_length > 0) then
409  if (present(n)) then
