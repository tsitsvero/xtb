38 subroutine cube(n,nmo,nbf,xyz,at,cmo,eval,occ,fname,basis)
39  use xtb_type_basisset
40  use xtb_setparam
41  use xtb_mctc_blas, only : blas_gemm
42  implicit none
43  type(tbasisset), intent(in) :: basis
44 
45  real*8, intent ( in ) :: xyz(3,n)
46  real*8, intent ( in ) :: eval(nmo)
47  real*8, intent ( in ) :: occ(nmo)
48  real*8, intent ( in ) :: cmo(nbf,nmo)
49  integer, intent( in ) :: at(n)
50  integer, intent( in ) :: n,nmo,nbf
51  character*(*) fname
52 
53  real*8 ,allocatable ::ptmp(:,:)
54  real*8 ,allocatable ::p(:,:)
55  real*8 ,allocatable ::c(:,:)
56  real*4 ,allocatable ::cb(:,:,:)

69  call timing(t0,w0)
70  write(*,*)
71  write(*,*)'cube file module (SG, 7/16)'
72  thr = cube_pthr ! Dmat pre-screen
73  step= cube_step ! grid step (Bohr)
74  intcut=8.00d0 ! primitive cut
75  intcut2=2.0d0*intcut
76 
77  nexp=100*int(intcut2) ! size of exp array ie arguments 0:nexp
78  allocate(array(0:nexp))
79  do i=0,nexp
80  dum=float(i)/100.0
81  array(i)=exp(-dum)
82  enddo
83 
84  ! do i=0,160
85  ! dum=float(i)/10.0
86  ! write(*,*) int(10*dum)
87  ! write(*,*) exp(-dum),array(int(10*dum)),fastexp(nexp,array,dum)
88  ! enddo
89  ! stop
90 
91  allocate(ptmp(nbf,nbf),p(nbf,nbf),c(nbf,nbf),matlist(2,nbf*(nbf+1)/2))
92  c(1:nbf,1:nmo) = cmo(1:nbf,1:nmo)
93  do m=1,nmo
94  do i=1,nbf
95  ptmp(i,m)=cmo(i,m)*occ(m)
96  enddo
97  enddo
98  if(nmo.ne.nbf)then
99  ptmp(1:nbf,nmo+1:nbf)=0
100  c(1:nbf,nmo+1:nbf)=0
101  endif
102  call blas_gemm('N','T',nbf,nbf,nbf, 1.0d0,c,nbf,ptmp,nbf,0.0d0,p,nbf)
103  deallocate(ptmp,c)
104 
105  nm=0
106  npri=0
107  do i=1,nbf
108  primstart(i)=npri ! start index for prims of AO i
109  npri=npri+basis%nprim(i)
110  do j=1,i
111  if(abs(p(j,i)).gt.thr)then
112  nm=nm+1
113  matlist(1,nm)=i
114  matlist(2,nm)=j
115  if(i.ne.j) p(j,i)=2.0d0*p(j,i)
116  endif
117  enddo
118  enddo
119  write(*,'('' cube_pthr : '',f7.3)')cube_pthr
120  write(*,'('' cube_step : '',f7.3)')cube_step
121  write(*,'('' non-zero P (%): '',f7.3,'' nmat:'',i8)') &
122  & 100.*float(nm)/float(nbf*(nbf+1)/2),nm
123 
124  px=maxval(xyz(1,1:n))+3.0
125  py=maxval(xyz(2,1:n))+3.0
126  pz=maxval(xyz(3,1:n))+3.0
127  nx=minval(xyz(1,1:n))-3.0
128  ny=minval(xyz(2,1:n))-3.0
129  nz=minval(xyz(3,1:n))-3.0
130  write(*,*)'Grid Boundaries (x y z) :'
131  write(*,*)px,py,pz
132  write(*,*)nx,ny,nz
133 
134  ! calculate step size and number of steps (step size approx 0.2-0.5)
135  xst=floor((abs(px)+abs(nx))/step)
136  xinc=(abs(px)+abs(nx))/xst
137  yst=floor((abs(py)+abs(ny))/step)
138  yinc=(abs(py)+abs(ny))/yst
139  zst=floor((abs(pz)+abs(nz))/step)
140  zinc=(abs(pz)+abs(nz))/zst
141  write(*,*)'Total # of points', (xst+1)*(yst+1)*(zst+1)
142  dr3=xinc*yinc*zinc
143 
144  write(*,*)'writing ',trim(fname)
145  call open_file(ifile,fname,'w')
146  write(ifile,*)'xtb spin/fo density'
147  write(ifile,*)'By S.G.'
148  write(ifile,101)n,nx,ny,nz
149  write(ifile,101)xst+1,xinc,0.0,0.0
150  write(ifile,101)yst+1,0.0,yinc,0.0
151  write(ifile,101)zst+1,0.0,0.0,zinc
152  do i=1,n
153  write(ifile,102)at(i),0.0,xyz(1,i),xyz(2,i),xyz(3,i)
154  enddo
155 
156  allocate(cb(0:zst,0:yst,0:xst))
157 
158  nfod=0
159  cb =0
160 
161  ! Dmat loop -----------------------------------
162  do m=1,nm
163  ii=matlist(1,m)
164  jj=matlist(2,m)
165  iat=basis%aoat(ii)
166  jat=basis%aoat(jj)
167  xyza(1:3)=xyz(1:3,iat)
168  xyzb(1:3)=xyz(1:3,jat)
169  npri=basis%nprim(ii)
170  nprj=basis%nprim(jj)
171  iprimcount=primstart(ii)
172  jprimcount=primstart(jj)
173  rab=(xyza(1)-xyzb(1))**2 &
174  & +(xyza(2)-xyzb(2))**2 &
175  & +(xyza(3)-xyzb(3))**2
176  ! prim loop
177  do iiii=1,npri
178  iii=iprimcount+iiii
179  ccc=p(jj,ii)*basis%cont(iii)
180  dum=rab*basis%alp(iii)
181  do jjjj=1,nprj
182  jjj=jprimcount+jjjj
183  est=dum*basis%alp(jjj)/(basis%alp(iii)+basis%alp(jjj))
184  if(est.lt.intcut)then
185  cc=ccc*basis%cont(jjj)
186  ! grid loops
187  gridp(1)=nx
188  do i=0,xst
189  gridp(1)=nx+(xinc*i)
190  dx1=xyza(1)-gridp(1)
191  dx2=xyzb(1)-gridp(1)
192  dxx1=dx1*dx1
193  dxx2=dx2*dx2
194  gridp(2)=ny
195  do j=0,yst
196  gridp(2)=ny+(yinc*j)
197  dy1=xyza(2)-gridp(2)
198  dy2=xyzb(2)-gridp(2)
199  dyy1=dy1*dy1
200  dyy2=dy2*dy2
201  gridp(3)=nz
202  r1xy=dxx1+dyy1
203  r2xy=dxx2+dyy2
204  do k=0,zst
205  gridp(3)=nz+(zinc*k)
206  dz1=xyza(3)-gridp(3)
207  dz2=xyzb(3)-gridp(3)
208  dzz1=dz1*dz1
209  dzz2=dz2*dz2
210  r1=r1xy+dzz1
211  ! primitive function value
212  ar1=basis%alp(iii)*r1
213  if(ar1.lt.intcut2)then ! exp(-16) < 1.d-7 i.e. zero
214  call primvalf(dx1,dy1,dz1,dxx1,dyy1,dzz1,ar1, &
215  & basis%lao(ii),nexp,array,f1)
216  r2=r2xy+dzz2
217  ar2=basis%alp(jjj)*r2
218  if(ar2.lt.intcut2)then
219  call primvalf(dx2,dy2,dz2,dxx2,dyy2,dzz2,ar2, &
220  & basis%lao(jj),nexp,array,f2)
221  cb(k,j,i)=cb(k,j,i)+cc*f1*f2
222  endif
223  endif
224  enddo
225  enddo
226  enddo
227  endif
228  enddo
229  enddo
230  enddo
231  ! Dmat loop end -----------------------------------
232 
233  ! write
234  cou=1
235  do i=0,xst
236  do j=0,yst
237  do k=0,zst
238  if (cou.lt.6) then
239  write(ifile,'(E14.8,1X)',advance='no')cb(k,j,i)
240  cou=cou+1
241  else
242  write(ifile,'(E14.8)')cb(k,j,i)
243  cou=1
244  endif
245  nfod=nfod+cb(k,j,i)*dr3
246  enddo
247  enddo
248  enddo
249  call close_file(ifile)
250 
251  101 format(i5,3f16.6)
252  102 format(i5,4f16.6)
253 
254  write(*,'('' N (numint) : '',f6.3)')nfod
255  call timing(t1,w1)
256  call prtime(6,t1-t0,w1-w0,'cube')
257 end subroutine cube
