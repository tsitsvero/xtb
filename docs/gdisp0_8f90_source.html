52  dnorm = 0.0_wp
53  do iref = 1, number_of_references(ati)
54  gw = weight_cn(wf, cn(iat), reference_cn(iref, ati))

62

67  gwk = 1.0_wp
68  else
69  gwk = 0.0_wp
70  endif
71  endif

239 subroutine d3_gradient(dispm, nat, at, xyz, npair, pairlist, zeta_scale, radii, &
240  & r4r2, weighting_factor, dispscale, cn, dcndr, energy, gradient)

272 
273  max_ref = maxval(dispm%nref(at))
274  allocate(gw(max_ref, nat), dgwdcn(max_ref, nat), c6(nat, nat), &
275  & dc6dcn(nat, nat), energies(nat), dedcn(nat), source=0.0_wp)
276 
277  call weight_references_d4(dispm, nat, at, weighting_factor, cn, gw, dgwdcn)
278 
279  !gw = gw*zeta_scale
280  !dgwdcn = dgwdcn*zeta_scale
281  call get_atomic_c6_d4(dispm, nat, at, gw, dgwdcn, c6, dc6dcn)
282 
283  !$omp parallel do default(none) schedule(runtime) &
284  !$omp reduction(+:energies, gradient, sigma, dEdcn) &
285  !$omp shared(at, xyz, npair, pairlist, zeta_scale, dispscale, radii, r4r2, c6, &
286  !$omp& dc6dcn) &
287  !$omp private(ij, img, iat, jat, ati, atj, r2, rij, r4r2ij, r0, t6, t8, t10, &
288  !$omp& d6, d8, d10, disp, ddisp, dE, dG, dS)
289  do img = 1, npair
290  iat = pairlist(1, img)
291  jat = pairlist(2, img)
292  ij = jat + iat*(iat-1)/2
293  ati = at(iat)
294  atj = at(jat)
295  rij = xyz(:, iat) - xyz(:, jat)
296  r2 = sum(rij**2)
297 
298  r4r2ij = 3*r4r2(ati)*r4r2(atj)
299  r0 = radii(lin(ati, atj))
300 
301  t6 = 1._wp/(r2**3+r0**3)
302  t8 = 1._wp/(r2**4+r0**4)
303 
304  d6 = -6*r2**2*t6**2
305  d8 = -8*r2**3*t8**2
306 
307  disp = (t6 + 2*r4r2ij*t8) * zeta_scale(ij)*dispscale
308  ddisp= (d6 + 2*r4r2ij*d8) * zeta_scale(ij)*dispscale
309 
310  de = -c6(iat, jat)*disp * 0.5_wp
311  dg = -c6(iat, jat)*ddisp*rij
312  ds = spread(dg, 1, 3) * spread(rij, 2, 3) * 0.5_wp
313 
314  energies(iat) = energies(iat) + de
315  dedcn(iat) = dedcn(iat) - dc6dcn(iat, jat) * disp
316  sigma = sigma + ds
317  if (iat /= jat) then
318  energies(jat) = energies(jat) + de
319  dedcn(jat) = dedcn(jat) - dc6dcn(jat, iat) * disp
320  gradient(:, iat) = gradient(:, iat) + dg
321  gradient(:, jat) = gradient(:, jat) - dg
322  sigma = sigma + ds
323  endif
324 
325  enddo
326  !$omp end parallel do
327 
328  call dgemv('n', 3*nat, nat, 1.0_wp, dcndr, 3*nat, dedcn, 1, 1.0_wp, gradient, 1)
329  !call dgemv('n', 9, nat, 1.0_wp, dcndL, 9, dEdcn, 1, 1.0_wp, sigma, 1)
